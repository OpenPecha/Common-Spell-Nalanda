༄། །བསྟན་བཅོས་བཞི་བརྒྱ་པ་ཞེས་བྱ་བའི་ཚིག་ལེའུར་བྱས་པ་བཞུགས་སོ། །༄༅༅། །རྒྱ་གར་སྐད་དུ། ཙ་ཏུཿཤ་ཏ་ཀ་ཤཱསྟྲཱ་ཀཱ་རི་ཀཱ་ནཱ་མ། བོད་སྐད་དུ། བསྟན་བཅོས་བཞི་བརྒྱ་པ་ཞེས་བྱ་བའི་ཚིག་ལེའུར་བྱས་པ། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །གང་ལ་འཇིག་རྟེན་གསུམ་མངའ་བདག །རང་ཉིད་འཆི་བདག་བྱེད་པོ་མེད། །ཡོད་དེས་རྣལ་བཞིན་གཉིད་ལོག་ན། །དེ་ལས་མ་རུངས་གཞན་ཅི་ཡོད། །འཆི་བར་བྱ་ཕྱིར་སྐྱེས་གྱུར་ཅིང་། །གཞན་དབང་འགྲོ་བའི་ངང་ཅན་ལ། །འཆི་བ་བྱ་བ་ལྟར་སྣང་སྟེ། །གསོན་པ་བྱ་བ་མིན་ལྟར་རོ། །ཁྱོད་ཀྱིས་འདས་དུས་ཐུང་ངུ་ལྟར། །མཐོང་ཞིང་མ་འོངས་གཞན་དུ་ན། །ཁྱོད་མཉམ་མི་མཉམ་ལྟར་སེམས་ཏེ། །གསལ་བར་འཇིགས་པའི་འོ་དོད་འདྲ། །གཞན་དག་ལ་ཡང་སྤྱི་མཐུན་ཕྱིར། །ཁྱོད་ལ་འཆི་བའི་འཇིགས་མེད་ན། །གཅིག་ལ་གནོད་གང་དེ་ཕྲག་དོག །སྒོ་ནས་སྡུག་བསྔལ་བསྐྱེད་འགྱུར་རམ། །ནད་བཅོས་རྒ་བ་གསོ་བྱ་དེས། །དེ་དག་ལ་ཁྱོད་མི་འཇིགས་མོད། །ཆད་པ་ཕྱི་མ་བཅོར་མེད་པས། །ཁྱོད་ཆེས་གསལ་བར་དེ་ལ་འཇིགས། །གསད་པར་བྱ་བའི་ཕྱུགས་རྣམས་ལྟར། །འཆི་བ་ཀུན་གྱི་ཐུན་མོང་ཞིང་། །གཞན་ཡང་འཆི་བར་སྣང་འགྱུར་ན། །ཁྱོད་ལ་འཆི་བདག་འཇིགས་ཅིས་མེད། །དུས་ལ་ངེས་པ་མེད་པའི་ཕྱིར། །བདག་རྟག་སྙམ་དུ་སེམས་ནས་ནི། །ནམ་ཞིག་དུས་ན་འཆི་བདག་གིས། །ཁྱོད་ལ་གནོད་པ་བྱེད་པར་འགྱུར། །མ་འོངས་པ་ཡི་དོན་ལྟ་ཞིང་། །གསོན་པ་ཟད་པ་མ་ཡིན་ན། །བདག་ཉིད་འཚོང་ལ་བདག་ཉིད་ནི། །བྱམས་ལྡན་མཁས་པ་སུ་ཞིག་སྨྲ། །བདག་ཉིད་གཏའ་མར་བཞག་བྱས་ཏེ། །སྡིག་ལས་ཅི་ཡི་ཕྱིར་ན་བྱེད། །ངེས་པར་ཁྱོད་ནི་མཁས་རྣམས་ལྟར། །བདག་ལ་འདོད་ཆགས་བྲལ་བར་ཟད། །སུ་ལའང་གསོན་པ་ཞེས་བྱ་བ། །སེམས་ཀྱི་སྐད་ཅིག་ལས་གཞན་མེད། །སྐྱེ་བོས་དེ་ངེས་མི་རིག་པ། །དེས་ན་བདག་ཉིད་ཤེས་པ་དཀོན། །ཁྱོད་རིང་གསོན་པ་ལ་དགའ་ཞིང་། །རྒན་ཉིད་ལ་ཁྱོད་མི་དགའ་ན། །ཨེ་མ་ཁྱོད་སྤྱོད་སྐྱེ་བོ་ནི། །འདྲ་ལ་དམ་པ་ལྟ་བུར་སྣང་འགྱུར། །ཁྱོད་རང་འཆི་ལ་མྱ་ངན་ན། །བུ་སོགས་རྣམས་ཕྱིར་ཅི་སྟེ་ཡིན། །རང་ཉིད་སྤྱོ་བྱ་སྨྲ་པོ་ནི། །ཅི་ལྟ་བུར་ན་སྨད་མི་འགྱུར། །གང་ཚེ་འགའ་ཞིག་མ་བཅོལ་བར། །རང་ཉིད་བུར་གྱུར་དེ་ཡི་ཚེ། །
མ་དྲིས་ཉིད་དུ་དེ་འགྲོ་བ། །རིག་པར་མི་འགྱུར་མ་ཡིན་ནོ། །གཏི་མུག་ཕྱིར་ན་བུ་ཡི་ནི། །རྣམ་པ་རྟོག་པ་མེད་པར་ཟད། །འདི་ཡི་འགྲོ་བར་སྤྲོ་བ་ཉིད། །རྒ་བས་སྟོན་པར་བྱེད་པར་འགྱུར། །ཇི་ལྟར་ཕ་ལ་བུ་སྡུག་པ། །དེ་ལྟར་དེ་ལ་སྐྱེ་མ་ཡིན། །འཇིག་རྟེན་འདི་ནི་འོག་ཏུ་འགྲོ། །དེ་ཕྱིར་མཐོ་རིས་ཀྱང་རྙེད་དཀའ། །གང་ཚེ་རྗེས་སུ་མི་མཐུན་པ། །འགའ་ཡང་སྡུག་ཅེས་བྱ་ཡོད་མིན། །དེ་ཚེ་ཆགས་པ་ཚོང་དང་ནི། །འདྲ་བ་འབའ་ཞིག་སྐྱེ་བར་ཟད། །བྲལ་བས་བསྐྱེད་པའི་སྡུག་བསྔལ་ནི། །མི་རྣམས་སྙིང་ལས་མྱུར་ལྡོག་སྟེ། །ཆགས་ལའང་སྡུག་བསྔལ་བྲལ་བ་ཡིས། །མཚོན་པའི་མི་བརྟན་པ་ཉིད་ལྟོས། །གནོད་པས་གཟིར་བར་བྱས་ཟིན་ཀྱང་། །ཡོན་ཏན་མེད་སྙམ་ཤེས་བཞིན་དུ། །ཁྱོད་བདག་ཉིད་ལ་ཕྱི་ཆོས་འགྱུར། །དེ་ཡང་བདག་ལ་མི་རིགས་སོ། །སྡུག་བསྔལ་རྣམ་པར་འགེད་ལྟ་བུ། །འཇིག་རྟེན་འདི་ནི་ཡོངས་སུ་འཁྱམ། །སྡུག་བསྔལ་གྱུར་པའི་སྐྱེ་བོ་ལ། །སྡུག་བསྔལ་འགེད་པས་ཅི་ཞིག་བྱ། །གང་གི་འདུ་བ་དགའ་ཡིན་དེའི། །འབྲལ་བ་ཅི་ལྟར་དགའ་མ་ཡིན། །འདུ་བ་དང་ནི་འབྲལ་བ་དག །ལྷན་ཅིག་རྒྱུ་བར་སྣང་མིན་ནམ། །འདས་ལ་ཐོག་མ་ཡོད་མིན་ལ། །མ་འོངས་ཐ་མ་ཡོད་མིན་ན། །ཅིས་ན་ཁྱོད་ཀྱིས་འདུ་མཐོང་ཞིང་། །རིང་པོར་གྱུར་ཀྱང་འབྲལ་བ་མིན། །སྐད་ཅིག་ལ་སོགས་དུས་དེ་རྣམས། །ངེས་པར་དགྲ་དང་འདྲར་འགྱུར་ཏེ། །དེས་ན་དགྲར་གྱུར་དེ་དག་ལ། །རྣམ་ཀུན་ཁྱོད་འདོད་ཆགས་མི་བྱ། །བློ་ངན་བྲལ་བའི་འཇིགས་པ་ཡིས། །ཁྱིམ་ནས་ངེས་པར་མི་འབྱུང་ན། །མཁས་པ་སུ་ཞིག་ངེས་པར་ནི། །བྱ་དགོས་ཆད་པས་བྱེད་པར་འགྱུར། །ཁྱོད་འདི་བྱས་ནས་གསལ་བར་ནི། །ནགས་སུ་འགྲོ་བྱ་སྙམ་སེམས་ན། །གང་ཞིག་བྱས་ཀྱང་གཏང་བྱ་བ། །དེ་བྱས་ཡོན་ཏན་ཅི་ཞིག་ཡོད། །བདག་ནི་འཆིའོ་སྙམ་སེམས་པ། །གང་ལ་ངེས་པར་ཡོད་གྱུར་པ། །དེ་ནི་ཆགས་པ་ཡོངས་བཏང་ཕྱིར། །འཆི་བདག་ལ་ཡང་ག་ལ་འཇིགས། །བྱང་ཆུབ་སེམས་དཔའི་རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་རྟག་པར་འཛིན་པ་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་དང་པོའོ།། །།ལུས་ནི་དགྲ་ལྟར་མཐོང་མོད་ཀྱི། །དེ་ལྟ་ན་ཡང་དེ་ཡང་བསྲུང་།ཁྲིམས་ལྡན་ཡུན་རིང་འཚོ་བ་ནི། །
དེ་ལས་བསོད་ནམས་ཆེན་པོ་བྱེད། །མི་རྣམས་སྡུག་བསྔལ་ལུས་ཉིད་ལས། །སྐྱེ་ཞིང་བདེ་བ་གཞན་ལས་ན། །སྡུག་བསྔལ་ཀུན་གྱི་སྣོད་གྱུར་པ། །ཁོག་པ་ལ་ཁྱོད་ཅི་སྟེ་གུས། །གང་ཚེ་བདེ་བ་སྡུག་བསྔལ་ལྟར། །ཤས་ཆེར་མི་ལ་མི་འབྱུང་ན། །དེ་ལྟར་སྡུག་བསྔལ་ཆེན་པོ་དེ། །ཆུང་ངོ་སྙམ་དུ་སེམས་བྱེད་དམ། །འཇིག་རྟེན་བདེ་ལ་མངོན་ཕྱོགས་ཤིང་། །བདེ་གྱུར་དག་ཀྱང་རྙེད་དཀའ་སྟེ། །དེས་ན་སྐྱེ་བོ་འཇིག་འདི་ལ། །སྡུག་བསྔལ་རྗེས་སུ་སྙེགས་པ་འདྲ། །སྡུག་བསྔལ་འདོད་པས་རྙེད་འགྱུར་གྱི། །འདོད་པས་བདེ་བ་ག་ལ་ཡིན། །ཁྱོད་ཅིས་དཀོན་པ་ལ་གུས་ཤིང་། །མོད་ལ་ཅི་སྟེ་ཁྱོད་མི་འཇིགས། །བདེ་བར་གྱུར་པའི་ལུས་པོ་ནི། །སྡུག་བསྔལ་སྣོད་དུ་གྱུར་ཏེ་དེས། །ལུས་ལ་གུས་དང་དགྲ་ལ་གུས། །དེ་དག་གཉིས་ནི་མཚུངས་པར་སྣང་། །ལུས་ནི་ཤིན་ཏུ་འགོར་བས་ཀྱང་། །བདེ་བའི་བདག་ཏུ་མི་འགྱུར་ཏེ། །རང་བཞིན་གཞན་གྱིས་ཟིལ་གནོན་ཅེས། །བྱ་བར་རིགས་པ་མ་ཡིན་ནོ། །མཆོག་ལ་ཡིད་ཀྱི་སྡུག་བསྔལ་ཏེ། །ཐ་མལ་རྣམས་ལ་ལུས་སྐྱེས་སོ། །སྡུག་བསྔལ་གཉིས་ཀྱིས་འཇིག་རྟེན་འདི། །ཉིན་རེ་ཉིན་རེ་འཇོམས་པར་བྱེད། །བདེ་བ་རྟོག་པའི་དབང་གྱུར་ཅིང་། །རྟོག་པ་སྡུག་བསྔལ་དབང་གྱུར་པ། །དེ་ཕྱིར་ཀུན་ན་སྡུག་བསྔལ་ལས། །ཤིན་ཏུ་སྟོབས་ལྡན་ཅི་ཡང་མེད། །ཇི་ལྟ་ཇི་ལྟར་དུས་འགྲོ་བ། །དེ་ལྟ་དེ་ལྟར་སྡུག་བསྔལ་འཕེལ། །དེ་ཕྱིར་བདེ་བ་ལུས་འདི་ཡི། །གཞན་པོ་ལྟ་བུར་སྣང་བར་འགྱུར། །སྡུག་བསྔལ་གྱི་ནི་རྒྱུ་མང་པོ། །ནད་རྣམས་དང་གཞན་སྣང་གྱུར་པ། །དེ་སྙེད་དུ་ནི་མི་རྣམས་ལ། །བདེ་བའི་རྒྱུ་དག་སྣང་མ་ཡིན། །འཕེལ་བཞིན་པ་ཡི་བདེ་བ་ལ། །ཇི་ལྟར་བཟློག་པ་མཐོང་འགྱུར་བ། །དེ་ལྟར་སྡུག་བསྔལ་འཕེལ་བཞིན་ལ། །བཟློག་པ་ཡོད་པ་མ་ཡིན་ནོ། །རྐྱེན་དང་ལྷན་ཅིག་བདེ་བ་ལ། །བཟློག་པ་མཐོང་བ་སྡུག་བསྔལ་ལ། །རྐྱེན་རྣམས་ཉིད་དང་ལྷན་ཅིག་ཏུ། །བཟློག་པ་ཡོད་པ་མ་ཡིན་ནོ། །ཁྱོད་འཆི་བཞིན་ལ་དུས་སོང་ཞིང་། །འགྲོ་བ་དང་ནི་འགྲོར་འགྱུར་ན། །འཆི་བཞིན་པ་ནི་བདེ་འགྱུར་ཞེས། །རྣམ་པ་ཀུན་ཏུ་འང་རིགས་མ་ཡིན། །ལུས་ཅན་རྣམས་ལ་བཀྲེས་སོགས་ཀྱིས། །གནོད་པ་རྟག་ཏུ་ཡོད་འགྱུར་ན། །གནོད་བཞིན་པ་ནི་བདེའོ་ཞེས། །རྣམ་པ་ཀུན་ཏུ་འང་རིགས་མ་ཡིན། །
ནུས་པ་མེད་པས་འབྱུང་ཀུན་ལ། །ཚོགས་པ་ཞེས་བྱ་སྐྱེ་འགྱུར་ཏེ། །འགལ་བ་རྣམས་ལ་བདེའོ་ཞེས། །རྣམ་པ་ཀུན་ཏུ་མི་རིགས་སོ། །གྲང་སོགས་རྣམས་ལ་ཕྱིར་འཚོ་བ།! །རྟག་ཏུ་ཡོད་པ་མ་ཡིན་ན། །བརླག་བཞིན་པ་ནི་བདེའོ་ཞེས། །རྣམ་པ་ཀུན་རྩོལ།!མེད་པར་སྤྱོད་པ་ཞེས། །བྱ་འགའ་ས་སྟེང་ཡོད་མིན་ན། །ལས་བྱེད་བདེ་བ་ཡིན་ནོ་ཞེས། །རྣམ་པ་ཀུན་ཏུ་འང་མི་རིགས་སོ། །འདི་དང་གཞན་དུ་བདག་ཉིད་ནི། །རྟག་ཏུ་སྡིག་ལས་བསྲུང་བར་བྱ། །ངན་འགྲོ་ཡོད་ན་བདེ་བ་ཞེས། །རྣམ་པ་ཀུན་ཏུ་རིགས་མ་ཡིན། །བཞོན་པ་སོགས་ལ་མི་རྣམས་ཀྱི། །བདེ་བ་རྟག་ཏུ་ཡོད་མིན་ཏེ། །དང་པོར་གང་ལ་རྩོམ་མེད་པ། །དེ་ལ་མཐར་འཕེལ་ག་ལ་ཡོད། །གསེར་གྱི་སྣོད་དུ་སྐྱུག་པ་ན། །ཇི་ལྟར་འགའ་ཞིག་དགའ་འགྱུར་བ། །དེ་བཞིན་སྡུག་བསྔལ་ཕྱིར་འཆོས་དེ། །བདེའོ་སྙམ་དུ་སེམས་པར་བྱེད། །རྩོམ་པས་སྐྱེས་པ་ལྡོག་འགྱུར་ན། །སྡུག་བསྔལ་རྩོམ་ལའང་བདེ་ཅི་ཡོད། །དེ་ཕྱིར་ཐུབ་པས་སྐྱེ་འཇིག་དག །སྡུག་བསྔལ་ཡིན་ཞེས་གསུངས་པ་སྙམ། །སོ་སོའི་སྐྱེ་བོས་སྡུག་བསྔལ་ནི། །བདེ་བས་གཡོགས་ལྟར་མི་མཐོང་ན། །གང་གིས་སྡུག་བསྔལ་སྒྲིབ་འགྱུར་བ། །བདེ་བ་ཞེས་བྱ་ཅི་ཡང་མེད། །ཁྱོད་སྡུག་ཆགས་བྲལ་མེད་དོ་ཞེས། །སོ་སོའི་སྐྱེ་བོ་བསྟན་བྱ་དེས། །ངེས་པར་དེ་བཞིན་གཤེགས་རྣམས་ཀྱིས། །གཏི་མུག་ཀུན་གྱི་ཐ་མར་གསུངས། །མི་རྟག་པ་ལ་ངེས་པར་གནོད། །གང་ལ་གནོད་ཡོད་དེ་བདེ་མིན། །དེ་ཕྱིར་མི་རྟག་གང་ཡིན་པ། །ཐམས་ཅད་སྡུག་བསྔལ་ཞེས་བྱར་འགྱུར། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་བདེ་བར་འཛིན་པ་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་གཉིས་པའོ།། །།ཡུན་ནི་ཤིན་ཏུ་རིང་པོས་ཀྱང་། །ཡུལ་ལ་མུར་ཐུག་ཡོད་མིན་ཏེ། །སྨན་པ་ངན་ལྟར་ཁྱོད་ལུས་ལ། །ངལ་བ་འབྲས་བུ་མེད་པ་འབྱུང་། །བསྟེན་པར་བྱེད་པ་འགའ་ཞིག་ལ། །ས་སྲེད་ལྡོག་པ་མེད་ཇི་བཞིན། །བསྟན་བཞིན་པ་ཡི་མི་རྣམས་ལ། །འདོད་རེ་འཕེལ་བ་དེ་བཞིན་ནོ། །བུད་མེད་ཀུན་ལ་འཁྲིག་པ་ལ། །ཐ་དད་ཅུང་ཟད་ཡོད་མིན་ཞིང་། །གཟུགས་གཞན་གྱི་ཡང་ལོངས་སྤྱོད་ན། །ཁྱོད་ལ་མོ་མཆོག་གིས་ཅི་བྱེད། །
གང་ལ་གང་ཡིད་འོང་གྱུར་པ། །དེ་ནི་དེ་ཡིས་འཚེངས་པར་རློམ། །ཁྱི་སོགས་རྣམས་ལའང་ཐུན་མོང་པས། །བློ་གྲོས་ངན་པ་ཁྱོད་ཅིས་ཆགས། །ཁྱོད་ལ་ཡན་ལག་ཀུན་མཛེས་མོ། །གང་དེ་ཁྱོད་ལ་ངོ་མཚར་ལྟར། །ཇི་ལྟར་ཡང་ནི་འགྱུར་བ་མིན། །ཡོན་ཏན་ལྡན་པ་སྡུག་པ་དང་། །བཟློག་པ་ལ་ནི་བཟློག་མཐོང་ན། །རྣམ་པར་གནས་མེད་སྔ་མའམ། །ཅིག་ཤོས་ལས་ནི་གང་ཞིག་བདེན། །བླུན་པོའི་འདོད་ཆགས་ཡོན་ཏན་དང་། །ལྡན་པ་འབའ་ཞིག་ལ་མི་སྐྱེ། །རྒྱུ་མེད་པ་ལ་འཇུག་རྣམས་ཀྱི། །ལྡོག་པ་རྒྱུ་ལས་ག་ལ་ཡིན། །ཇི་སྲིད་གཞན་བཤེས་མ་གྱུར་པ། །དེ་སྲིད་དེ་ནི་ཁྱོད་ལ་འདུག། །ནད་ངན་དང་མཚུངས་བུད་མེད་ནི། །རྟག་ཏུ་རྐྱེན་ལས་བསྲུང་བར་བྱ། །དར་མར་རང་གིས་བྱས་པ་གང་། །དེ་ནི་རྒན་པོ་མི་འདོད་ན། །ཇི་ལྟ་བུར་ན་གྲོལ་བ་རྣམས། །དེ་ལ་ཤིན་ཏུ་གདུང་མི་འགྱུར། །མ་ཆགས་པ་ལ་བདེ་མེད་ཅིང་། །དེ་ཡང་བླུན་མིན་ལ་མེད་ན། །གང་ཡིད་རྟག་ཏུ་ཕྱིར་ཕྱོགས་པ། །དེ་ཡི་བདེ་བ་ཅི་འདྲ་ཞིག །ཁྱོད་ལ་གུས་བཞིན་རྟག་པར་ནི། །མོ་དང་ལྷན་ཅིག་ཕྲད་པ་མེད། །འདི་ངའི་གཞན་གྱི་མ་ཡིན་ཞེས། །ཡོངས་འཛིན་འདི་ཀོ་ཅི་ཞིག་ཡིན། །གལ་ཏེ་འདོད་ཆགས་བདེ་ཡིན་ན། །བུད་མེད་དག་གིས་དགོས་མེད་འགྱུར། །བདེ་བ་དོར་བྱ་ཞེས་བྱ་བར། །ཇི་ལྟར་ཡང་ནི་མི་མཐོང་ངོ་། །བུད་མེད་ལྷན་ཅིག་སྦྱོར་བ་ནའང་། །བདེ་བ་གཞན་ལས་སྐྱེ་འགྱུར་ན། །དེ་རྒྱུ་མཛའ་མོ་ཉིད་ཡིན་ཞེས། །བླུན་པོ་མིན་པ་སུ་ཞིག་འཛིན། །མཛེ་པོ་འཕྲུག་བཞིན་འདོད་ཆགས་ཀྱི། །ལྡོངས་པས་འདོད་པའི་སྐྱོན་མི་མཐོང་། །ཆགས་བྲལ་རྣམས་ལ་མཛེ་པོ་ལྟར། །ཆགས་ཅན་སྡུག་བསྔལ་ལྡན་པར་སྣང་། །མགོན་མེད་བཀྲེས་པས་ཉེན་རྣམས་ལ། །མུ་གེར་བྱ་བ་གང་བྱུང་བ། །དེ་ནི་བུད་མེད་ཕྲད་པའི་ཚེ། །ཆགས་ཅན་ཀུན་གྱིས་གཡོ་བ་ཡིན། །དྲེགས་པ་ཡིས་ནི་ཁ་ཅིག་ལ། །བཤང་ཁང་ལ་ཡང་ཆགས་སྐྱེ་ཞིང་། །བུད་མེད་ལ་ལ་ལ་ཆགས་པ། །འགའ་ཞིག་འགའ་ལ་ཕྲག་དོག་བྱེད། །མི་གཙང་བ་ལ་གཏི་མུག་ནི། །འབྱུང་ཞིང་ཁྲོ་བ་འབྱུང་རིགས་ཀྱི། །རྣམ་པ་ཀུན་ཏུ་འདོད་ཆགས་ནི། །འབྱུང་བར་རིགས་པ་མ་ཡིན་ནོ། །
མི་རྣམས་མ་གཏོགས་མི་གཙང་བའི། །སྣོད་ནི་སྨད་པར་འབྱུང་འགྱུར་ན། །གང་ལས་མི་གཙང་འབྱུང་འགྱུར་བ། །དེ་ནི་སྨད་པར་ཅིས་མི་སེམས། །དངོས་པོ་གཙང་བ་ཐམས་ཅད་ཀྱི། །ཐ་མ་གང་དུ་དམིགས་འགྱུར་བ། །དེ་ལ་གཙང་བ་ཡོད་དོ་ཞེས། །བློ་ལྡན་མི་ནི་གང་ཞིག་སྨྲ། །གང་ཞིག་བཤང་ཁང་ནང་འདུག་ཅིང་། །དེ་མེད་པར་ནི་མི་གནས་པ། །མི་གཙང་བ་ཡི་སྲིན་དེ་ལ། །རྨོངས་ལས་དྲེགས་པ་སྐྱེ་བར་ཟད། །ཐབས་གང་གིས་ཀྱང་ལུས་ཀྱི་ནི། །ནང་ཕྱུགས་གཙང་མར་མི་འགྱུར་ན། །ཁྱོད་ནི་ནང་ལ་འབད་བྱེད་པ། །དེ་ལྟར་ཕྱི་རོལ་ལ་མ་ཡིན། །གལ་ཏེ་མཛེ་ཅན་བཞིན་དེ་ལྟར། །གཅིན་ཅན་ཀུན་ལ་མཚུངས་མིན་ན། །མཛེ་ཅན་ཇི་བཞིན་གཅིན་ཅན་དེ། །དེ་ལྟར་མི་ཀུན་གྱིས་སྤོང་འགྱུར། །ཡན་ལག་ཉམས་པ་འགའ་ཞིག་ནི། །སྣ་ཡི་ཚབ་མས་དགར་འགྱུར་ལྟར། །མི་གཙང་ཕྱིར་འཆོ་མེ་ཏོག་ལ། །སོགས་ལས་འདོད་ཆགས་དེ་བཞིན་འདོད། །གང་དུ་འདོད་ཆགས་བྲལ་སྐྱེ་བ། །དེ་གཙང་ཞེས་བྱར་མི་རིགས་ལ། །ངེས་པར་འདོད་ཆགས་རྒྱུར་གྱུར་པའི། །དངོས་དེ་གང་ན་ཡོད་མ་ཡིན། །མདོར་ན་མི་རྟག་མི་གཙང་དང་། །སྡུག་བསྔལ་བ་དང་བདག་མེད་ཅེས། །བྱ་བ་བཞི་པོ་ཐམས་ཅད་ནི། །གཅིག་ཉིད་ལ་ནི་སྲིད་པར་འགྱུར། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་གཙང་བར་འཛིན་པ་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་གསུམ་པའོ།། །།བདག་གམ་བདག་གི་སྙམ་དྲེགས་པ། །སྲིད་པར་དམ་པ་སུ་ལ་འགྱུར། །གང་ཕྱིར་ལུས་ཅན་ཐམས་ཅད་ལ། །ཡུལ་རྣམས་ཐམས་ཅད་མཚུངས་ཕྱིར་རོ། །ཚོགས་བྲན་དྲུག་ཆོས་བསྟབས་གྱུར་པ། །ཁྱོད་ཀྱི་དྲེགས་པ་ཅི་ཞིག་ཡིན། །གང་ངམ་དེ་རུ་བྱ་བ་ནི། །དབང་དུ་བྱས་ལ་རག་ལས་འགྱུར། །ཇི་ལྟར་གསོ་བྱ་འཐོབ་ཐོབ་པ། །རྗེ་བོ་གཏོང་བར་སེམས་དེ་བཞིན། །རྗེ་བོ་སྦྱིན་བྱ་བྱིན་བྱས་ནས། །བདག་ནི་གཏོང་བ་པོ་སྙམ་རློམ། །གཞན་རྣམས་ཀྱིས་ན་སྡུག་བསྔལ་གནས། །ཁྱོད་ལ་ཕྱིན་ཅི་ལོག་ཏུ་དྲན། །གཞན་གྱི་ལས་ཀྱིས་འཚོ་ཁྱོད་ཀྱི། །དགའ་བ་གང་གིས་བསྐྱེད་པར་འགྱུར། །འཇིག་རྟེན་གྱི་ནི་བསྲུང་པོ་དང་། །བསྲུང་བྱར་ས་བདག་སྣང་གྱུར་ན། །ཅི་སྟེ་གཅིག་གིས་དྲེགས་ཡོད་ཅིང་། །
ཅི་སྟེ་གཞན་གྱི་དྲེགས་བྲལ་མིན། །རིགས་ཀུན་ནང་ནས་རང་ལས་ལ། །དགའ་བས་འཚོ་རྣམས་རྙེད་པ་དཀའ། །གལ་ཏེ་ཁྱོད་ལ་མི་དགེར་འགྱུར། །དེས་ན་ཁྱོད་ལ་འགྲོ་བཟང་དཀོན། །གཞན་གྱིས་སྐུལ་ཏེ་བྱེད་པ་གང་། །དེ་ནི་ས་སྟེང་བླུན་ཞེས་བྱ། །ཁྱོད་དང་མཚུངས་པའི་གཞན་དབང་གཞན། །འགའ་ཡང་ཡོད་པ་མ་ཡིན་ནོ། །བསྲུང་བ་ང་ལ་རག་ལས་ཞེས། །འཇིག་རྟེན་ལས་ནི་གླུ་ལེན་ཞིང་། །རང་ཉིད་ཀྱིས་ཀྱང་སྡིག་བྱེད་ན། །དེ་མཚུངས་བརྩེ་མེད་སུ་ཞིག་ཡོད། །སྐྱེ་བོ་སྡིག་ཏོ་བྱེད་རྣམས་ལ། །གལ་ཏེ་བརྩེ་བ་བྱ་མིན་ན། །བྱིས་པ་སོ་སོའི་སྐྱེ་བོ་ནི། །ཐམས་ཅད་བསྲུང་བྱར་མི་འགྱུར་རོ། །བདག་ཉིད་དགའ་བ་བསྐྱེད་པ་ཡི། །རྒྱུ་ནི་གང་དུ་ཡོད་མི་འགྱུར། །ལུང་ལ་སོགས་པའི་རྒྱུ་རྣམས་ཀྱི། །བསོད་ནམས་མིན་པ་འཇིག་ཡོད་མིན། །ཡང་དག་བསྲུང་བྱེད་ཅེས་བྱ་བ། །གལ་ཏེ་ས་བདག་ཆོས་ཡིན་ན། །ཉོན་མོངས་གཟོ་མཁན་རྣམས་ལ་ཡང་། །ཅི་སྟེ་ཆོས་སུ་འགྱུར་མ་ཡིན། །འཇིག་རྟེན་ས་བདག་རག་ལས་ཀྱི། །ས་བདག་དེ་ནི་སྨད་དེ་དཔེར། །དམ་པས་འཇིག་རྟེན་ཀུན་གྱི་མ། །སྲིད་པ་ཡི་ནི་སྲེད་པ་བཞིན། །བླུན་མིན་རྒྱལ་སྲིད་མི་ཐོབ་ལ། །བླུན་ལ་བརྩེ་བ་ཡོད་མིན་ན། །བསྲུང་པོ་ཡིན་ཡང་མི་ཡི་བདག །བརྩེ་བ་མེད་པར་ཆོས་མི་གནས། །དྲང་སྲོང་རྣམས་ཀྱི་སྤྱོད་པ་ཀུན། །མཁས་པས་བྱེད་པ་མ་ཡིན་ཏེ། །གང་གི་ཕྱིར་ན་དེ་དག་ལ། །དམན་འབྲིང་ཁྱད་འཕགས་ཡོད་ཕྱིར་རོ། །སྔར་གྱི་ས་བདག་དགེ་རྣམས་ཀྱི། །འཇིག་རྟེན་བུ་རྣམས་བསྐྱངས་པ་དེ། །རྩོད་དུས་ཆོས་ལ་བརྟེན་རྣམས་ཀྱི། །དེང་སང་རི་དགས་དགོན་པར་བྱས། །གླགས་སུ་སྣུན་པའི་རྒྱལ་པོ་ལ། །གལ་ཏེ་སྡིག་པ་ཡོད་མིན་ན། །རྐུན་པོ་གཞན་རྣམས་ལ་ཡང་དེ། །དང་པོ་ཉིད་དུ་ཡོད་མ་ཡིན། །ཆང་སོགས་དག་ལ་བདོག་པ་ཀུན། །ཡོངས་གཏོང་མཆོད་པ་མ་ཡིན་ན། །གཡུལ་ངོར་བདག་ཀྱང་ཡོངས་གཏོང་བས། །མཆོད་པ་སྙམ་དུ་སེམས་སམ་ཅི། །རྒྱལ་པོ་འཇིག་རྟེན་མགོན་ཁྱོད་ལ། །མགོན་དགའ་ཡོད་པ་མ་ཡིན་ན། །མགོན་ཉིད་དེ་ཡི་བདག་ཉིད་ན། །མ་བཏང་སུ་ཞིག་དགའ་བར་འགྱུར། །རྒྱལ་པོ་ཤི་བར་གྱུར་པ་ལའང་། །གྲགས་པས་ཡོན་ཏན་འགའ་མི་འབྱུང་། །
ཁྱོད་ཡོན་མེད་དང་ཁྱི་འཚེད་ལ། །གྲགས་པ་ཆེན་པོ་ཅི་སྟེ་མེད། །གང་ཚེ་ཀུན་གྱི་དབང་ཕྱུག་ཉིད། །བསོད་ནམས་ཀྱིས་བསྐྱེད་དེ་ཡི་ཚེ། །འདི་ནི་དབང་ཕྱུག་རྟེན་མིན་ཞེས། །བརྗོད་པར་རྙེད་པ་མ་ཡིན་ནོ། །འཚོ་བའི་ཐབས་སོ་ཅོག་ལ་ཡང་། །འཇིག་རྟེན་ན་ནི་རིགས་ཞེས་བསྙད། །དེས་ན་སེམས་ཅན་ཐམས་ཅད་ལ། །རིགས་ལ་དབྱེ་བ་ཡོད་མ་ཡིན། །འདས་དུས་ཤིན་ཏུ་རིང་བ་དང་། །མི་མོ་རྣམས་ཡིད་གཡོ་བ་སྟེ། །དེའི་ཕྱིར་རྒྱལ་རིགས་ཞེས་བྱ་བ། །རིགས་ལས་འགའ་ཡང་ཡོད་མ་ཡིན། །གལ་ཏེ་ལས་ཀྱིས་དམངས་རིགས་ཀྱང་། །རྒྱལ་རིགས་ཞེས་བྱར་འགྱུར་ན་ནི། །ལས་ཀྱིས་དམངས་ཀྱང་བྲམ་ཟེར་ནི། །གང་གིས་འགྱུར་བ་མིན་པར་སེམས། །རྒྱལ་པོའི་སྡིག་ལ་དབང་ཕྱུག་ལྟར། །འགེད་པ་ཡོད་པ་མ་ཡིན་ན། །གཞན་གྱི་དོན་དུ་མཁས་པ་ནི། །སུ་ཞིག་ཕྱི་མར་འཇོམས་པར་བྱེད། །དབང་ཕྱུག་གིས་བསྐྱེད་ང་རྒྱལ་ནི། །མཉམ་དང་ཁྱད་པར་འཕགས་པ་ཡི། །ཕ་རོལ་ནུས་ལྡན་བལྟ་བྱ་སྟེ། །དམ་པའི་སྙིང་ལ་མི་གནས་སོ། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་བདག་ཏུ་འཛིན་པ་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཞི་པའོ།། །།སངས་རྒྱས་རྣམས་ཀྱི་གཡོ་བ་ནི། །རྒྱུ་མེད་དགའ་ཡང་ཡོད་མིན་ཏེ། །དབུགས་ཀྱང་སེམས་ཅན་རྣམས་ལ་ནི། །སྨན་སླད་ཁོ་ནར་ཡང་དག་འབྱུང་། །འཇིག་རྟེན་ཀུན་ལ་འཆི་བདག་སྨྲ། །ཇི་ལྟར་འཇིགས་པ་སྐྱེ་འགྱུར་བ། །དེ་བཞིན་ཀུན་མཁྱེན་སྨྲ་འདི་ནི། །འཆི་བདག་ལ་ཡང་འཇིགས་པ་བསྐྱེད། །ཐུབ་པ་མཛད་དང་མཛད་མིན་དང་། །གསུང་བྱ་གསུང་བྱ་མིན་མངའ་བ། །དེས་ན་ཀུན་མཁྱེན་ཀུན་མཁྱེན་ནི། །མིན་ཞེས་བྱ་བ་ལ་རྒྱུ་ཅི་ཡོད། །སེམས་མ་རྟོགས་པར་འགྲོ་སོགས་ལ། །གང་ཕྱིར་བསོད་ནམས་ལ་སོགས་ལ། །མ་མཐོང་དེ་ཕྱིར་ལས་ཀུན་ལ། །ཡིད་ནི་དོན་པོར་བསྒྲུབ་པར་བྱ། །བསམ་པས་བྱང་ཆུབ་སེམས་དཔའ་ལ། །དགེ་བ་འམ་འོན་ཏེ་མི་དགེའང་རུང་། །ཐམས་ཅད་དགེ་ལེགས་ཉིད་འགྱུར་ཏེ། །གང་ཕྱིར་ཡིད་དེའི་དབང་གྱུར་ཕྱིར། །བྱང་ཆུབ་སེམས་དཔའ་སེམས་དང་པོ། །ས་སྟེང་སྐྱེ་བོ་ཐམས་ཅད་ནི། །འཁོར་ལོས་སྒྱུར་བ་ཉིད་འགྱུར་བ། །དེ་དག་བསོད་ནམས་ལས་ཁྱད་འཕགས། །རིན་ཆེན་རང་བཞིན་མཆོད་རྟེན་ནི། །འཇིག་རྟེན་རྙེད་མཐོ་འགས་བྱས་པ། །
དེ་བས་སེམས་ཅན་གཅིག་འདུལ་པོ། །མཆོག་ཏུ་བཀྲབས་པར་བརྗོད་པ་ཡིན། །དེ་ལ་ཕན་འདོད་བླ་མ་ཡིས། །སློབ་མ་ལ་ནི་སྲིད་ཞུ་བྱ། །གང་ཕྱིར་དེས་ཕན་མི་ཤེས་པ། །དེ་ཕྱིར་སློབ་མ་ཞེས་བྱའོ། །ཁྲོ་ཡང་འབྱུང་པོས་བཟུང་བ་ལ། །སྨན་པ་འཁྲུག་པ་མིན་པ་ལྟར། །ཐུབ་པ་ཉོན་མོངས་དགྲར་གཟིགས་ཀྱི། །ཉོན་མོངས་དང་འབྲེལ་གང་ཟག་མིན། །གང་ཞིག་གང་གང་ལ་དགའ་བ། །དེ་ཡི་དེ་དེ་སྔར་སྤྱད་བྱ། །ཉམས་པར་གྱུར་པ་དམ་ཆོས་ཀྱི། །སྣོད་ནི་ཅིས་ཀྱང་མ་ཡིན་ནོ། །ཇི་ལྟར་མ་ནི་ཁྱད་པར་དུ། །ནད་ཐེབས་བུ་ལ་གདུང་གྱུར་པ། །དེ་བཞིན་བྱང་ཆུབ་སེམས་དཔའི་བརྩེ། །དམ་པ་མིན་ལ་ཁྱད་པར་དུའོ། །ལ་ལའི་སློབ་མར་གྱུར་པ་སྟེ། །ལ་ལ་ཡི་ནི་བླ་མར་འགྱུར། །ཐབས་རིག་སྣ་ཚོགས་ཐབས་ཀྱིས་ནི། །མ་རྟོགས་སེམས་ཅན་རྟོགས་པར་བྱེད། །ཇི་ལྟར་སྨན་པ་མཁས་གྱུར་པས། །ནད་པ་བསྒྲུབ་མེད་དཀོན་དེ་བཞིན། །བྱང་ཆུབ་སེམས་དཔའ་སྟོབས་རྙེད་ནས། །གདུལ་བྱ་མ་ཡིན་ཤིན་ཏུ་དཀོན། །བྱང་ཆུབ་སེམས་དཔའི་ཡུལ་དུ་ནི། །གལ་ཏེ་འགའ་ཞིག་མ་བསྐུལ་བར། །ངན་སོང་འགྲོ་བར་འགྱུར་ན་དེ། །བློ་ལྡན་གཞན་གྱིས་སྨད་བྱར་འགྱུར། །གང་ཞིག་ཉེན་པ་གཞན་དག་ལ། །སྙིང་བརྩེ་ལེགས་ཞེས་མི་འདོད་པ། །དེ་རྗེས་བརྩེ་བས་མགོན་མེད་པ། །དག་ལ་ཇི་ལྟར་སྦྱིན་པ་གཏོང་། །འགྲོ་ལ་ཕན་ཕྱིར་གང་ཞིག་ལ། །འཇིག་རྟེན་སྲིད་དུ་གནས་ཡོད་པ། །དེ་ལ་སོར་གནས་དག་ཕུང་ན། །སྡང་ལ་བསམ་དག་ཅི་ཞིག་ཡོད། །གང་ལ་སྐྱེ་བ་ཐམས་ཅད་དུ། །མངོན་ཤེས་ལྔ་ཆར་ཡང་ཡོད་པ། །དེ་ཡི་དམན་པར་དམན་ལྟའི་དངོས། །འདི་ནི་ཤིན་ཏུ་བྱ་དཀའ་བའོ། །ཐབས་ཀྱིས་ཀུན་ཚེ་བསོད་ནམས་གང་། །ཤིན་ཏུ་ཡུན་རིང་བསགས་པ་དེ། །ཐམས་ཅད་མཁྱེན་པའི་གཞལ་བྱ་ཡང་། །མིན་ཞེས་དེ་བཞིན་གཤེགས་པས་གསུངས། །འཆི་དང་ཆོས་དང་སྲིད་པ་གཞན། །སྦྱིན་པའི་སྒྲ་ཡིས་གསལ་བྱེད་པ། །དེ་ཕྱིར་བྱང་ཆུབ་སེམས་དཔའ་ལ། །སྦྱིན་པའི་སྒྲ་ནི་རྟག་ཏུ་སྙན། །འདིར་བྱིན་པ་ཡི་སྦྱིན་པ་ལས། །འབྲས་བུ་ཆེན་པོ་འབྱུང་འགྱུར་ཞེས། །
ལེན་དང་ཡོངས་སུ་གཏོང་བ་ནི། །ཚོང་སོགས་བཞིན་དུ་སྨད་པར་འགྱུར། །གང་ལ་སྔར་བྱས་སྡིག་པ་ཡང་། །ཡོད་ཉིད་ཡོད་པ་མ་ཡིན་པ། །དགེ་བའི་ལས་ཅན་དེ་ལ་ནི། །བསྒྲུབ་མིན་ཞེས་བྱ་ཡོད་མ་ཡིན། །རླབས་ཆེན་ཡིད་ཅན་དེ་ལ་ནི། །འདི་ནའང་གནོད་པ་ཡོད་མིན་ཏེ། །དེས་ན་དེ་ལ་སྲིད་པ་དང་། །མྱ་ངན་འདས་ལ་ཁྱད་པར་མེད། །གང་ལ་ཀུན་ཚེ་སེམས་དབང་ནི། །ཉིད་ལས་སྐྱེ་བ་ཡོད་གྱུར་པ། །དེ་ཀོ་རྒྱུ་ཅིས་འཇིག་རྟེན་ནི། །ཀུན་གྱི་མངའ་བདག་འགྱུར་མ་ཡིན། །ཕུལ་འགྲོ་ལས་ཆེས་ཕུལ་འགྲོ་བ། །འགའ་ཞིག་འཇིག་རྟེན་འདིར་ཡང་སྣང་། །དེས་ན་བསམ་མི་ཁྱབ་མཐུ་ཡང་། །ངེས་པར་ཡོད་ཅེས་རྟོགས་པར་གྱིས། །བླུན་ལ་ཤིན་ཏུ་ཟབ་ཆོས་ལ། །སྐྲག་པ་སྐྱེ་བ་ཇི་ལྟ་བ། །དེ་བཞིན་གཞན་ལ་ཤིན་ཏུ་ནི། །རྨད་བྱུང་ཆོས་ལ་སྐྲག་པ་སྐྱེ། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་ལྔ་པའོ།། །།བདེ་བས་འདོད་ཆགས་འཕེལ་འགྱུར་ཞིང་། །སྡུག་པས་ཁྲོ་བ་འཕེལ་འགྱུར་ན། །ཅིས་བདེ་དཀའ་ཐུབ་ཅན་མིན་ཞིང་། །སྡུག་པ་དཀའ་ཐུབ་ཅན་ཅིས་ཡིན། །འདོད་ཆགས་ལས་ནི་སྡུད་པ་སྟེ། །ཁྲོ་བའི་ལས་ནི་རྩོད་པའོ། །འབྱུང་བ་ཀུན་ལ་རླུང་བཞིན་དུ། །གཏི་མུག་ལས་ནི་གསོས་འདེབས་པའོ། །མ་འཕྲད་པ་ལ་འདོད་ཆགས་ནི། །སྡུག་བསྔལ་སྦུངས་མེད་ལས་ཁྲོ་སྡུག །ཡོངས་མི་ཤེས་ལས་གཏི་མུག་སྟེ། །དེ་དག་གིས་དེ་དག་མི་རྟོག། །ཇི་ལྟར་བད་ཀན་ལ་མཁྲིས་དང་། །ལྷན་ཅིག་ཕྲད་པ་མེད་མཐོང་བ། །དེ་བཞིན་ཁྲོ་ལ་འདོད་ཆགས་དང་། །ལྷན་ཅིག་ཕྲད་པ་མེད་པར་མཐོང་། །འདོད་ཆགས་བྲན་བཞིན་བཀོལ་བྱ་སྟེ། །གང་ཕྱིར་མི་ཕངས་དེའི་སྨན་ཕྱིར། །ཁྲོ་ལ་རྗེ་བོ་བཞིན་ལྟ་སྟེ། །དེ་སྨན་ཕངས་པ་ཡིན་ཕྱིར་རོ། །དང་པོར་གཏི་མུག་ཡང་དག་འབྱུང་། །བར་དུ་ཁྲོ་བ་འབྱུང་གྱུར་ཞིང་། །ཕྱིས་ནི་འདོད་ཆགས་འབྱུང་བ་སྟེ། །ཉིན་ལ་རིམ་པ་གསུམ་དུའོ། །འདོད་ཆགས་མཛའ་མིན་མཛའ་འདྲ་བ། །དེས་དེ་ལ་ཁྱོད་མི་འཇིགས་ཏེ། །མི་ཕན་མཛའ་བོ་ཁྱད་པར་དུ། །སྐྱེས་བུས་སྤང་བར་བྱ་མིན་ནམ། །འདོད་ཆགས་རྒྱུ་ལས་སྐྱེ་འགྱུར་ཞིང་། །རྐྱེན་ལས་ཀྱང་ནི་སྐྱེ་བ་སྟེ། །འདོད་ཆགས་རྐྱེན་ལས་སྐྱེ་བ་གང་། །
དེ་ནི་སྒྲུབ་སླ་ཅིག་ཤོས་མིན། །ཁྲོ་བ་བསྟན་ཞིང་ངེས་པར་ནི། །མི་དགེ་ཉེས་ཆེན་བྱེད་པ་སྟེ། །དེ་ལྟར་རྣམ་ཀུན་དབྱེ་ཤེས་པས། །ཉོན་མོངས་རྣམས་མཐར་བྱེད་པར་འགྱུར། །ལུས་ལ་ལུས་དབང་ཇི་བཞིན་དུ། །གཏི་མུག་ཀུན་ལ་གནས་གྱུར་ཏེ། །དེ་ཕྱིར་ཉོན་མོངས་ཐམས་ཅད་ཀྱང་། །གཏི་མུག་བཅོམ་པས་བཅོམ་པར་འགྱུར། །རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་ནི། །མཐོང་ན་གཏི་མུག་འབྱུང་མི་འགྱུར། །དེ་ཕྱིར་འབད་པ་ཀུན་གྱིས་འདིར། །གཏམ་དེ་ཁོ་ན་བསྙད་པར་བྱ། །རྟག་ཏུ་ས་འཛིན་སོགས་དགའ་ལ། །གཏོང་པོ་ཟ་པོ་གཙང་ཞིང་འགྱུར། །དེ་འདྲ་ལ་སོགས་མཚན་ཉིད་རྣམས། །འདོད་ཆགས་ཅན་གྱི་མི་ལ་སྣང་། །སངས་རྒྱས་རྣམས་ཀྱིས་ཆགས་ཅན་ལ། །ཟས་དང་ཆོས་གོས་གཙུག་ལག་ཁང་། །དགེ་ལེགས་ཡངས་པ་ཐམས་ཅད་དང་། །རྟག་ཏུ་བླ་མའི་ཐད་ནས་གསུངས། །ནུས་མེད་ལ་ཁྲོས་བདག་ཉིད་ལ། །མདོག་མི་སྡུག་པ་འབའ་ཞིག་བྱེད། །ནུས་པ་གང་ལའང་བརྩེ་མེད་ལ། །དེ་ནི་ཐ་ཆད་ཅེས་བྱར་བརྗོད། །ཡིད་དུ་མི་འོང་བ་ཡི་སྒྲ། །སྔར་བྱས་སྡིག་མཐར་བྱེད་པར་བརྗོད། །རྨོངས་ཤིང་དམ་པ་མ་ཡིན་པ། །བདག་ཉིད་རྣམ་པར་དག་མི་འདོད། །སྙན་པ་ཡིད་དུ་མི་འོང་བའང་། །ངོ་བོ་ཉིད་ཀྱིས་གནོད་མི་བྱེད། །དེ་ཕྱིར་རྣམ་རྟོག་ལས་བྱུང་བ། །གཞན་ལས་ཡིན་ནོ་སྙམ་དུ་རློམ། །ཇི་ལྟར་གཤེ་བ་པོ་ལ་ནི། །ཆད་པར་བྱ་བ་མངོན་བརྗོད་པ། །དེ་བཞིན་སྙན་པར་སྨྲ་པོ་ལ། །མཆོད་པར་བྱ་བར་ཅིས་མི་འགྱུར། །ཁྱོད་ཀྱིས་སྨད་བྱ་མ་སྨྲས་ཀྱང་། །གལ་ཏེ་གཞན་གྱིས་ཤེས་ནས་ནི། །སྨྲ་པོ་ལ་ཁྲོ་མི་རིགས་ན། །ཡང་དག་མིན་སྨྲ་ལ་ཅི་སྨོས། །དམན་ལས་མི་སྙན་ཞེས་བྱ་བ། །ངེས་པར་འབྱུང་བ་མ་ཡིན་ཏེ། །དེ་ཕྱིར་དམན་འབྱུང་མི་སྙན་པ། །ཡན་གར་ཉུང་ཟད་ལྟ་བུར་སྣང་། །གཞན་གནོད་བྱས་པས་བདག་ལ་ནི། །ཡོན་ཏན་ཅུང་ཟད་ཀྱང་མེད་ན། །ཁྱོད་ཁྲོ་ཡོན་ཏན་མེད་པ་ལ། །གུས་པ་ཞེན་པ་རྐྱང་པར་ཟད། །གལ་ཏེ་བཟོད་པས་འབད་མེད་པར། །བསོད་ནམས་ཆེན་པོ་ཐོབ་འགྱུར་ན། །གང་ཞིག་དེ་ལ་གེགས་བྱེད་པ། །དེ་མཚུངས་བླུན་པོ་སུ་ཞིག་ཡོད། །
ཁྲོ་བ་ཞེས་བྱ་ཁྱད་པར་དུ། །སྟོབས་ལྡན་དག་ལ་སྐྱེ་མི་འགྱུར། །དེས་ན་གཞན་འཇོམས་ཁྲོ་བ་ལ། །ཅི་ནས་ཁྱོད་གུས་གྱུར་ཏ་རེ། །གང་ཞིག་ཁྲོ་གནས་ལ་བཟོད་པ། །དེ་ལ་བསྒོམ་པ་སྐྱེ་བར་འགྱུར། །ཡོན་ཏན་གནས་ལ་འཇིགས་ཞེས་པ། །ཁྱོད་ཀྱི་བླུན་པ་འབའ་ཞིག་ཡིན། །སུ་ཞིག་ཁྱད་བསད་མཐར་བྱས་ནས། །འཇིག་རྟེན་ཕ་རོལ་འགྲོ་བ་ཡོད། །དེ་ཕྱིར་བདག་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་དྲུག་པའོ།། །།སྡུག་བསྔལ་རྒྱ་མཚོ་འདི་ལ་མཐའ། །རྣམ་པ་ཀུན་ཏུ་ཡོད་མིན་ན། །བྱིས་ཁྱོད་འདིར་ནི་བྱིང་བ་ལ། །འཇིགས་པ་ཅིས་ན་སྐྱེ་མི་འགྱུར། །ལང་ཚོ་རྒྱབ་ཏུ་བྱུང་གྱུར་ནས། །སླར་ཡང་མདུན་དུ་འབྱུང་བར་འགྱུར། །གནས་ཀྱང་འཇིག་རྟེན་འདི་ལ་ནི། །འགྲོས་ཀྱི། །འགྲོ་བ་གཞན་ཡང་ཡོད་མིན་ན། །གཞན་གྱི་དབང་དང་འཇིགས་མེད་པར། །བློ་དང་ལྡན་པ་སུ་ཞིག་འགྱུར། །མ་འོངས་ཐ་མ་ཡོད་མིན་ཞིང་། །ཀུན་ཚེའང་སོ་སོའི་སྐྱེ་བོ་སྟེ། །ཁྱོད་འདས་ཇི་ལྟ་དེ་བཞིན་དུ། །དེ་ཡང་མི་འགྱུར་དེ་ལྟར་བྱོས། །ཉན་པོ་མཉན་བྱ་འཆད་པོ་རྣམས། །འབྱུང་བ་ཤིན་ཏུ་རྙེད་དཀའ་སྟེ། །དེས་ན་མདོར་ན་འཁོར་བ་ནི། །མཐའ་མེད་མ་ཡིན་མཐའ་བཅས་མིན། །ཕལ་མོ་ཆེར་ནི་མི་རྣམས་ནི། །དམ་པ་མ་ཡིན་ཕྱོགས་ཡོངས་འཛིན། །དེས་ན་སོ་སོ་སྐྱེ་བོ་རྣམས། །ཕལ་ཆེར་ངེས་པར་ངན་འགྲོར་འགྲོ། །ས་སྟེང་སྡིག་པའི་རྣམ་སྨིན་ནི། །ཅོ་འདྲི་ཁོ་ན་ཡིན་པར་མཐོང་། །དེས་ན་སྲིད་པ་བསོད་ནམས་དང་། །མཚུངས་པར་དམ་པ་རྣམས་ལ་སྣང་། །རྣམ་ཤེས་གནས་པ་མེད་པ་ལས། །གལ་ཏེ་སྨྱོན་པར་འགྱུར་ན་ནི། །སྲིད་གནས་སྨྱོན་པ་མ་ཡིན་ཞེས། །མཁས་པ་སུ་ཞིག་སྨྲ་བར་འགྱུར། །འགྲོ་ལ་སོགས་པའི་སྡུག་བསྔལ་ལས། །བཟློག་ལ་ཉམས་པ་མཐོང་འགྱུར་ཏེ། །དེས་ན་ལས་ཀུན་ཟད་པ་ལ། །བློ་གྲོས་ལྡན་པ་བློ་བསྐྱེད་དོ། །
གང་ཚེ་འབྲས་བུ་གཅིག་གིས་ཡང་། །ཐོག་མའི་རྒྱུ་ནི་སྣང་མིན་པ། །དེ་ཚེ་གཅིག་ལ་ཡང་རྒྱས་པར། །མཐོང་ནས་གང་ལ་འཇིགས་མི་འབྱུང་། །འབྲས་བུ་ཐམས་ཅད་ངེས་པར་ནི། །འགྲུབ་པར་འགྱུར་བ་མ་ཡིན་ཞིང་། །གྲུབ་ལ་ངེས་པར་མཐར་འགྱུར་ན། །དེ་ཡི་དོན་དུ་ཅི་སྟེ་འཇོམས། །ལས་ནི་འབད་པས་བྱེད་འགྱུར་ཞིང་། །བྱས་ཟིན་འབད་པ་མེད་པར་འཇིག །དེ་ལྟར་གྱུར་ཀྱང་ཁྱོད་ལ་ཀོ། །ལས་ལ་ཆགས་བྲལ་འགའ་ཡོད་མིན། །འདས་ལ་བདེ་བ་ཡོད་མིན་ཞིང་། །མ་འོངས་པ་ལའང་ཡོད་མིན་ལ། །ད་ལྟར་བྱུང་བའང་འགྲོ་ཉིད་དེ། །ཁྱོད་ཀྱི་ལས་འདི་གང་ཕྱིར་ཡིན། །མཁས་པ་རྣམས་ལ་མཐོ་རིས་ཀྱང་། །དམྱལ་དང་མཚུངས་པར་འཇིགས་སྐྱེ་འགྱུར། །རྣམ་པ་ཀུན་ཏུ་དེ་རྣམས་ལ། །སྲིད་གང་འཇིགས་པ་མི་བསྐྱེད་དཀོན། །གལ་ཏེ་བྱིས་པའང་རྣམ་ཀུན་ཏུ། །འཁོར་བའི་སྡུག་བསྔལ་ཤེས་གྱུར་ན། །སྐད་ཅིག་དེ་ལ་སེམས་དང་ནི། །ལྷན་ཅིག་གཏན་ཏུ་འཇིག་པར་འགྱུར། །སེམས་ཅན་རློམ་པ་ཅན་མིན་དཀོན། །རློམ་ཅན་སྙིང་རྗེ་ལྡན་པ་མེད། །དེས་ན་སྣང་ནས་སྣང་བར་ནི། །འགྲོ་བ་ཤིན་ཏུ་རྙེད་དཀར་གསུངས། །འདི་ན་ཡུལ་ལོགས་གྱུར་པ་ལ། །ཡུལ་ནི་དམིགས་པར་གྲགས་གྱུར་ཏེ། །ཕྱིན་ཅི་ལོག་ཏུ་གྱུར་པའི་ཆོས། །རྒྱུ་ཅི་ཞིག་གིས་ཡང་དག་འདོད། །བསོད་ནམས་འབྲས་བུ་ཕྱུག་པ་སྟེ། །དེ་གཞན་ལས་ཡོངས་བསྲུང་བྱ་ན། །རྟག་ཏུ་གཞན་ལས་བསྲུང་བྱ་གང་། །དེ་ནི་ཇི་ལྟར་བདག་གིར་འགྱུར། །འཇིག་རྟེན་ཚུལ་ལུགས་གང་དང་གང་། །དེ་དང་དེ་ལ་ཆོས་རྗེས་འཇུག །དེ་ཡི་ཕྱིར་ན་ཆོས་པས་ཀྱང་། །འཇིག་རྟེན་སྟོབས་ལྡན་ལྟ་བུར་སྣང་། །ཡུལ་ནི་དགེ་བ་ཡིད་འོང་ན། །ཡུལ་དེའང་ངན་པར་བརྩི་བ་ཡིན། །གང་ཞིག་བཏང་བས་ཤིས་འགྱུར་བ། །དེ་གྲུབ་པ་ཡི་ཅི་ཞིག་དགོས། །གང་ལ་བཀའ་ལུང་དགོས་མེད་པ། །དེ་ལ་ཆོས་དོན་མེད་པར་འགྱུར། །གང་ཡང་བཀའ་ལུང་དོན་གཉེར་ཉིད། །དེ་ནི་སྐྱེ་ནང་བླུན་ཞེས་བྱ། །མ་འོངས་པ་ཡི་འབྲས་མཐོང་ནས། །ཆོས་འདོད་ཆགས་པས་ཁྱོད་ཆགས་ན། །མ་འོངས་མཐའ་ཉིད་མཐོང་གྱུར་ནས། །འཇིགས་པ་མེད་དམ་ཅི་ཞིག་བྱེད། །གླ་བ་ལྟ་བུ་བསོད་ནམས་ནི། །རྣམ་ཀུན་རྔན་པ་དང་མཚུངས་པ། །གང་དག་དགེ་བའང་འདོད་པ། །དེ་དག་མི་དགེ་ཇི་ལྟར་བྱེད། །
གང་གིས་འགྲོ་བ་འཁྲུལ་འཁོར་གྱི། །ཚོགས་འདྲ་སྒྱུ་མའི་སྐྱེས་བུ་ལྟར། །མཐོང་བར་གྱུར་པ་དེ་དག་ཆེས། །གསལ་བར་གོ་འཕང་དམ་པར་འགྲོ། །གང་དག་ལ་ནི་འཁོར་བ་ན། །ཡུལ་རྣམས་ཀྱིས་ཀྱང་དགའ་མེད་པ། །དེ་དག་ལ་དང་རྣམ་ཀུན་ཏུ། །དགའ་བ་འཐད་པ་མ་ཡིན་ཉིད། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་མི་ཉིད་ཀྱིས་འདོད་པ་ལོངས་སྤྱོད་ལ་ཞེན་པ་སྤང་བའི་ཐབས་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བདུན་པའོ།། །།ཇི་ལྟར་མི་མཐུན་མི་རྣམས་ལ། །མཛའ་བ་ཡུན་རིང་མི་གནས་པ། །དེ་བཞིན་ཀུན་ལ་སྐྱོན་ཤེས་ལ། །འདོད་ཆགས་ཡུན་རིང་མི་གནས་སོ། །ལ་ལ་དེ་ཉིད་ལ་ཆགས་ཏེ། །ལ་ལ་དེ་ཉིད་ལ་སྡང་ཞིང་། །ལ་ལ་དེ་ཉིད་ལ་རྨོངས་པ། །དེ་ཕྱིར་འདོད་པ་དོན་མེད་པའོ། །རྟོག་པ་མེད་པར་འདོད་ཆགས་ལ། །སོགས་ལ་ཡོད་ཉིད་ཡོད་མིན་ན། །ཡང་དག་དོན་དེ་རྟོག་པ་ཞེས། །བློ་དང་ལྡན་པ་སུ་ཞིག་འཛིན། །འགའ་ལའང་གང་དང་ལྷན་ཅིག་ཏུ། །བཅིངས་པ་ཞེས་བྱ་ཡོད་མིན་ཏེ། །གཞན་དང་ལྷན་ཅིག་བཅིངས་པ་ལ། །བྲལ་བར་རིགས་པ་མ་ཡིན་ནོ། །བསོད་ནམས་ཆུང་ངུ་ཆོས་འདི་ལ། །ཐེ་ཚོམ་ཟ་བར་ཡང་མི་འགྱུར། །ཐེ་ཚོམ་ཟ་བ་ཙམ་ཞིག་གིས། །སྲིད་པ་ཧྲུལ་པོར་བྱས་པར་འགྱུར། །ཆོས་གང་ཞིག་ལ་ཐུབ་པ་ཡིས། །ཐར་བའི་བར་དུ་འཕེལ་བ་ཉིད། །གསུངས་དེར་གང་ལ་གུས་མེད་པ། །དེ་ནི་གསལ་བར་བློ་ལྡན་མིན། །བདག་ནི་མྱ་ངན་འདའ་འགྱུར་ཞེས། །སྟོང་ཉིད་སྟོང་ལྟར་མཐོང་མིན་ཏེ། །ལོག་ལྟས་མྱ་ངན་མི་འདའ་བར། །དེ་བཞིན་གཤེགས་པ་རྣམས་གསུང་ངོ་། །གང་ལས་འཇིག་རྟེན་བསྟན་བྱུང་བ། །དེ་ལས་འཇུག་པ་གསུངས་པ་སྟེ། །གང་ལས་དོན་དམ་བསྙད་འབྱུང་བ། །དེ་ལས་ལྡོག་པ་གསུངས་པའོ། །ཀུན་ཡོད་མ་ཡིན་ཅི་བྱ་ཞེས། །ཁྱོད་ལ་འཇིགས་པ་སྐྱེ་འགྱུར་གྲང་། །གལ་ཏེ་བྱ་བ་ཡོད་ན་ནི། །ཆོས་འདི་ཟློག་བྱེད་མི་འགྱུར་རོ། །ཁྱོད་ལ་རང་ཕྱོགས་ཆགས་ཡོད་ཅིང་། །གཞན་གྱི་ཕྱོགས་ལ་མི་དགའ་ན། །མྱ་ངན་འདས་པར་མི་འགྲོ་སྟེ། །གཉིས་སྤྱོད་ཞི་བར་ཡོངས་མི་འགྱུར། །བྱེད་མེད་མྱ་ངན་འདའ་འགྱུར་ཞིང་། །བྱེད་པས་ཡང་སྲིད་འགྱུར་ཏེ་དེས། །བསམ་བྲལ་མེད་པས་མྱ་ངན་ལས། །
འདས་པ་ཐོབ་སླ་ཅིག་ཤོས་མིན། །གང་ལ་འདིར་སྐྱོ་ཡོད་མིན་པ། །དེ་ལ་ཞི་གུས་ག་ལ་ཡོད། །རང་ཁྱིམ་ལས་བཞིན་སྲིད་པ་ནི། །འདི་ནས་འབྱུང་བའང་བྱ་བར་དཀའ། །ལ་ལ་སྡུག་བསྔལ་ཟིལ་མནན་ཅིང་། །འཆི་བར་འདོད་པ་དག་སྣང་སྟེ། །དེ་ཚེ་དེ་དག་གཏི་མུག་ཕྱིར། །གོ་འཕང་དམ་པར་མི་འགྲོར་ཟད། །སྦྱིན་པ་དམན་པ་ལ་གསུངས་ཤིང་། །འབྲིང་ལ་ཚུལ་ཁྲིམས་གསུངས་པ་སྟེ། །མཆོག་ལ་ཞི་བ་གསུངས་གྱུར་པ། །དེས་ན་རྟག་ཏུ་མཆོག་ཏུ་བྱོས། །བསོད་ནམས་མིན་པ་དང་པོར་ཟློག། །བར་དུ་བདག་ནི་ཟློག་པ་དང་། །ཕྱི་ནས་ལྟ་ཞི་ཀུན་བཟློག་པ། །གང་གིས་ཤེས་དེ་མཁས་པ་ཡིན། །དངོས་པོ་གཅིག་གི་ལྟ་པོ་གང་། །དེ་ནི་ཀུན་གྱི་ལྟ་པོར་བཤད། །གཅིག་གི་སྟོང་ཉིད་གང་ཡིན་པ། །དེ་ཉིད་ཀུན་གྱི་སྟོང་པ་ཉིད། །ཆོས་ཆགས་དེ་བཞིན་གཤེགས་རྣམས་ཀྱི། །མཐོ་རིས་འདོད་པ་རྣམས་ལ་གསུངས། །ཐར་པ་འདོད་རྣམས་ལ་དེ་ཉིད། །སླད་གྱུར་གཞན་དུ་སྨོས་ཅི་དགོས། །བསོད་ནམས་འདོད་པས་སྟོང་པ་ཉིད། །ཀུན་ཚེ་བརྗོད་པར་བྱ་མིན་ཏེ། །གནས་མ་ཡིན་པར་སྦྱར་བའི་སྨན། །དུག་ཏུ་འགྱུར་བ་མ་ཡིན་ནམ། །ཇི་ལྟར་ཀླ་ཀློ་སྐད་གཞན་གྱིས། །གཟུང་བར་མི་ནུས་དེ་བཞིན་དུ། །འཇིག་རྟེན་པ་ཡི་མ་གཏོགས་པར། །འཇིག་རྟེན་གཟུང་བར་ནུས་མ་ཡིན། །ཡོད་དང་མེད་དང་ཡོད་མེད་དང་། །གཉི་ག་མིན་ཞེས་ཀྱང་བསྟན་ཏེ། །ནད་ཀྱི་དབང་གིས་ཐམས་ཅད་ཀྱང་། །སྨན་ཞེས་བྱ་བར་འགྱུར་མིན་ནམ། །ཡང་དག་མཐོང་ན་གནས་མཆོག་ལ། །ཅུང་ཟད་མཐོང་ན་བཟང་འགྲོ་སྟེ། །དེ་ཕྱིར་ནང་བདག་བསམ་པ་ལ། །མཁས་པ་རྟག་ཏུ་བློ་གྲོས་བསྐྱེད། །དེ་ཉིད་ཤེས་པས་གལ་ཏེ་འདིར། །མྱ་ངན་འདས་པ་མ་ཐོབ་ཀྱང་། །སྐྱེ་བ་ཕྱི་མར་འབད་མེད་པར། །ངེས་པར་ཐོབ་འགྱུར་ལས་བཞིན་ནོ། །བསམ་བཞིན་པ་ནི་བྱ་བ་ན། །ཀུན་ལ་གྲུབ་པ་ཤིན་ཏུ་དཀོན། །འདི་ན་འང་མྱ་ངན་འདས་མེད་མིན་། ་སྦྱོར་དང་གྲོལ་རྣམས་རྙེད་པར་དཀའ། །ལུས་ལ་ཡོན་ཏན་མེད་ཐོབ་ནས། །འདོད་ཆགས་ཡུན་རིང་མི་གནས་ཏེ། །ལམ་དེ་ཉིད་ཀྱིས་ཐམས་ཅད་ཀྱང་། །ཟད་པར་འགྱུར་བ་མ་ཡིན་ནམ། །
ཇི་ལྟར་ས་བོན་མཐའ་མཐོང་ཞིང་། །དེ་ལ་ཐོག་མ་ཡོད་མིན་ལྟར། །དེ་བཞིན་རྒྱུ་ནི་མ་ཚང་ཕྱིར། །སྐྱེ་བའང་འབྱུང་བར་མི་འགྱུར་རོ། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་སློབ་མ་སྦྱོང་བ་སྟེ་རབ་ཏུ་བྱེད་པ་བརྒྱད་པའོ།། །།ཐམས་ཅད་འབྲས་བུའི་དོན་སྐྱེ་བ། །དེས་ན་རྟག་ཡོད་མིན་དེའི་ཕྱིར། །ཐུབ་པ་མ་གཏོགས་ཇི་ལྟའི་དངོས། །དེ་བཞིན་གཤེགས་པ་ཡོད་མ་ཡིན། །གང་ཞིག་གང་ན་ནམ་དུ་ཡང་། །མ་བརྟེན་པར་ནི་ཡོད་ཉིད་མེད། །དེས་ན་ནམ་ཡང་གང་ཞིག་ཏུའང་དུ། །རྟག་པ་འགའ་ཡང་ཡོད་མ་ཡིན། །རྒྱུ་མེད་པར་ནི་དངོས་པོ་མེད། །རྒྱུ་ལྡན་རྟག་པ་ཡོད་མིན་པ། །དེས་ན་རྒྱུ་མེད་ལས་གྲུབ་ནི། །དེ་ཉིད་མཁྱེན་པས་གྲུབ་མིན་གསུངས། །མི་རྟག་བྱས་པར་མཐོང་གྱུར་ནས། །མ་བྱས་གལ་ཏེ་རྟག་ན་ནི། །བྱས་པ་ཡོད་པ་ཉིད་མཐོང་ནས། །རྟག་པ་ཡོད་ཉིད་མིན་པར་འགྱུར། །མཁའ་ལ་སོགས་རྣམས་རྟག་གོ་ཞེས། །སོ་སོའི་སྐྱེ་བོ་དག་གིས་རྟོགས། །མཁས་པ་རྣམས་ཀྱིས་དེ་དག་ལ། །འཇིག་རྟེན་པས་ཀྱང་དོན་མི་མཐོང་། །ཕྱོགས་ཞེས་བྱ་བ་ཕྱོགས་ཅན་ནི། །ཀུན་ལ་གནས་པ་མ་ཡིན་ཏེ། །དེའི་ཕྱིར་ཕྱོགས་ཅན་ལ་ཕྱོགས་ནི། །གཞན་ཡང་ཤིན་ཏུ་གསལ་བར་ཡོད། །གང་ཞིག་ཡོད་ན་དངོས་པོ་ལ། །འཇུག་དང་ལྡོག་པའང་དམིགས་པར་འགྱུར། །དེ་ནི་གཞན་གྱི་དབང་གྱུར་ཏེ། །དེས་ན་འབྲས་བུར་ཡང་འགྱུར་རོ། །འབྲས་བུ་མེད་པར་རྒྱུ་ལ་ནི། །རྒྱུ་ཉིད་ཡོད་པ་མ་ཡིན་ཏེ། །དེ་ཡི་ཕྱིར་ན་རྒྱུ་རྣམས་ཀུན། །འབྲས་བུ་ཉིད་དུ་ཐལ་བར་འགྱུར། །རྒྱུ་ནི་རྣམ་པར་འགྱུར་བ་ན། །གཞན་གྱི་རྒྱུ་རུ་འགྱུར་བ་སྟེ། །གང་ལ་རྣམ་པར་འགྱུར་ཡོད་པ། །དེ་ནི་རྟག་ཅེས་བྱར་ཡོད་མིན། །རྟག་པ་གང་གི་རྒྱུ་ཡིན་པའི། །དངོས་མེད་མ་བྱུང་བ་ལས་སྐྱེ། །རང་ཉིད་འབྱུང་བར་གྱུར་དེ་ལ། །རྒྱུ་ནི་ལྡོག་པར་འགྱུར་བ་སྟེ། །དངོས་པོ་རྟག་པ་ལས་སྐྱེས་པ། །ཅི་ལྟ་བུར་ན་མི་རྟག་འགྱུར། །ནམ་ཡང་རྒྱུ་དང་འབྲས་བུ་གཉིས། །མཚན་ཉིད་མི་མཐུན་མཐོང་མ་ཡིན། །གང་གི་ཕྱོགས་འགའ་རྒྱུ་ཡིན་ཞིང་། །ཕྱོགས་འགའ་རྒྱུ་མ་ཡིན་དེས་ན། །དེ་ནི་སྣ་ཚོགས་འགྱུར་ན་ཀོ། །སྣ་ཚོགས་རྟག་པར་ག་ལ་རིགས། །#རྒྱུ་ནི་(214) <«པེ་»«སྣར་»ག་ལ་རིགས། །
རྒྱུ་ནི་>ཟླུམ་པོ་གང་ཡིན་པ། །དེ་ནི་འབྲས་བུ་ལ་ཡོད་མིན། །དེས་ན་བདག་ཉིད་ཀུན་སྦྱོར་བ། །རྡུལ་ཕྲན་རཁང་་ལ་མི་འཐད་དོ། །རགྱིར་ཕྲན་གཅིག་གི་གནས་གང་ཡིན། །དེ་ནི་གཞན་གྱི་ཡང་མི་འདོད། །དེ་ཕྱིར་རྒྱུ་དང་འབྲས་བུ་གཉིས། །བོང་ཚོད་མཉམ་པར་འདོད་མ་ཡིན། །གང་ལ་ཤར་གྱི་ཕྱོགས་ཡོད་པ། །དེ་ལ་ཤརཔ།ྱི་ཆ་ཡང་ཡོད། །གང་གི་རྡུལ་ལ་ཕྱོགས་ཡོད་ན། །དེས་རྡུལ་རྡུལ་ཕྲན་མིན་པར་བསྙད། །མདུན་གྱིས་ལེན་ཅིང་རྒྱབ་ཀྱིས་ནི། །གཏོང་བར་གྱུར་ན་དེ་དག་གཉིས། །གང་ལ་ཡོད་པ་མ་ཡིན་པ། །དེ་ནི་འགྲོ་པོར་ཡང་མི་འགྱུར། །གང་ལ་དང་པོ་ཡོད་མིན་ཞིང་། །གང་ཞིག་ལ་དཀྱིལ་ཡོད་མིན་ལ། །གང་ལ་ཐ་མ་ཡོད་མིན་པ། ཡི་ན་མེད་དེ་ནི་གང་གིས་མཐོང་། །འབྲས་བུ་ཡིས་ནི་རྒྱུ་བཤིག་པ། །དེས་ན་རྒྱུ་ནི་རྟག་མ་ཡིནརྒྱུ་ཡོད་པ།་ན་རྒྱུ་ཡོད་པ། །དེ་ན་འབྲས་བུ་ཡོད་མ་ཡིན། །ཐོགས་ལྡན་དངོས་པོ་རྟག་པ་ནི། །གང་དུ་འང་སྣང་བ་མ་ཡིན་ཏེ། །དེས་ན་ནམ་ཡང་སངས་གསུངས།ྒྱས་རྣམས། །རྡུལ་ཕྲན་རྟག་པ་ཉིད་མི་གསུང་། །འཆིང་དང་བཅིངས་དང་ཐབས་ལས་གཞན། །ཐར་པ་གལ་ཏེ་ཡོད་ན་ནི། །དེ་ལས་ཅི་ཡང་མི་སྐྱེ་སྟེ། །དེས་ན་དེ་ཐར་ཞེས་མི་བརྗོད། །མྱ་ངན་འདས་ལ་ཕུང་པོ་རྣམས། །ཡོད་མིན་གང་ཟག་སྲིད་མ་ཡིན། །གང་དུ་མྱ་ངན་འདས་གྱུར་པ། །མ་མཐོང་དེར་མྱང་འདས་གང་ཞིག །སྲིད་དང་བྲལ་ལ་ཐར་པའི་ཚེ། །ཤེས་ཡོད་ཡོན་ཏན་ཅི་ཞིག་ཡོད། །ཤེསདེ་ད་པ་ཡི་ཡོད་པ་བདག་ཅིག་བར་ཡོད་པ་མིན་དང་མཚུངས། །ཐར་པར་བདག་གཅིག་ཡོད་ན་ནི། །ཤེས་ཡོད་ས་བོན་སྲིད་པ་ཡོད། །ཡིན་ན་ནི་སྲིད་པ་ལ། །བསམ་པའང་ཡོད་པ་མ་ཡིནི་ནོ། །སྡུག་བསྔལ་ལས་ཐར་མི་རྣམས་ལ། །གཞན་ལ་ཡོད་པ་མིན་པར་ངེས། །དེ་ཕྱིར་རྣམ་པ་ཀུན་དུ་འང་བདག །ཟད་པ་ལེགས་ཞེས་བྱ་བར་བརྗོད། །འཇིག་རྟེན་པ་འདི་ཉིད་བླ་ཡི། །དོན་དམ་རྣམ་ཀུན་མ་ཡིན་ཏེ། །འཇིག་རྟེན་པ་ལ་ཅུང་ཟད་ཡོད། །དམ་པའི་དོན་ལ་ཡོད་མ་ཡིན། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་དངོས་པོ་རྟག་པ་དགག་པ་བསྒོམ་པ་བསྟན་་སྟེ་རབ་ཏུ་བྱེད་པ་དགུ་པའོ།། །།གང་ཚེ་ནད་བདག་བུད་མེད་མིན། །སྐྱེས་མིན་མ་ནིང་མ་ཡིན་པ། །དེ་ཚེ་མི་ཤེས་འབའ་ཞིག་ལས། །ཁྱོད་བདག་ཕོའོ་སྙམ་དུ་སེམས། །
གང་ཚེ་འབྱུང་བ་ཐམས་ཅད་ལ། །ཕོ་མོ་མ་ནིབསྟེན་མིན་པ། །དེ་ཚེ་ཅི་སྟེ་དེ་དག་ཉིད། །བརྟེན་ནས་ཕོ་མོ་མ་ནིང་ཡིན།དག་ོད་ཀྱི་བདག་གང་ངའི་བདག་མིན། །དེས་དེ་བདག་མིན་མ་ངེས་ཕྱིར། །དངོས་པོ་མི་རྟག་པ་རྣམས་ལ། །རྟོག་པ་སྐྱེ་བར་འགྱུར་མིན་ནམ། །སྐྱེ་དང་སྐྱེ་བར་སྐྱེས་བུ་ནི། །ལུས་བཞིན་རྣམ་པར་འགྱུར་བར་འགྱུར། །ཁྱོད་ཀྱི་དེ་ལ་ལུས་ལས་གཞན། །ཉིད་དང་རྟག་ལ།་མི་རིགས་སོ། །རེག་དང་མི་ལྡན་དངོས་པོ་ལས། །བསྐུལ་བ་ཞེས་བྱ་མི་སྐྱེ་སྟེ། །དེ་ཕྱིར་ལུས་ཀྱི་གཡོ་བ་ལ། །སྲོག་ནི་བྱེད་པ་པོར་མི་འགྱུར། །མི་འཚེ་བ་དང་བདག་རྟག་པ། །འཏུ་ལ་རྒྱུ་ནི་ཅི་ཡོད་སྙམ། །རྣམ་པ་ཀུན་ཏུ་འང་ཤིང་སྲིན་ལས། །རྡོ་རྗེ་བསྲུང་བྱར་མི་འགྱུར་རོ། །སྐྱེ་བ་དྲན་པ་ཡོད་པའི་ཕྱིར། །གལ་ཏེ་ཁྱོད་ཀྱི་བདག་རྟག་ནབདག་ྔོན་བྱས་པ་ཡི་རྨ་མཐོང་ནས། །ཁྱོད་ཀྱི་ལུས་ཀོ་ཅིས་མི་རྟག །སེམས་པ་ཅན་དང་ལྡན་ཡི་ཀྱང་། །གལ་ཏེ་ཤེས་པོ་ཉིདཞེས་་ན། །དེ་ཡིས་སེམས་པ་ཅན་སེམས་པ། །མིན་ཞིང་སྐྱེས་བུའང་རྟག་མི་འགྱུར། །བདེ་བ་ལ་སོགས་དང་ལྡན་སྲོག །བདབདེ་་བཞིན་དབཞིན།ཏེ་ཚོགས་མཐོང་། །དེ་ཕྱིར་བདེ་བ་ལ་སོགས་བཞིན། །རྟག་པ་ཉིད་དུའང་མི་རུང་ངོ་། །གལ་ཏེ་ཤེས་པ་ཡོད་རྟག་ན། །བྱེད་འགྱུར་པར་འགྱུར་བ་སྟེ། །གལ་ཏེ་མེ་ནི་རྟག་འབྱུང་ན། །བུད་ཤིང་གིས་དོན་ཡོད་མི་འགྱུར། །རྫས་ནི་བྱ་བ་ཇི་བཞིན་དུ། །འཇིག་པའི་བར་དུ་གཡོ་བ་མེད། །དེས་ན་སྐྱེས་བུ་ཡོད་ཤེས་པ། །ཡོད་པ་མེད་ཅེས་བྱར་མི་རིགས། །གཞན་དུ་ཤེས་པ་ཡོད་པའི་ཁམས། །མཐོང་ཞིང་གཞན་དུ་ཤེས་ཡོད་དེ། །ལྕགས་ཀྱི་ཞུ་ཉིད་བཞིན་དེའི་ཕྱིར། །སྐྱེས་བུ་རྣམ་པར་འགྱུར་བར་འགྱུར། །ཤེས་ཡོད་ཡིད་ཙམ་ཞིག་ལ་སྟེ། །སྐྱེས་བུ་ནམ་མཁའ་བཞིན་དུ་ཆེ། །དེ་ཕྱིར་དེ་ཡི་ངོ་བོ་ནི། །ཤེས་པ་ཡོད་ཉིད་མིན་ལྟར་མཐོང་། །གལ་ཏེ་ཀུན་ལ་བདག་ཡོད་ན། །གན་གྱིས་དེ་ཅིས་ངར་མི་རྟོག །དེ་ཉིད་ཀྱིས་ནི་དེ་ཉིད་ལ། །སྒྲིབ་ཅེས་བྱ་བར་མི་འཐད་དོ། །གང་གི་ཡོན་ཏན་བྱེད་པོ་དང་། །རྣམ་ཀུན་ཤེས་ཡོད་མ་ཡིན་པ། །དེ་དག་དང་ནི་སྨྱོན་པ་ལ། །ཁྱད་པར་ཅི་ཡང་ཡོད་མ་ཡིན། །
ཡོན་ཏན་རྣམས་ཀྱིས་རྣམ་ཀུན་ཏུ། །ཁྱིམ་ལ་སོགས་དག་བྱེད་ཤེས་ཀྱི། །ཟ་བ་རྣམ་པར་མི་ཤེས་པ། །འདི་ལས་མི་རིགས་གཞན་ཅི་ཡོད། །བྱ་བ་དང་ལྡན་རྟག་པ་མེད། །ཀུན་ཏུ་སོང་ལ་བྱ་བ་མེད། །བྱ་བ་མེད་པ་མེད་དང་མཚུངས། །བདག་མེད་ལ་ཁྱོད་ཅིས་མི་དགའ། །ལ་ལར་ཀུན་ཏུ་སོང་བར་སྐྱེ་བུ་རྡུལ་ཙམ་ཞིག།་ལུས་ཙམ་ཞིག །ལ་ལར་རྡུལ་ཙམ་ཞིག་མཐོང་སྟེ། །ཤེས་རབ་ཅན་གྱིས་མེད་པར་མཐོངཔ་ །རྟག་ལ་གནོད་པ་ག་ལ་ཡོད། །གནོད་མེད་ཐར་ཡི་ག་ལཔ་ོད། །དེས་ན་གང་གི་བདག་རྟག་པ། །དེ་ལ་ཐར་པ་མི་རིགས་སོ། །གལ་ཏེ་བདག་ཅེས་བྱ་ཡོད་ན། །བདག་མེད་སྙམ་པ་མི་རིགས་ཤིགྱུར་ེ་ཉིད་རིག་པ་ངེས་པ་ལས། །མྱ་ངན་འམེད་གྱུར་ཞེས་བྱའང་བརྫུན། །གལ་ཏེཉིད་ལ་བ་ཡོད་ཉིད་ན། །དེ་ནི་སྔར་ཡང་མེད་པ་མེད། །མིབྱར་་པ་ལ་གང་མཐོང་བ། །དེ་ནི་རང་བཞིན་ཞེས་བྱར་བཤད། །གལ་ཏེ་མི་རྟག་ཆད་ན་ནི། །ད་དུང་རྩྭ་སོགས་ཅི་སྟེ་ཡོད། །འདི་ནི་གལ་ཏེ་བདེན་འགྱུར་ན། །སུ་ལའང་གཏི་མུག་ཀྱང་མི་འབྱུང་། །བདག་ནི་ཡོད་པར་གྱུར་ནའང་གཟུགས། །གཞན་ལས་འབྱུང་བར་སྣང་གྱུར་ལ།འཇིགས་པ་སྣང་གྱུར་་སྣང་གྱུར་ཞིང་། །གཞན་ལས་རྣམ་པར་འཇིག་པར་སྣང་། །ཇི་ལྟར་ས་བོན་བྱས་པ་ལས། །མྱུ་གུ་བྱས་པ་སྐྱེ་འགྱུར་བ། །དེ་བཞིན་མི་རྟག་ཐམས་ཅད་ནི། །མི་རྟག་དག་ལས་འབྱུང་བར་འགྱུར། །གང་ཕྱིར་དངོས་པོ་འཇུག་འགྱུར་བ། །དེས་ན་ཆད་པར་མི་འགྱུར་ཞིང་། །གང་ཕྱིར་དངོས་པོ་ལྡོག་འགྱུར་བ། །དེས་ན་རྟག་པར་མི་འགྱུར་རོ། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་བདག་དགག་པ་བསྒོམ་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅུ་པའོ།། །།མ་འོངས་བུམ་ལ་ད་ལྟར་བའི། །བུམ་ཡོད་མ་ཡིན་འདས་པ་མེད། །གང་ཕྱིར་དེ་གཉིས་མ་འོངས་པ། །དེ་ཕྱིར་མ་འོངས་ཡོད་མ་ཡིན། །གལ་ཏེ་ཞིག་པ་མ་འོངས་པའི། །ངོ་བོར་མ་འོངས་ལ་ཡོདཇི་ །མ་འོངས་བདག་ཉིད་དུ་ཡོད་གང་། །དེ་ནི་ཅི་ལྟར་འདས་པར་འགྱུར། །གང་གི་ཕྱིར་ན་མ་འོངས་དངོས། །མ་འོངས་བདག་ཏུ་གནས་འགྱུར་བ། །དེ་ཡི་ཕྱིར་ན་ད་ལྟར་བར། །འགྱུར་ཏེ་མ་འོངས་ཉིད་མི་སྲིད། །མ་འོངས་པ་ཡོད་འདས་པ་ཡོད། །
ད་ལྟར་བ་ཡོད་ཅི་ཞིག་མེད། །གང་གག་ུས་ཀུན་ཡོད་ཉིད་པ། །དེ་ཡི་མི་རྟག་ཉིད་གང་ལས། །འདས་པ་ལས་ནི་འདས་གྱུར་པ། །ཅི་ཡི་ཕྱིར་ན་འདས་པར་འགྱུར། །འདས་པ་ལས་ནི་མ་འདས་པ། །ཅི་ཡི་ཕྱིར་ན་འདས་པར་འགྱུར། །གལ་ཏེ་མ་འོངས་སྐྱེས་ཡོད་ན། །སྐྱེས་ར་ད་ལྟར་བར་མི་འགྱུར། །ཅི་སྟེ་དེ་ལ་སྐྱེ་མེད་ན། །མ་འོངས་རྟག་པར་འགྱུར་རམ་ཅི། །སྐྱེ་བ་མེད་ཀྱང་འཔ་ག་པ་ལས། །གལ་ཏེ་མ་འོངས་མི་རྟག་ན། །འདས་ལ་འཇིག་པ་ཡོད་མིན་ཏེ། །དེ་ནིབ།ྟག་པར་ཅིས་མི་རྟོག །འདས་པ་དང་ནི་ད་ལྟར་བ། །འདི་ནི་མི་རྟག་འགྱུར་མིན་ལ། །དེ་དག་ལས་གཞན་རྣམ་པ་ནི། །གསུམ་པའང་དེ་ལ་ཡོད་མ་ཡིན། །དངོས་པོ་གང་ཞིག་ཕྱིས་སྐྱེ་བ།ཕྱ་་ནི་གལ་ཏེ་སྔརཀྱིས་་ན། །དེ་ཡི་ཕྱིར་ན་ཕྱྭ་སྨྲ་བ། །རྣམས་ཀྱི་ཕྱོགས་ནི་ལོག་མི་འགྱུར། །གང་ཞིག་འབྱུང་བར་བྱེད་པ་སྟེ། །སྔར་ཡོད་ཅེས་བྱར་མི་རིགས་ཏེ། །གལ་ཏེ་ཡོད་པ་སྐྱེ་འགྱུར་ན། །སྐྱེས་ཟིན་པ་ཡང་འབྱུང་བར་འགྱུར། །མ་འོངས་པ་ཡི་དངོས་མཐོང་ན། །དངོས་པགྱུར་པ་ཅིས་མི་མཐོང་། །གང་ལ་མ་འོངས་ཡོད་འགྱུར་བ། །དེ་ལ་རིང་བ་ཡོད་མ་ཡིན། །གལ་ཏེ་བྱས་པ་མེད་ཀྱང་ཆོས། །ཡོད་ན་ངེས་སྡོམ་དོན་མེད་འགྱུར། །ཅི་སྟེ་ཅུང་ཟད་བྱེད་ན་ཡང་། །འཅི་་བུ་ཡོད་པ་མི་སྲིད་དོ། །མི་རྟག་ཡིན་ན་ཇི་ལྟ་བུར། །འབྲས་བུ་ཡོད་ཅེས་བྱ་བར་འགྱུར། །གང་ལ་ཐོག་མ་མཐའ་ཡོད་པ། །དེ་ནི་འཇཐར་རྟེན་མི་རྟག་བརྗོད། །བསྒྲིམ་པ་མེད་པར་ཐལ་འལྟར་ར་ཏེ། །གྲོལ་ལ་མ་འོངས་ཡོད་མ་ཡིན། །དེ་ལྟ་ཡིན་ན་ཆགས་མེད་པར། །འདོད་ཆགས་ཀྱང་ནི་འབྱུང་བར་འགྱུར། །འབྲས་བུ་ཡོད་ཉིད་གང་འདོད་དང་། །འབྲས་བུ་མེད་ཉིད་གང་འདོད་ལ། །ཁྱིམ་གྱི་དོན་དུ་ཀ་བ་ལ། །སོགས་པའི་རྒྱན་ནི་དོན་མེད་འགྱུར། །དངོས་རྣམས་ཡོངས་སུ་འགྱུར་བ་ཡང་། །ཡིད་ཀྱིས་ཀྱང་ནི་འཛིན་མི་འགྱུར། །དེ་ལྟ་ན་ཡང་ད་ལྟར་བ། །ཡོད་པར་མི་མཁས་རྣམས་ཀྱིས་རྟོག །གནས་མེད་དངོས་པོ་ག་ལཔོ་ །མི་རྟག་པས་གནས་ག་སྙིང་པོར་ལ་ཏེ་དང་པོར་གནས་གྱུར་ན། །ཐ་མར་རྙིངས་པར་མི་འགྱུར་རོ། །ཇི་ལྟར་རྣམ་ཤེས་གཅིག་གིས་ནི། །
དོན་གཉིས་རྣམ་པར་མི་ཤེས་པ། །དེ་བཞིན་རྣམ་ཤེས་གཉིས་ཀྱིས་ནི། །དོན་གཅིག་རྣམ་པར་མི་ཤེས་སོ། །གལ་ཏེ་དུས་ལ་གནས་ཡོད་ན། །གནས་པ་དུས་སུ་མི་འགྱུར་རོ། །ཅི་སྟེ་གནས་མེད་ན་གནས་པ། །མེད་པར་མཐའ་ཡང་ཡོད་མ་ཡིན། །གལ་ཏེ་མི་རྟག་དངོས་གཞན་ན། །དངོས་པོ་མི་རྟག་པར་མི་འགྱུར། །གཅིག་ན་མི་རྟག་ཉིད་གང་ཡིན། །དེ་ཉིད་དངོདེས།་ག་ལ་གནས། །མི་རྟག་ཉིད་གར་སྟོབས་ཆུང་དེར། །གནས་པ་སྟོབས་ཆུང་མ་ཡིན་ན། །ཕྱི་ནས་དེ་དག་ངེས་པར་གོ །བཟློག་པར་ཅི་ཡིས་མཐོང་བར་འགྱུར། །གལ་ཏེ་མི་རྟག་སྟོབས་ཆུང་མིན། །དངདང་།ོ་ཀུན་ལ་ཡོད་ན་ནི། །ཀུན་ལ་གནས་པ་མེད་པའམ། །ཡང་ན་ཐམས་ཅད་མི་རྟག་མིན། །གལ་ཏེ་རྟག་ཏུ་མི་རྟག་ཉིད། །ཡོད་ན་རྟག་ཏུ་གཕྱི་ནས་མིན། །ཡང་ན་རྟག་པར་གྱུར་ཟིན་ནས། །ཕྱི་ནས་མི་རྟག་པར་འགྱུར་རོ། །གལ་ཏེ་གནས་པ་མི་རྟག་དང་། །ལྷན་ཅིག་དངོས་པོ་ལ་ཡོད་ན། །མི་རྟག་ལོག་པར་འགྱུར་བའམ། །ཡང་ན་གནས་པ་བརྫུན་པར་འགྱུར། །མཐོངཀྱི་དངོས་པོ་མི་སྣང་ཞིང་། །སླར་ཡང་སེཡིས་སྐྱེ་མིན་དེས་ན། །དྲན་པ་ཞེས་བྱ་ལོག་པ་ཡི། །དོན་ལ་ལོག་པ་ཁོ་ན་འབྱུང་། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་དུས་དགག་པ་བསྒོམ་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅུ་གཅིག་པའོ།། །།གཟུར་གནས་བློ་ལྡན་དོན་གཉེཔོར་འི། །ཉནརྣམས་སྣོད་ཅེས་བྱ་བར་བཤད། །སྨྲ་པོ་ཡོན་ཏན་རྣམ་གཞན་དུ། །མི་འགྱུར་ཉན་པོ་ལ་ཡང་མིན། །སྲིད་དང་སྲིད་ཐབས་ཞི་བ་ཡི། །ཐབས་དང་དེ་བཞིན་ཞི་གསུངས་ཏེ། །འཇིག་རྟེན་ཡོངས་སུ་མི་ཤེས་གང་། །དེ་ནི་ཐུབ་པའི་ལྟ་བུར་མངོན། །ཐམས་ཅད་བཏང་བས་མྱ་ངན་ལས། །འདའ་བར་ཡ་མཚན་ཅན་ཀུན་འདོད། །ཀུན་སུན་འབྱིན་ལ་དེ་དག་ནི། །མི་དགར་འགྱུར་བ་རྒྱུ་ཅི་ཞིག །གང་ཞིག་གཏོང་ཐབས་མི་ཤེས་དེ། །ཅི་ཞིག་གཏོང་བར་བྱེད་པར་འགྱུར། །དེས་ན་ངེས་པར་ཐུབ་པ་ཡིས། །གཞན་དུ་ཞི་བ་མེད་ཅེས་གསུངས། །སངས་རྒྱས་ཀྱིས་གསུངས་ལྐོག་གྱཡི་། །གང་ཞིག་ཐེ་ཚོམ་སྐྱེ་འགྱུར་བ། །དེ་ཡིས་སྟོང་པ་ཉིད་བསྟེན་ཏེ། །འདི་ཉིད་ཁོ་ནར་དཀའ་ཆེས་བྱ། །གང་མ་ས་འཇིག་རྟེན་འདི་མཐོང་དཀའ། །དེ་ནི་གཞན་ལ་བླུན་པ་ཉིད།རིངས་དག་དེ་རྗེས་འགྲོ་དེ་དག །ཤིན་ཏུ་ཡུན་རིངས་བསླུས་པར་འགྱུར། །མྱ་ངདཀའ་ས་པར་རང་འགྲོ་གང་། །དེ་ནི་ཤིན་ཏུ་བྱ་དཀའ་བྱེད། །
འདྲེན་པའི་སླད་བཞིན་དམ་པ་ནི། །མ་ཡིན་ཡིད་འགྲོ་སྤྲོ་མ་ཡིན། །མ་མཐོང་སྐྲག་པ་མི་རྩོམ་སྟེ། །མཐོང་ན་རྣམ་ཀུན་དེ་ལྡོག་འགྱུར། བསྒྲུབ་་ངེས་པར་བྱིས་ད་ཅིག །ཤེཔ་ལ་སྐྲག་པ་བརྗོད་པར་བྱ། །བྱིས་རྣམས་ངེས་པ་ཁོ་ནར་ནི། །འཇུག་བྱེདཡིསས་ལ་གོམས་པ་སྟེ། །དེའཇིགས་མས་པ་མེད་པ་ཡིས། །ལྡོག་བྱེདའགའ་་ལ་འཇིགས་པར་འགྱུར། །གང་ཞིག་གཏི་མུག་འབའ་ཞིག་གིས། །བསྒྲིབས་ཤིང་དེ་ཉིད་གེགས་བྱེད་པསྨོས་ཅི་དགོས་ལ་དགེ་ལེགས་འགྲོ་བ་ཡང་། །མེད་ན་ཐར་པ་ག་ལ་ཡོད། །ཚུལ་ཁྲིམས་ལས་ནི་ཉམས་བླ་ཡི། །ལྟ་ལས་ཅིས་ཀྱང་མ་ཡིན་ཏེ། །ཚུལ་ཁྲིམས་ཀྱིས་ནི་མཐོ་རིས་འགྲོ། །ལྟ་བས་གོ་འཕང་མཆོག་ཏུ་འགྱསྟོང་དམ་པ་མིན་ལ་བདག་འཛིན་མཆོག །བདག་མེད་སྟོནབ།་མ་ཡིན་ཏེ། །གཅིག་ནི་ངན་འགྲོ་ཉིད་འགྲོ་ལ། །ཐ་མལ་མ་ཡིན་ཞི་ཉིད་དུའོ། །ཞི་སྒོ་གཉིས་པ་མེད་པ་དང་། །ལྟ་བ་ངན་རྣམས་འཇིག་བྱེད་ཅིང་། །སངས་རྒྱས་ཀུན་གྱི་ཡུལ་འགྱུར་ལ། །བདག་མེད་ཅེས་ནི་བྱ་བར་བརྗོད། །འཇིག་ི་ཡི་ནི་མིང་ལས་ཀྱང་། །དམ་བསྐྱེད་པ་འཇིགས་པ་སྐྱེ། །ཞན་ལ་འཇིགས་པ་མི་སྐྱེད་པའི། །སྟོབས་ལྡན་ཞེས་བྱ་གང་ཀྱི་ཐོང་། །ཆོས་འདི་དེ་བཞིན་གཤེགས་རྣམས་ཀྱིས།འདི་ད་པའི་ཆེད་དུ་མ་གསུབསྲེགས་ཏེ་མེ་ཡི་འདིས་ནི་གཞན་སྨྲ་རྣམས། །སྲེག་སྟེ་མེ་ཡིས་བུད་ཤིང་བཞིན། །ཆོས་འདི་གང་གིས་ཤེས་གྱུར་པ། །དེ་ནི་གཞན་ལའཇིག་འ་སྟེ། །དེས་ན་བདག་ལ་ཆོས་འདི་ནི། །འཇིགས་པའི་སྒོ་དང་འདྲ་བར་སྣང་། །དེ་ཉིད་དུ་བདག་མེད་སྙམ་དུ། །གང་་ལྟར་གང་ལ་དགོངས་གནས་པ། །དེ་ནི་ཡོད་པས་ག་ལ་དགའ། །མེད་པས་འཇིགས་པར་ག་ལ་འགྱུར། །དོན་མིན་ས་བོན་གྱུར་པ་ཡི། །མུ་སྟེགས་ཅན་མང་མཐོང་ནས་ནི། །ཆོས་འདོད་པ་ཡི་སྐྱེ་བོ་ལ། །སུ་ཞིག་སྙིང་བརྩེ་སྐྱེ་མི་རྣམས་ཀྱི་ཡང་གོས་མེད་བྲམ་ཟེ་སྟེ། །གསུམ་པོ་རྣམས་ཀྱིས་ཆོས་ཡིད་དང་། །མིག་དང་རྣ་བ་ཡིས་འཛིན་པ། །དེ་ཕྱིར་ཐུབ་པའི་གཞུང་ལུགས་སྨྲ། །ཇི་ལྟར་བྲམ་ཟེ་རྣམས་ལ་ཆོས། །ཕལ་ཆེར་ཕྱི་འཆོས་བརྗོད་པ་ལྟར། །དེ་བཞིན་གཅེར་བུ་རྣམས་ལ་ཆོརིག་ལ་ཆེར་བླུན་པ་བརྗོད་ཟེ་ན། །ཇི་ལྟར་རིགས་པ་བླངས་པ་ལས། །
བྲམ་ཟེར་གུས་པ་སྐྱེ་བ་ལྟར། །དེ་བཞིན་ཉོན་མོངས་བླངས་པ་ལས། །གཅེར་བུ་པ་ལ་བརྩེ་བར་འགྱུར། །སྡུག་བསྔལ་ལས་ཀྱི་རྣམ་སྨིན་ཕྱིར། །ཇི་ལྟར་ཆོས་སུ་མི་འགྱུར་བ། །དེ་བཞིན་སྐྱེ་བ་ལས་ཀྱི་ནི། །རྣམ་སྨིན་ཡིན་ཕྱིར་ཆོས་མ་ཡིན། །ཆོས་ནི་མདོར་ན་མི་འཚེ་བར། །དེ་བཞིན་གཤེགས་པར་རྣམས་ཀྱིས་གསུངས། །སྟོང་ཉིད་མྱ་ངན་འདས་པ་སྟེ། །འདིར་ནི་དེ་གཉིས་འབའ་ཞིག་གོ། །རང་ཕྱོགས་འཇིག་རྟེན་ཐམས་ཅད་ལ། །སྐྱེས་ས་བཞིན་དུ་སྡུག་འགྱུར་ན། །དེ་ཡི་ལྡོགབསྔལ་་བྱེད་པའི་རྒྱུ། །ཁྱོད་ལ་གང་གིས་སྡུག་པར་འགྱུར། །རིགས་པའི་དོན་ནི་གཞན་ལས་ཀྱང་།སྟེང་ལྡན་ལེགས་པ་འདོད་པས་བླང་། །ཉི་མ་ས་སྟེངས་མིག་ལྡན་ལ། །ཀུན་གྱི་སྤྱི་ཐུན་མ་ཡིན་ནམ། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་ལྟ་བ་དགག་པ་བསྒོམ་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅུ་གཉིས་པའོ།། །།གཟུགས་མཐོང་ཚེ་ན་བུམ་པ་ནི། །ཐམས་ཅད་ཁོ་ན་མཐོང་མི་འགྱུར། །བུམ་པ་མངོན་སུམ་ཞེས་བྱ་བའང་། །དེ་ཉིད་རིག་པ་སུ་ཞིག་སྨྲ། །རྣམ་པར་དཔྱད་པ་འདི་ཉིད་ཀྱིས། །བློ་མཆོག་ལྡན་པས་དྲི་ཞིམ་དང་། །མངར་དང་འཇམ་པ་ཐམས་ཅད་དག །སོ་སོར་དགག་པར་བྱ་བ་ཡིན། །གལ་ཏེ་གཟུགས་མཐོང་བ་ཡིས་དེ།ཡིས་ས་ཅད་མཐོང་བར་འགྱུར་ན་ནི། །མ་མཐོང་བ་ཡིས་ཅིས་མི་འགྱུར། །གཟུགས་ནི་འབའ་ཞིག་ཁོ་ན་ལ། །མངོན་སུམ་ཉིད་ནི་ཡོད་མ་ཡིན། །གང་ཕྱིར་དེ་ལ་ཕ་རོལ་གྱི། །ཆ་དང་ཚུ་རོལ་དབུས་ཡོད་ཕྱིར། །རྡུལ་ལ་ཆ་ཤས་ཡོད་མེད་ཅེས། །དཔྱད་པ་དེར་ཡང་འཇུག་པར་འགྱུར། །དེ་ཕྱིཔར་སྒྲུབ་པར་བྱ་བ་ཡིས། །བསྒྲུབ་བྱ་འགྲུབ་པར་མི་འཐད་དོ། །ཀུན་ཀྱང་ཡན་ལག་ཏུ་གྱུར་ནས། །སླར་ཡང་ཡན་ལག་ཅན་དུ་འན་ུར། །དེ་ཕྱིར་ཡི་གེ་བརྗོད་པ་ཡང་། །འདི་ལ་ཡོད་པ་མ་ཡིན་ནོ། །གལ་ཏེ་མདོག་ལས་དབྱིབས་གཞན་ན། །ཇི་ལྟ་བུར་ན་དབྱིབས་འཛིན་འགྱུར། །འོན་ཏེ་གཞན་མིན་ན་ལུས་ཀྱིས། །མདབལྟ་ཀྱང་ཅི་སྟེ་འཛིན་མི་འགྱུར། །གཟུགས་ནི་ལྟ་ཞིག་མ་གཏོགས་པར། །གཟུགས་ཀྱི་རྒྱུ་ནི་མི་སྣང་ངོ་། །གལ་ཏེ་དེ་ལྟར་འགྱུར་ན་ནི། །གཉི་ག་ཡང་ནི་མིག་ཉིད་ཀྱིསབསྟན་ཛིན་པར་ཅི་ཡི་ཕྱིར་མི་འགྱུར། །ས་ནི་བརྟན་ཞེས་བྱ་བར་མཐོང་། །དེ་ཡང་ལུས་ཀྱིས་འཛིན་པར་འགྱུར། །དེས་ན་རག་པ་འབའ་བྱ་ནི། །ས་འོ་ཞེས་ནི་བྱ་བར་བརྗོད། །བལའགར་ར་སྐྱེས་པས་བུམ་པ་ནི། །འདི་ལ་ཡོན་ཏན་འགར་མི་འགྱུར། །དེས་ན་བལྟ་བྱ་ཉིད་སྐྱེ་ལྟར། །ཡོའགྱུར་་ངོ་བོའང་ཡོད་མ་ཡིན། །མིག་ནགིས་བྱུང་གྱུར་སྣ་དེ་བཞིན། །མིག་གིས་མཐོང་གི་གཞན་གྱིས་མིན། །དེས་ན་ངེས་པར་ལས་རྣམ་སྨིན། རྟག་ཏུ་ཡིད་མེད་ཅི་སྟེ་མིན། །སྔར་མཐོང་བ་ཡི་དོན་གང་ཞིག །ཡིད་ཀྱིས་སྨིག་རྒྱུ་ལྟར་འཛིན་པ། །དེ་ནི་ཆོས་ཀུན་རྣམ་གཞག་ལ། །འདུ་ཤེས་ཕུང་པོ་ཞེས་བྱའོ། །མིག་དང་གཟུགས་ལ་བརྟེན་ནས་ཡིད། །སྒྱུ་མ་བཞིན་དུ་སྐྱེ་བར་འགྱུར། །གང་ལ་ཡོད་པ་ཉིད་ཡོད་དེ། ལ་ས་སྟེང་ས་བྱར་མི་རིཅན་མིན་ཅང་་ཚེ་མཁས་པ་ས་སྟེངས་ན། །ངོ་མཚར་མིན་པ་འགའ་མེད་པ། །དེ་ཚེ་དབང་རྟོགས་དེ་འདྲ་ལ། །ཡ་མཚན་ཞེས་བྱ་ཅི་ཞིག་ཡོད། །མགལ་མེའི་འཁོར་ལོ་སྤྲུལ་པ་དང་། །རྨི་ལམ་སྒྱུ་མ་ཆུ་ཟླ་དང་། །ཁུག་རྣ་ནང་གི་བྲག་ཅ་དང་། །སྨིག་རྒྱུ་སྤྲིན་དང་སྲིད་པ་མཚུངས། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་དབང་པོ་དང་དོན་དགག་པ་བསྒོམ་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅུ་གསུམ་པའོ།། །།དངོས་པོ་གང་ཞིག་གཞན་འགའ་ལའང་། །རག་ལས་པར་ནི་མི་འགྱུར་ན། །དེ་ཡི་བདག་ཉིད་འགྲུབ་འགྱུར་ན། །དེ་ནི་གང་ནའང་ཡོད་མ་ཡིན། །གཟུགས་ཉིད་བུམ་ཞེས་གཅིག་མ་ཡིན། །གཟུགས་ལྡན་བུམ་གཞན་ཡོད་མིན་ལ། །བུམ་པ་ལ་གཟུགས་ཡོད་མིན་ཞིང་། །གཟུགས་ལ་བུམ་པ་ཡོད་མ་ཡིན། །གཉིས་མཚན་མི་མཐུན་མཐོང་ནས་ནི། །གལ་ཏེ་དངོས་ལས་བུམ་གཞན་ན། །མ་ཡིན་དེ་ལྟར་དངོས་པོ་ཡང་། །བུམ་པ་ལས་གཞན་ཅིས་མི་འགྱུར། །གཏུ་ེ་གཅིག་བུམ་མི་འདོད་ན། །བུམ་པའང་གཅིག་པུ་མི་འགྱུར་རོ། །ལྡན་འདི་མཚུངས་པའང་མ་ཡིན་པ། །དེས་ཀྱང་གཅིག་ཏུ་མི་འགྱུར་རོ། །གང་ཚེ་རྫས་ཅི་ཙམ་གཟུགས་ན། །དེ་ཚེ་གཟུགས་ཆེན་ཅི་སྟེ་མིན། །གལ་ཏེ་ཕྱིར་རྒོལ་གཞན་མིན་ན། །གཞུང་ལུགས་བསྙད་པར་བྱ་བར་འགྱུར། །མཚན་ཉིད་ཀྱིས་ཀྱངདེར་་གཞི་ནི། །གང་དུ་གྲུབ་པ་ཡོད་མིན་པ། །དེར་ནི་གྲངས་སོགས་ཐ་དད་པར། །དངོས་པོ་ཡོད་པ་མ་ཡིན་ནོ། །མཚན་ཉིད་རྣམས་དང་སོ་སོ་ནི། །མིན་ཕྱིར་བུམ་པ་གཅིག་མ་ཡིན། །རེ་རེའི་བུམ་པ་མེད་ན་ནི། །མང་ཉིད་འཐད་པར་མི་འགྱུར་རོ། །རེག་ལྡན་རེག་དང་མི་ལྡན་དང་། །ལྷན་རྣམས་སྦྱོར་བ་ནི།་མེད། །དེ་ཕྱིར་གཟུགས་སོགས་རྣམས་སྦྱོར་བ། །རྣམ་པ་ཀུན་ཏུ་རིགས་མ་ཡིན། །གཟུགས་ནི་བུམ་པའི་ཡན་ལག་སྟེ། །དེས་ན་རེ་ཞིག་དེ་བུམ་མིན། །གང་ཕྱིར་ཡན་ལག་ཅན་མེད་པ། །
དེས་ན་ཡན་ལག་ཀྱང་ཡོད་མིན། །གཟུགས་རྣམས་ཀུན་ལ་གཟུགས་ཉིད་ནི། །མཚན་ཉིད་མི་མཐུན་མ་ཡིན་ན། །གཅིག་ལ་བུམ་པ་ཡོད་འགྱུར་ཞིང་། །གཞན་དག་ལ་མིན་རྒྱུ་ཅི་ཞིག །རོ་སོགས་དག་ལས་གཟུགས་གཞན་ཞིང་།མ་གཏོགས་ས་མིན་ཞེས་ཁྱོད་འདོད་ན། །དེ་རྣམས་མེད་པར་རང་མེད་གང་། །དེ་གཟུགས་ལས་གཞན་ཇི་ལྟར་མིན། །བུམ་པ་ལ་ནི་རྒྱུ་ཡོད་མིན། །རང་ཉིད་འབྲས་བུར་མི་འགྱུར་ཏེ། །དེ་ཕྱིར་གཟུགས་སོགས་ལས་གཞན་དུ། །འགྲུབ་་འགའ་ཡང་ཡོད་མ་ཡིན། །བུམ་པ་རྒྱུ་ལསགྱུར་འགྱུར་ཞིང་། །རྒྱུ་ནི་གཞན་ལས་གྲུབ་འགྱུར་ན། །གང་ལ་རབསྐྱེད།ྲུབ་མེད་པ། །དེ་ཡིས་གཞན་པོ་ཇི་ལྟར་སྐྱེད། །འཕྲོད་པ་འདུས་པར་གྱུར་ཀྱང་གཟུགས། །དྲི་ཉིད་དུ་ནི་མི་འཐད་དེ། །དེས་ན་བུམ་པ་བཞིན་ཚོགས་པ། །གཅིག་ཉིད་དུ་ནི་མི་རིགས་སོ། །ཇི་ལྟར་གཟུགས་སོགས་མ་གཏོགས་པར། །བུམ་པ་ཡོད་པ་མ་ཡིན་པ། །དེ་བཞིན་རླུང་ལ་སོགས་པ་ནི། །མ་གཏོགས་གཟུགས་ཀྱང་ཡོད་མ་ཡིན། །ཚ་བ་མེ་ཉིད་དུ་འགྱུར་ཏེ། །ཚ་བ་མིན་པ་ཇི་ལྟར་བསྲེག །དེས་ན་བུད་ཤིང་ཞེས་བྱ་བ། །ཡོད་མིན་དེ་མེད་མེ་ཡོད་མིན། །གལ་ཏེ་ཟིལ་མནན་ཚ་བ་ཡང་། །དེ་ཡང་ཅི་སྟེ་མམེ་མི་འགྱུར། །འོན་ཏེ་མི་ཚ་ན་དངོས་གཞན། །དེ་ལ་ཡོད་ཅེས་བྱར་མི་རུང་། །གལ་ཏེ་རྡུལ་ལ་ཤིང་མེད་ལའང་ །དེས་ན་ཤིང་མེད་མེ་ཡོད་དོ། །གལ་ཏེ་དེ་ལ་ཤིང་ཡོད་ན། །གཅིག་གི་བདབརྟག་རྡུལ་ཡོད་མིན། །དངོས་པོ་གང་གང་ཡོངས་བཏགས་པ། །དེ་དང་དེ་ལ་གཅིག་ཉིད་མེད། །གང་གིས་གཅིག་ཀྱང་ཡོད་མིལ་པ། །དེས་ན་དུ་མ་དག་ཀདེར་ེད། །འདོད་ེ་གང་ན་དངོས་དེ་རྣམས། །མེད་དེར་གཅིག་ཡོད་ཅེས་འདོད་ན། །གང་གིས་ཀུན་ལ་གསུམ་ཉིད་ཡོད། །དེས་ན་གཅིག་ཉིད་ཡོད་མ་ཡིན། །ཡོད་དང་མེད་དང་ཡོད་མེད་དང་། །ཡོད་མེད་མིན་ཞེས་བྱ་བའི་ཚུལ། །འདི་ནི་མཁས་པས་གཅིག་ཉིད་སོགས། །དལྟ་རྟག་ཏུ་སྦྱར་བར་བྱ། །ཇི་ལྟར་རྒྱུནན།་བལྟ་ཉེས་ན། །རྟག་ལྟ་ཉེས་ན་བྱར་འགྱུར་བ། །དེ་བཞིན་ཚོགས་ལ་བལྟ་ཉེས་པ། །དངོས་པོ་ཡོད་ཅེས་བྱ་བར་འགྱུར། །གང་ལ་བརྟེན་ནས་འབྱུང་ཡོད་པ། །དེ་ནི་རང་དབང་མི་འགྱུར་རོ། །འདི་ཀུན་རང་དབང་མེད་པ་སྟེ། །
དེས་ན་བདག་ནི་ཡོད་མ་ཡིན། །འབྲས་བུ་མེད་པར་དངོས་པོ་ནི། །ཀུན་ལ་འདུ་བ་ཡོད་མ་ཡིན། །འབྲས་བུའི་དོན་དུ་འདུ་བ་གང་། །དེ་ནི་འཕགས་ལམ་འདུས་པའོ། །སྲིད་པའི་ས་བོན་རྣམ་ཤེས་ཏེ། །ཡུལ་རྣམས་དའདོད་ནས་སྐྱེས་ི་སྤྱོད་ཡུལ་ལོ། །ཡུལ་ལ་བདག་མེད་མཐོང་ནས་འགྱུར་བ། །དེས་ན་ཡོད་པ་ག་ལ་སྐྱེ། །འབྲས་བུ་ཡིས་ནི་རྒྱུ་བཤིག་པ། གྲུབ་་མེད་པ་སྐྱེ་མི་འགྱུར། །གཔའང་བ་པ་ལས་སྒྲུབ་ཡོད་མ་ཡིན། །དེ་ཕྱིར་ཡོད་པ་མི་སྐྱེའོ། །དེ་ཚེ་སྐྱེ་བ་ཡོད་མིན་ཞིང་། །གཞན་ཚེ་སྐྱེ་བ་ཡོད་མ་ཡིན། །དེ་ཚེ་གཞན་ཚེ་མི་སྐྱེ་ན། །ནམ་ཞིག་སྐྱེ་བ་ཡོད་པར་འགྱུར། །དེ་ལ་དེ་ཡི་དངོས་པོ་རུ། །ཇི་ལྟར་སྐྱེ་བ་ཡོད་མིན་ལྟར། །དེ་བཞིན་གཞན་གྱི་དངོས་པོར་ཡང་། །དེ་ལ་སྐྱེ་བ་ཡོད་མ་ཡིན། །ཐོག་མ་བར་དང་ཐ་མ་རྣམས། །སྐྱེ་བའི་སྔ་རོལ་སྲིད་མ་ཡིན། །གཉིས་གཉིས་དག་ནི་མེད་པ་རུ། །ཇི་ལྟར་རེ་རེ་རྩོམ་པར་འགྱུར། །གཞན་གྱི་དངོས་པོ་མེད་པར་ནི། །བདག་གི་དངོས་པོ་འགཉིས་ཀ་བ་མེད། །དཡོད་ྱིར་ན་བདག་གཞན་ནི། །གཉིས་ལས་འབྱུང་བ་ཡོད་མ་ཡིན། །སྔོན་དང་ཕྱིས་དང་ཅིག་ཅར་ཅེས། །བརྗོད་པ་ཉིད་དུ་མི་ནུས་པ། །དེ་ཕྱིར་སྐྱེ་དང་བུམ་པ་ལ། །ཅིག་ཅར་འབྱུང་བ་ཡོད་མ་ཡིན།ཪྙིང་སྐྱེས་པ་ཡི་ཕྱིར་ན་ནི། །སྔར་སྐྱེས་རྙིངས་པར་མི་འགྱུར་ཞིང་། །ཕྱི་ནས་ཀུན་ཏུ་སྐྱེས་པ་ཡང་། །ཕྱི་ནས་སྐྱེས་པར་མི་འགྱུར་རོ། །ད་ལྟར་བ་ཡི་དངོས་པོ་ནི། །དེ་ཉིད་ལས་འབྱུང་མ་ཡིན་ཞིང་། །མ་འོངས་ལས་འབྱུང་ཡོད་མིན་ལ། །འདས་པ་ལས་ཀྱང་ཡོད་མ་ཅི་།སྐྱེས་པ་ལ་ནི་འོང་བ་དང་། །དེ་བཞིན་འགགས་ལྟ་བུར། །སྲིད་པ་སྒྱུ་མ་འདྲ་མ་ཡིན། །སྐྱེ་དང་གནས་དང་འཇིག་པརིམ་གྱིས་ིག་ཅར་འབྱུང་བ་ཡོད་མིན་ཞིང་། །རིམས་ཀྱིས་འབྱུང་བའང་ཡོད་མིན་ལ་བ་ །ནམ་ཐམས་ྱུང་བ་ཡོད་པར་འགྱུར། །སྐྱེ་ལ་སོགས་པ་ཐམས་ཅད་ལ། །སླར་ཡང་ཐམས་ཅད་འབྱུང་བ་སྟེ། །དེ་ཕྱིར་འཇིག་པ་སྐྱེ་འདྲ་དང་མཚན་མེད་པ་འཇིག་དང་འདྲ་བར་སྣང་། །མཚན་ལསག་ན་གཞི་གཞན་ཞེ་ན། །མཚན་གཞི་མི་རྟག་ཉིད་གང་ལས། །ཡང་ན་བཞི་ཀའང་ཡོད་པ་ཡི། །ངོ་བོར་གསལ་བར་ཡོད་མ་ཡིན། །དངོས་པོ་དངོས་ལས་མི་སྐྱེ་སྟེ། །དངོས་པོ་དངོས་མེད་ལས་མི་སྐྱེ། །དངོས་མེད་དངོས་མེད་མི་སྐྱེ་སྟེ། །དངོས་མེད་དངོས་ལས་མི་སྐྱེའོ། །དངོས་པོ་དངོས་པོར་མི་འགྱུར་ཏེ། །དངོས་མེད་དངོས་པོར་མི་འགྱུར་རོ། །དངོས་མེད་དངོས་མེད་མི་འགྱུར་ཏེ། །དངོས་པོཕྱེད་་མེད་མི་འགྱུར་རོ། །སྐྱེ་བཞིན་པ་ནི་བྱེད་སྐྱེས་ཕྱིར། །སྐྱེ་བཞིན་པ་ནི་སྐྱེ་མི་འགྱུར། །ཡང་ན་ཐམས་ཅད་སྐྱེ་བཞིན་པ། །ཉིད་ནི་ཡིན་པར་ཐལ་བར་འགྱུར། འབྲས་བུ་ཡོད་པ་མ་ཡིན་པ། །དེ་ཡི་ཚེ་ན་འཇུག་པ་དང་། །ལྡོག་པ་འཐད་པ་ཉིད་མི་འགྱུར། །རྣདོན་དམ་དུ་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་འདུས་བྱས་ཀྱི་དོན་དགག་པ་བསྒོམ་པ་བསྟན་པ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅོ་ལྔ་པའོ།། །།རྒྱུ་འགའ་ཞིག་གིས་སྟོང་པ་ཡང་། །སྟོང་པ་མཀྱི། །དེ་་སྣང་། །རབ་ཏུ་བྱེད་པ་ཐམས་ཅད་ཀྱིས། །འདི་ནི་སོ་སོར་འགོག་པར་བྱེད། །གང་ཚེ་སྨྲ་པོ་བརྗོད་བྱ་ནའངབརྟེན་ྟོང་པ་ཞེས་བྱར་མི་རུང་ངོ་། །གང་ཞིག་རྟེན་ནས་ཀུན་འབྱུང་བ། །དེ་ནི་གསུམ་ལའང་ཡོད་མ་ཡིན། །གལ་ཏེ་སྟོང་པའི་ཉེས་པ་ཡིས། །སྟོང་པ་མ་ཡིན་འགྲུབ་འགྱུར་ན། །མི་སྟོང་ཉེས་པས་སྟོང་པ་ཉིད། །ཅི་སྟེ་འགྲུབ་པར་འགྱུར་མ་ཡིན། །གཞན་གྱི་ཕྱོགས་ནི་བཟློག་པ་དང་། །རང་གི་ཕྱོགས་ནི་བསྒཅིག །གང་ །གལ་ཏེ་སུན་འབྱིན་བྱེད་ཕྱོགས་གཅིག །དགའ་ན་སྒྲུབ་བྱེད་ལ་ཅིས་མིན། །གལ་ཏེ་ཡོངས་སུ་བརྟགས་པ་ན། །མེད་གང་དེ་ཕྱོགས་མི་འགྱུར་ན། །དེས་ན་གཅིག་ཉིད་ལ་སོགས་གསུམ། །ཀུན་ཀྱང་ཕྱོགས་མ་ཡིན་པར་འགྱུར། །གང་དུ་མངོན་སུམ་གྱིས་བུམ་དེར། །སགཞན་འི་གཏན་ཚིགས་དོན་མེད་འགྱུར། །འདི་ན་གཞུང་ལུགས་ལས་བྱུང་བའི། །གཏན་ཚིགས་ཡོད་མིན་གཞན་དུ་བྱུང་མི་སྟོང་མེད་པར་སྟོང་པ་ནི། །གང་ལས་འབྱུང་བ་ཉིད་དུ་འགྱུར། །ཇི་ལྟ་བུར་ན་ཅིག་ཤོས་ནི། །མེད་པར་གཉེན་པོ་འབྱུང་བར་འགྱུར། །གལ་ཏེ་ཕྱོགས་ཉིད་དུ་འགྱུར་ན། །ཕྱོགས་མེད་ཕྱོགས་ཀྱི་ངོ་བོར་འགྱུར། །དེ་ལྟར་ཕྱོགས་མེད་ཉིད་མེད་ན། །གང་ཞིག་མི་མཐུན་ཕྱོགས་སུ་འགྱུར།ཇི་ལ་ཏེ་དངོས་པོ་ཡོད་མིན་ན། །མེ་ནི་ཚ་བར་ཅི་ལྟར་འགྱུར། །ཚ་བ་མེ་ཡང་ཡོད་མིན་ཞེས། །གོང་ཉིད་དུ་ནི་བཟློག་པར་བྱས། །དངོས་པོ་མཐོང་ནས་དངོས་པོ་ནི། །མེད་པ་ཞེས་བྱ་བཟློག་འགྱུར་ན། །དེ་ལྟར་ཕྱོགས་ནི་བཞི་ཆར་ལ། །ཉེས་པ་སྤངས་པ་གང་ཞིག་མཐོང་། །རྡུལ་ཕྲན་ལ་ཡང་ཡོད་པའི་དངོས། །གང་དུ་མེད་དེར་ཇི་ལྟར་འབྱུང་། །སངས་རྒྱས་རྣམས་ལ་མེད་ཉིད་ཀྱང་། །དེ་ཉིད་ཕྱིར་ན་མ་འབྲེལ་པའོ། །གལ་ཏེ་ཀུན་ལ་གཉིས་མིན་ན། །གཞན་གང་ཞིག་ལ་ཡོད་དངོས་ཡོད། །གལ་ཏེ་ཁྱོད་ལའང་དེ་རིགས་ན། །
བརྒལ་བ་གཞན་པོ་ཅི་སྟེ་བྱེད། །དངོས་པོ་ཀུན་གྱི་དངོས་མེད་ལ། །རྣམ་པར་དབྱེ་བ་མི་རིགས་ཏེ། །རྫས་རྣམས་ཀུན་ལ་གང་མཐོང་བ། །དེ་ནི་རྣམ་དབྱེར་མི་འགྱུར་རོ། །མེད་ཕྱིར་གཞན་གྱི་ཕྱོགས་ལ་ལན། །བརྗོད་པ་མིན་པར་གྲགས་ན་ནི། །གཏན་ཚིགས་ཀྱིས་བཟློག་རང་ཕྱོགས་ཀྱང་། །ཅི་ཡི་ཕྱིར་ན་སྒྲུབ་མི་འགྱུབླ་།སུན་འབྱིན་བྱེད་པའི་གཏན་ཚིགས་རྙེད། །སླ་ཞེས་འཇིག་རྟེན་བརྗོད་ན་ཡང་། །གཞན་གྱི་ཕྱོགས་ལ་སྐྱོན་བརྗོད་པ། །ཉིད་དུ་ཅི་ཡི་ཕྱིར་མི་ནུས། །ཡོད་ཅེས་བྱ་བ་ཙམ་ཞིག་གིས། །གལ་ཏེ་དེ་ཉིད་དངོས་ཡོད་ན། །མེད་ཅེས་བྱ་བ་ཙམ་ཞིག་གིས། །མེད་པར་ཡང་ནི་ཅིས་མི་འགྱུར། །གང་ཕྱིར་ཡོད་བདོག་ཅེས་མིང་བཏགས། །དེ་ཕྱམི་ེད་པར་མི་འགྱུར་ན། །གང་ཕྱིར་ཡོད་ཅེས་མིང་བྱས་པ། །དེ་ཕྱིར་ཡོད་པར་མི་འགྱུར་རོ། །གལ་ཏེ་ཐམས་ཅད་འཇིག་རྟེན་པས། །བརྗོད་པས་འཇིག་རྟེན་པ་ཡོད་ན། །དངོས་གང་དེ་ཉིད་དུ་ཡོད་པ། །དེ་ནི་འཇིག་རྟེན་པར་ཅིས་འགྱུར། །གལ་ཏའགྱུར་་ཀུན་མེད་པའི་ཕྱིར། །མེད་པ་ཉིད་དུ་འགྱུར་ན་ནི། །དེ་ལྟ་ཡིན་དང་ཕྱོགས་རྣམས་ཀུན། །དངོས་པོ་མེད་པར་མི་རིགས་སོ། །དངོས་པོ་མེད་ཕྱིར་དངོས་མེད་ལ། །སྲིད་པ་ཡོད་པར་མི་འགྱུར་རོ། །དངོས་པོ་མེད་པར་དངོས་པོ་ནི། །མེད་པར་གང་ལས་འགྲུབ་པར་འགྱུར། །སྟོང་པ་གཏན་ཚིགས་ལས་འགྱུར་བ། །དེ་ཕྱིར་སྟོང་པར་མི་འགྱུར་ན། །དམ་བཅའ་གཏན་ཚིགས་ལས་གཞན་ནི། །མ་ཡིན་དེས་ན་གཏན་ཚིགས་མེད། །གང་ཕྱིར་སྟོང་པའི་དཔེ་ཡོད་པ། །དེས་ན་སྟོང་པར་མི་འགྱུར་ན། །བྱ་རོག་ཇི་ལྟ་དེ་བཞིན་དུ། །བདག་ཀྱང་ནག་ཅེས་བརྗོད་ནུས་སམ། །གལ་ཏེ་རང་བཞིན་གྱིས་དངརྟོགས་ད། །སྟོང་མཐོང་ཡོན་ཏན་ཅི་ཞིག་ཡོད། །རྟོག་པས་མཐོང་བ་འཆིང་བ་སྟེ། །དེ་ནི་འདི་རུ་དགདེ་ཉིད་ར་བྱ། །གཅིག་ཡོད་གཅིག་མེད་ཅེས་བྱ་བ། །དེ་མིན་འཇིག་རྟེན་པ་ཡང་མིན། །དེས་ན་འདི་ཡོད་འདི་མེད་ཅེས། །བརྗོད་པ་ཉིད་དུ་ནུས་མ་ཡིན། །ཡོད་དང་མེད་དང་ཡོད་མེད་ཅེས། །གང་ལ་ཕྱོགས་ནི་ཀླན་མིན་པ། །
དེ་ལ་ཡཔ་་ནི་རིང་པོ་ནའང་། །གླན་ཀ་བརྗོད་པར་ནུས་མ་ཡིན། །རྣལ་འབྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པ་ལས་སློབ་དཔོན་དང་སློབ་མ་རྣམ་པར་གཏན་ལ་དབབ་པ་བསབྱང་ཆུབ་སེམས་དཔའི་རྣལ་འབྱོར་སྤྱོད་པ་བསྟན་བཅོས་བཞི་བརྒྱ་པའི་ཚིག་ལེའུར་བྱས་པ་ཞེས་བྱ་བ། སིངྒ་ལའི་གླིང་དུ་པདྨའི་སྦུབས་ལས་བརྫུས་ཏེ་འཁྲུངས་ཤིང་། རང་དང་གཞན་གྱི་གྲུབ་པའི་མཐའ་རྒྱ་མཚོའི་ཕ་རོལ་ཏུ་སོན་པ་ཡང་དག་པ་དང་། ཡང་དག་པ་མ་ཡིན་པའི་ལྟ་བ་རྣམ་པར་འབྱེད་པས་དབུ་མའི་ལམ་རབ་དུ་གསལ་བར་མཛད་པ། འཕགས་པ་ཀླུའི་ཞབས་ཀྱི་སྲས་སུ་གྱུར་པའི་སློབ་དཔོན་འཕགས་པ་ལྷའི་ཞལ་སྔ་ནས་ཀྱིས་མཛད་ྱོར་སྤྱོད་པ་བཞི་བརྒྱ་པའི་ཚིག་ལེའུར་བྱས་པ་རྫོགས་སོ།། །།ཁ་ཆེའི་གྲོང་ཁྱེར་དཔེ་མེད་ཀྱི་དབུང་རིན་ཆེན་སྦས་པའི་ཀུན་དགའ་ར་བར། རྒྱ་ལོ་ཙཱ་ི་མཁན་པོ་སུ་སྨྲ་ཛཱ་ན་དང་། བོད་ཀྱི་ལོཙྪ་བ་པ་ཚབ་ཉི་པའོགྲགསའདིས་བསྟན་པ་དང་སེམས་ཅན་ལ་ཕན་ཐོགས་པར་གྱུར་ཅིག། །།ིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ།། །།སརྦ་མངྒ་ལཾ།། །།
