༄༅། །ཞེན་ལོག་སྒྲོན་མ་བཞུགས་བ་མ་མང་དུ་བཞི་བ ། དབང་པོ བྱུང་། དུ་བ་དུ་ཟླ་ནི། །ཀ་ཀློང་དུ་བྱ་བ་ག ག་ཚོད་ག་འི་ཁྱོན་དུ་མ་གིnལ་བ་ན་ཕྱོགས༄༅། །མཆོད་གནས་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ལ་འཇུག་ཐབས་དོན་འདི་བརྟནཔར་བྱ། །གང་ཞིག་བདག་གཞན་སྡུག་བསྔལ་སྤང་འདོད་ཅིང་། །བདེ་བ་འདོད་པས་དམ་ཆོས་སྤྱད་པར་བྱ། །མི་རྟག་ཆོསལ་བྱེ་བྲག་བསམ་ལས་འདས། །ཅི་ཕྱིར་འཇིག་རྟེན་འདི་དག་མི་རྟག་སྟེ། སྣོད་ཀྱི་འཇིག་རྟེན་རི་རབ་ལྷུན་པོ་དང་། །གསེར་གྱི་རི་བདུན་རོལ་མཚོ་ཆེནཔོ་བདུན། །གླིང་བཞི་གླིང་ཕྲན་ལྕགས་རི་མུ་ཁྱུད་ཡང་། །འཇིག་པའི་དུས་ཐན་བསྐལ་པའི་མེ་བྱུང་བས། །ཐལ་བའི་ལྷག་མ་ཙམ་ཡང་མི་ལུས་ཏེབརྟན་ཞིང་ངེས་པ་འདྲ་སྟེ་བརྟག་མི་མར་ཆུ་དང་རླུང་གིས་འཇིག་པས་ན། བཟོད།ལོ་ཟླ་ཚེས་གྲངས་ཤོལ་ནི་མི་ཐེབས་པས། ། ལོ་ཟླ་འདས་ཞིང་འཇིགཔའི་དུས་དང་ཉེ། །དུས་བཞི་འཁོར་བ་སྡོད་པ་མེད་པས་ནི། །དབྱར་དྲོ་སྐྱེལ་དགུན་གྲང་སྐམ་པ་དང་། །ལོ་ལེགས་ལོ་ཉེས་ཐན་ཞོད་ངེས་པ་མེད། །རྩི་ཞེན་ལོག་སྒྲོན་མ་བཞུགས།  དགེཐོག་ལྗོན་ཤིང་ལོ་འདབ་རྒྱས་པ་འདྲ། །དུས་ཀྱི་དྲོད་བྲལ་རླུང་གིས་མི་རྟག་གོ། །རྟག་པ་ལགས་ན་ཅི་ཕྱིར་འགྱུར་ཞིང་འགྲོ། །དེ་ལ་སོགས་པ་མི་རྟག་དགོངས་ནསགདའ། –་ཁོ །བཅུད་ཀྱི་འཇིག་རྟེན་འདི་ཡང་མི་རྟག་སྟེ། །ཉིན་ཞག་འདས་ཤིངཤི་དང་ཕར་ཕར་ཉེ། །འཁོར་བའི་སྡུག་བསྔལ་གསུམ་གྱིས་རྟག་ཏུ་འཁོར། ། སྐྱེརྒ་ན་འཆི་རང་ཐག་རི་མོ་འདྲ། །འཆི་བདག་བདུད་ནི་སོ་བྱ་ཉ་སྡོད་འདྲ། ཨེ འཇིག་རྟེན་འདི་ན་ཆེ་ཆུང་ཕྱུག་དབུལ་ཀུན། །འཆི་བདག་བདུད་ཀྱི་ལག་ཏུ་མིའགྲོ་མེད། །། སྔ་མ་ཤི་བའི་མི་རྣམས་ཀུན་གྱིས་ཀྱང་། །རྟག་ཏུ་རེ་ཞིང་བུ་དང་ནོར་བསགས་ཀྱང་། ། །རང་ནི་དགེ་སྡིག་གཉིས་དང་འགྲོགས་ཏེ་འགྲོ། ། བུ་ནོར་ཕྱིར་ལུས་བདག་ལ་ཕན་མེད་པས། །དེ་ཕྱིར་བུ་ནོར་ཟས་ལ་ཆགས་མི་བྱ། ཡོད་པས་དགེ་བའི་ལས་ལ་མི་འབད་ན། །དེ་ནི་འཇིག་རྟེན་འདི་ཡི་གཏེར་བསྲུང་ཡིན།ཤེས་པས་དམ་པའི་ཆོས་ལ་མི་སྤྱོད་ན། །མི་རྟག་པས་གཅམ་》བུ་དོནམེད་གླེན་པ་ཡིན། ། དེ་བས་ནོར་མེད་དབུལ་ལོ་བསྙེན་དགའ་དང་། །བྲམ་ཟེ་ལྡན་པ་ཕུལ་བ་དག་ལ་སྤྱོད། ་འཇིག་རྟེན་འདི་ཡི་ཕ་མ་བུ་ཚ་དང་། །ཕུ་ནུ་མཛའ་བཤེས་བྲན་ཁོལ་ནོར་རྣམས་ཀུན། །སྡུག་བསྔལ་སྐལ་ལེན་འགའ་ཡི་པ་མི་འོང་སྟེ། ལྷ་ལ།འཆི་བའི་ཕྱིར་ནི་གཅིག་ཀྱང་མི་འོང་ངོ༌། །དོན་དེ་མ་རིགསེམས་ཅན་མང་པོ་ཀུན། །ཚེ་འདིའི་འཁོར་བའི་ཉེ་བཤེས་ཆེད་དག་ཏུ། །འ་ཡི སྡིག་པའི་བྱེ་བྲག་རྣམ་པ་མང་པོ་སྤྱོད། །ཕན་མེད་རང་ནི་ངན་སོང་གསུམ་དུལྟུང་། །དེ་ཚེ་འགྱོད་ཀྱང་ཕན་མེད་ཉོན་མོངས་སྣོན། །ཚེ་འདིར་ཕྲད་པའི་ལསམཐུན་ཉེ་བཞི་རྣམས། །འགྲོན་པོ་གནས་མཐུན་ཙམ་སྟེ་སོ་སོར་འགྲོ། །སྡིགདང་བསྡོས་ནས་ཟས་དང་ནོརབསགས་ཀྱང་། །སྡིག་པའི་རྣམ་སྨིན་རྨོངས་པབདག་གིས་ཁུར། ལོ།བསགས་པའི་ནོར་ནི་ཕྱིར་ལུས་བུ་ཚས་ཁྱེར། །སྦྲང་བུས -  བསྟན་འགྱུར། རྒྱུད།  ཚུ། ་འབད་དེ་སྦྲང་རྩི་བསགས་པ་འདྲ། །དེ་བས་ལོངས་སྤྱོད་མེད་ཅིང་དབུལ་བ་ཡོང་།རྒྱུ་འབྲས་མི་སླུ་ཆོས་ལ་ཡིད་ཆེས་པ། ། །མོས་པས་དགེ་བ་དད་པའི་ཡོན་ཕུལ་ནས། དགེ་སློང་གསེར་འབྱིག་ལྟ་བུར་གྱུར་པ་ཡོད། །དེ་ཕྱིར་འཕགས་པའི་ནོར་བདུནདང་དུ་བླང་། །དད་དང་ཚུལ་ཁྲིམས་གཏོང་དང་ཐོས་པ་དང༌། ཁ ངོ་ཚ་ཤེས་ཤིང་ཁྲེལ་ཡོད་ཤེས་རབ་ནི། –་།འཕགས་པའི་ནོར་བདུན་ལགས་པར་ཐུབཔས་གསུངས། །དཔེར་ན་ནོར་གྱི་ནང་ན་ཆོག་ཤེས་པ། །རབ་མཆོག་ལགསཔར་ལྷ་མིའི་སྟོན་པས་གསུངས། །ཀུན་དུ་ཆོག་ཤེས་མཛད་ཅིང་ཆོག་མཁྱེན་ན། ནོར་མི་བདོག་ཀྱང་ཡང་དག་འབྱོར་པར་འགྱུར། ། ཌི་།བདག་ལ་ཤིན་ཏུ་ཕངས་སོ་པའི་ལུས་འདི་ཡང་། །གོས་རྙིང་ལམ་དུ་བོར་བ་བཞིན་དུ་འགྲོ། །ད་ལྟ་ས་སྟེངས་འདི་ན་གནས་པ་ཀུན། །ལོ་བརྒྱ་ཚུན་ཆད་གཅིག་ཀྱང་མི་ལུས་སོ། །ཌ་ཌ བརྒྱ་ཐུབ་འབུམ་ལ་གཅིག་ཀྱང་མི་འབྱུང་ན། །ངཚ་།བརྒྱ་ལ་སྲིད་ཀྱང་ཕལ་ཆེརཡེངས་འགྱུར་ཏེ། ཌ་པ །མཚན་མོ་ཉལ་བས་བརྒྱ་ཕྱེད་གཉིད་དབང་དུ། །སོང་སྟེདེ་ལ་དྲན་མེད་ཤི་དང་འདྲ། །ལྷག་མ་ལྔ་བཅུ་ཡོད་དུ་ཟད ཀྱང་ནི། །ར་རོཉལ་དང་འདུ་བྱེད་རྣམས་ཀྱི་རྐྱེན། །ལྔ་བཅུ་ཕྱེད་ནི་དེར་ཡང་ཟད་པས་ནི། །ཆ་ལེགས་པར་དྲན་པ་ཉི་ཤུ་རྩ་ལྔ་ཡོད། །དེ་ཡང་རང་ལ་དབང་བྱར་མ་མཆིས་ཏེ། དེ་རིང་དོ་ནུབ་འཆིའམ་གཏོལ་མེད་ ཀྱི། །ད་ལྟ་ཉིད་དུ་བརྟུན་ཏེ་འབད་ལབྱའོ།ཚེ་འདི་ཟིལ་པ་འདྲ་སྟེ་མེད་ལ་ཁད། ཟླ།དབུགས་ནི་ན་བུན་འདྲ་སྟེཡལ་ལ་ཁད། ། །ལུས་ནི་སྒྱུ་མ་འདྲ་སྟེ་སྣང་ལ་མེད། །སེམས་ནི་གློག་དང་འདྲ་སྟེསྐད་ཅིག་མ། །ཕུང་པོ་མི་རྟག་འཆི་ཞིང་འགྱུར་བ་སྟེ། །དར་བའི་མདོག་ནིཉོན་མོངས་ནད་ཀྱིས་འཕྲོག །འཆི་བའི་སྲོག་ནི་འཆི་བདག་བདུད་ཀྱིས་འཕྲོག དཔེར་ན་ཆུ་འགྲམ་གཡང་སའི་སྡོང་པོ་འདྲ། །རིང་པོར་མི་ཐོགས་བྱ་ཁྱི་ངུར་པ —ཚེ ཞེན་ལོག་སྒྲོན་མ་བཞུགས། །ཡདང་། ལ། །སྲིན་བྱ་འུག་པ་བྱ་རྒོད་ཁྭ་སྟེག་དང་། །ཁྱི་ལྟོགས་ཝ་དང་འབུ་སྦྲངགཟན་དུ་འགྱུར། ་།དེ་བཞིན་སྒྱུ་མ་མི་རྟག་མི་རྟོགས་པར། » །ཕུང་པོ་འདི་ཡི་ཆེད་དུ་སྡིག་སྤྱོད་པ། །མ་རིག་གླེན་པ་ཉོན་མོངས་སྙིང་རེ་རྗེ། །ཟས་ནོརའབྲེལ་པའི་ལོངས་སྤྱོད་འཇའ་ཚོན་འདྲ། ཡབ། ། རྨི་ལམ་ལོངས་སྤྱོད་རྙེད་དངམཚུངས་པ་སྟེ། །དེ་ལྟར་འདུས་བྱས་ཐམས་ཅད་མི་རྟག པས། །དམ་པའི་ཆོས་ལ་མྱུར་དུ་འབད་པར་བྱའོ། །དེ་བས་འཕགས་པའི་གཟུགས་བརྙན་མཆོདགནས་ལ། །མོས་པའི་དད་པས་དགེ་ལ་བརྩོན་པ་ནི། །བླ་མེད་བྱང་ཆུབ་ཐོབཔར་འགྱུར་བའི་དཔེ། །རྒྱལ་པོ་གསལ་ཐུབ་ལྟ་བུར་གྱུར་པའོ། །དེ་ཕྱིར་མཆོདཔའི་ལས་ལ་བརྩོན་པར་བྱ། །ཅིས་ཏེ་མི་ལུས་ཐོབ་པ་ཤིན་ཏུ་དཀའ། –་ཏེ་།ཁོམལྡན་མི་ལུས་ཐོབ་པར་གྱུར་པ་ན། །ཕྱི་མའི་ཕན་པ་འདི་ལ་མ་བསྒྲུབས་ན། ཕྱིས་ནི་མཐོ་རིས་ཐོབ་པར་ག་ལ་འགྱུར། །ངན་སོང་སྐྱེས་ནས་སྡུག་བསྔལ་འབྱུངབའི་ཚེ། །ཚེ་འདིའི་བཙན་ཕྱུག་བདེ་སྐྱིད་ཅི་ལ་ཕན། ། ། སྔ་མར་ཅི་སྤྱད་དྲན་པས་ཤིན་ཏུ་སྡུག ། དེ་ཕྱིར་འཁོར་བའི་ཆོས་ལ་མ་ཡེངས་པར། ། ཞེ་སྡང་ཆེནཔོ་མེ་ལྟར་འབར་བ་ལ། །སྙིང་རྗེ་ཆེན་པོའི་ཆུ་ཡིས་མྱུར་དུ་སོད། །འདོདཆགས་སེར་སྣའི་མདུད་པ་དམ་པོ་དེ། །ཆགས་མེད་སྦྱིན་པ་ཆེན་པོའི་སེམས་ཀྱིསཁྲོལ། ལོ །གཏི་མུག་མུན་པ་ཆེན་པོ་འཐིབས་པ་དེ། །ོ །ཤེས་རབ་གསལ་བའིསྒྲོན་མེ་ཆེན་པོ་ཐེག །ཉོན་མོངས་ཀུན་ལ་གཉེན་པོ་བཅུག་ནས་ནི། །འཁོརབའི་ཆུ་བོ་ཆེན་པོ་བཞི་རྣམས་ལ། །བརྩོན་འགྲུས་གྲུ་བཙུགས་ནས་ནི་རྒལ་བར་བྱ།དོ་ཅིག་ཡོད་པས་སང་ཕོད་ཡོད་པ་བཞིན། – ། ཚེ་འདི་ཡོད་པས་ཕྱི་མ་ཡོད་པར་ངེས། སང་ཕོད་ལོ་དང་སྐུ་ཚེ་སླད་མ་ནི། ཚེ་ཕྱིའི།གང་སྔར་འོང་བ་སུ་ལའང་གཏོལ་མེད་པས༑།སང་ཕོད་དོན་དུ་སོ་ནམ་འབད་པ་བཞིན། །ཕྱི་མའི་དོན་དུ་དམ —བསྟན་འགྱུརརྒྱུད། ་ཚུ༽  ་ཆོས་སྤྱད་པར་རིགས། །དེ་རིང་དོ་ནུབ་ཁོ་ན་མི་འཆི་ཞེས། །བག་མེད་བདེ་བར་འདུག་པ་རིགས་མ་ཡིན། །ཚེ་འདི་མི་སྡོད་རི་གཟར་ཆུ་དང་འདྲ། །རིངཔོར་མི་རྟག་ཚྭ་ཁའི་ཟིལ་པ་འདྲ། །ཡང་ན་སྐྱེ་ཤི་ཆུ་ཡི་ཆུ་བུར་འདྲ། ་ཀོ །འཁོརབར་སྐྱེ་བ་ཤིང་རྟའི་འཁོར་ལོ་འདྲ། །དེ་བས་རང་གི་ལུས་ལ་མ་ཆགས་པར། སྐྱབས་ཀྱི་སྦྱིན་པས་འགྲོ་ལ་ཕན་བཏགས་ན། ། བླ་མེད་བྱང་ཆུབ་ཐོབ་པར་འགྱུརབའི་དཔེ། །རྒྱལ་པོ་ཤི་བྱིན་ལྟ་བུར་གྱུར་པའང་ཡོད།  །མི་ཡི་ཆོས་ལུགསལེགས་པར་སྤྱད་པས་ན། །མཐོ་རིས་ལྷ་ཡུལ་བགྲོད་པ་ཐག་མི་རིང༌། ། ེ །མཐོརིས་ཐར་པའི་གོ་འཕང་གང་ཡིན་པ། །འཛེག་?ན་ཐར་པ་གམ་ན་འདུག་པས་ན། དེ་བས་མི་དགེ་བཅུ་སྤང་དགེ་བཅུ་སྤྱོད། །གྲགས་པ་ཉམས་པར་འགྱུར་བའིཆོས་དྲུག་སྤོང་། ། བྱ་།དགེ་ལ་མི་སྐུར་སྡིག་ལ་མི་བསྟོད་དེ། །འབྲས་བུ་གང་སྤྱད་དེ་དང་མཉམ་པར་གསུངས། །ངན་སོང་སྡུག་བསྔལ་ཤེས་ཀྱང་མི་འཇིགསཔར། །མི་རྟག་ལུས་ལ་ཆགས་ནས་སྡིག་སྤྱོད་པ། །ཨེ་མའོ་རང་གིས་རང་རླགསྙིང་རེ་རྗེ། ། ། ཡི་དྭགས་བྱོལ་སོང་སྡུག་བསྔལ་སྐྱི་རེ་བུང་། །དམྱལ་ཁམསབཅོ་བརྒྱད་སྡུག་བསྔལ་ཡ་རེ་ཆ། ། སྐྱེ་རྒ་ན་འཆིའི་སྡུག་བསྔལ་རྒྱུན་རེ་རིང་། ཕུང་པོ་སྒྱུ་མ་ཟས་ནོར་བསྙེན་མ་ཡིན། །འཁྲུལ་པའི་གཉེན་བཤེས་འཁོར་བའིབཤལ་ཐག་པས། །དེ་རིང་རང་དབང་ཡོད་པའི་མི་རྣམས་ཀྱིས། །རང་དངའབྲེལ་པའི་ཟས་ནོར་གཉེན་བཤེས་ལ། །སེར་སྣས་མི་བཅིང་ཆགས་སེམས་མི་སྐྱེདདོ། ། །སེམས་ཅན་རྩལ་ཤུགས་མེད་ལ་སྐུར་མི་བྱ། །ཆང་ནི་ཧ་ལའི་དུགབཞིན་རབ་ཏུ་བསྲུང་། །གཞན་གྱི་བུད་མེད་དག་ལ་ཆགས་མི་བྱ། ཞེ་།རྣམ་སྨིནམེད་པའི་སོ་ནམ་མི་བོར་ཞིང་། ཁ །གཞན་གྱི་ནོར་ལ་བརྩི་བདག་ཆོམ་རྐུན་སྤང༌།དོན་དུ་ལུས་ངག་ཡིད་གསུམ་སྤྱོད་པ་གཟབ། །མདོར་ན་མཚམས་མེད་ལྔ་དང་ཉེ —༽ང ཞེན་ལོག་སྒྲོན་མ་བཞུགས། ཋགྱུརབ་ལྔ། །ལྕི་བཞི་ལོག་བརྒྱད་ཁ་ན་མ་ཐོ་གསུམ། །འདི་སྤྱོད་ངན་སོང་གསུམ་དུགདོན་མི་ཟ། ། མི་རིགས་སྡིག་པ་སྔོན་དུ་སྤྱོད་པ་ལས། ་ཆ། །འགྱོད་པ་དྲག་པོསེམས་ནི་བསྐྱེད་ནས་སུ། ཁ །སྡིག་པ་འཕྲོ་བཅདདགེ་བ་སྤྱོད་པ་ནི། །འབྲསབུ་མཐོ་རིས་ཐོབ་པར་གདོན་མི་ཟ། །དེ་ཡང་ཅིས མངོན་སྟོན་པའི་བཀའ་ལསནི། །དང་པོ་རྒྱལ་པོ་མ་སྐྱེས་དགྲ་ཡིས་སུ། །མཚམས་མེད་པ་ཡི་སྡིག་པ་བྱསཔ་ལས་།ནགས་ཀྱི་དྲང་སྲོང་དག་ལས་སེམས་བསྐྱེད་ནས། ཁ།འགྱོད་པའི་སེམས་ཀྱིས་འབྲས་བུ་ཐོབ་པ་འདྲའོ། །དེ་ཕྱིར་ངེས་པ་འཕྲལ་བཤགས་དགེ་ལ་སྦྱོར། འོ་།གཞན་ཡང་དང་པོ་ཁ་ན་མ་ཐོ་བའི། །ཚིག་གིས་སྐུར་པ་བཏབ་སྟེ་སྨད་པ་ལས། ། ལ།མོད་ལ་འགྱོད་དེ་བཤགས་པ་ཕུལ་ན་ནི། །བུ་མོ་རྡོ་རྗེ་ལྟ་བུརསྐྱེས་པའོ། །མདོར་ན་གཞན་དོན་སྤྱོད་པར་གསུངས་པ་སྟེ། །བདག་ཕྱིརགཞན་ལ་གནོད་བྱས་བདེ་བར་འགྱུར། །མ་རིག་སྐྱེ་བོ་རང་གི་དོན་བྱེད་དང་། སྟོན་པ་སངས་རྒྱས་གཞན་གྱི་དོན་བྱེད་གཉིས།  །འབྲས་བུ་ཆེ་ཆུང་ཁྱད་པར་དེ་འདྲ་བས།  །སེམས་ཅན་བཀྲེས་པས་ཉེན་པར་གྱུར་པ་ལ ། །ཟས་ནོར་ལུས་ཀྱིསྒོ་ནས་ཕན་འདོགས་པ། ལ། །རྒྱལ་པོ་ཤུད་ལོ་གར་ནེ་ལྟ་བུ་དེ།། ཀྱ །གཞན་དོན་སྤྱད་ན་འབྲས་བུ་ཐོབ་པར་འགྱུར། །འདི་དག་ཤེས་ནས་དགེ་མེད་སྡིག་སྤྱོད་པ། མི་ཤེས་སྡིག་སྤྱོད་སྡིག་པ་བརྒྱ་འགྱུར་ཆེ། ཁྱཁ་།དེ་བས་དོན་མེད་ལས་ལ་མ་བརྩོནཔར། ཁ།དགེ་བའི་ལས་ལ་རྟག་ཏུ་སྤྱད་པར་བྱ། ། རྒྱུ་འབྲས་ཤེས་རྐྱེན་དགེབའི་བཤེས་གཉེན་པས། །དེ་ཕྱིར་དགེ་བཤེས་དམ་པ་བཙལ་བར་བྱ། །དེ་ནི་བས་ལེགས་པ་ཀུན་སྟོན་ཉེས་པ་སྐྱོབ། །ཡང་དག་ལམ་སྟོན་འབྲས་བུ་སྐྱེད་བྱེད་པས། ཕ་མ་ས་མཁན་དུས་ཀྱི་དབྱར་དང་འདྲའོ། །དེ་ཕྱིར་དགེ་བཤེས་དམ་པ་རྟག་ཏུབསྟེན།ཀ། །ཆོས་ཀྱང་ཤེས་པས་མི་ཆོག་སྤྱད་པ་གཅེས། ངོ་། །ཅི་ཡི་ཕྱིར་ན་མི་ཆོས་( —བསྟན་འགྱུར། རྒྱུད།  ཚུ ་སོ་ནམ་འདྲ། །ཆོས་ཀྱིསབདག་བཤད་བདག་གིས་ཆོས་མ་བཤད། །དོནརྟོགས་བྱ་ཕྱིར་དང་པོ་མཉན་ཡང་དགོས། > །ངེས་པ་རྙེད་ནས་ཚིག་མང་བསླབ་མིདགོས།ཚིག་མི་ཟད་ཕྱིར་ཉན་པས་མཐར་ཐུག་མེད། །ཉན་ཅིང་འཆད་པསྤྱད་ཐབས་བཙལ་ཕྱིར་ཡིན། །མི་སྤྱོད་ཉན་འཆད་དོན་ཆུང་ཚི་རེ་ཆད། ཡོང་གཞན་ལ་འཆད་པ་ཚིག་གི་ཐན་པ་ཡིན། །གཞན་གྱི་ཐན་འབྱལ་རང་ལ་མི་ཕན་ཏེ །བྲོ་མཁན་སྒྲུང་མཁན་བྲོ་སྒྲུང་འཆད་པ་འདྲ། །དེ་ནི་མི་ནག་ཆོས་ཀྱི་ལོ་རྒྱུསམཁན།ཐོས་པས་བརྒྱགསནས་རང་སེམས་མ་ཐུལ་ན།  །མཁས་ཀྱང་ཕནམེད་སླར་ལ་འཁོར་བའི་རྒྱུ། །ཐོས་མང་ཚིག་ལ་མཁས་པ་ནེ་ཙོ་འདྲ། ག་པ། །འགྲོབ་གང་ཞིག་རང་ལ་མེད་པ་ཡི། །ནོར་བུ་རིན་པོ་ཆེ་ཡི་ཡོན་ཏན་དག ། ཇིསྙེད་བརྗོད་ཀྱང་རང་ལ་ཕན་མེད་འགྱུར། །དེ་བཞིན་ཆོས་ཤེས་ཆོས་དང་མི་ལྡནཔསཌ་།སྤྱད་ཐབས་ཆེ་བའི་ཡོན་ཏན་ཇི་སྙེད་ཅིག །བརྗོད་ཀྱང་རང་གི་དོན་ལ་ཕན་མེད་འགྱུར། །གཞན་ཡང་ཤེས་ནས་དམ་ཆོས་མ་སྤྱད་ན། །༽  ། རྒྱ་མཚོའི་ནང་དུ་སྐོམ་ནས་ཤི་བ་འདྲ། །གཞན་ཡང་ཁེ་ཕྱིར་གཞན་ལ་ཆོས་འཆད་པ།རིན་ཆེན་རྙེད་ཀྱང་སྤྲང་པོ འདྲ་ཞེས་གསུངས། །དཔེར་ན་སྨན་མཁན་ནད་ལ་མི་གཏོངདང་། །དབུལ་མོས་ཕྱུག་པོའི་ནོར་བགྲངས་པ་དང་འདྲ། ལམ་རྒྱུས་ཤེས་ཀྱང་མ་ཕྱིན་འདྲ་ཞེས་གསུངས། །དེ་བས་འགྲོ་བ་ཀུན་ཀྱང་ཆོས་ལགཟུད། ། ཕ་མ་བུ་ཚ་གཉེན་འདུན་ཐམས་ཅད་དང༌། །འཁོར་གཡོག་ལས་ཀྱིསའབྲེལ་པ་ཐམས་ཅད་ཀུན། །གྲོས་བཏབ་གལ་གྱིས་གཅུ་དེ་དགེ་ལགཟུད། །ཆོས་ལ་ཆུད་ན་བྱང་ཆུབ་སྤྱོད་པ་ཡིན། །དགེ་བ་ཉམ་ཆུང་སྡིགཔ་སྟོབས་ཆེན་དང་། ། བླ་མེད་བྱང་ཆུབ་སེམས་ནི་མ་བསྐྱེད་པར། །དགེ་བ་གཞན་གྱི་ཟིལ་གྱིས་མི་ནོན་པས། ། འ །འཁོར་བའི་བཙོན་ར་བསྡམས་ཤིང་ཉམ —ཞེན་ལོག་སྒྲོན་མ་བཞུགས། ། ཨཔྤཡེཐག་པའི། ། ལ། །བདག་གཞན་མི་བདེ་བསལ བར་འདོད་པ་དེས། །བྱང་ཆུབསེམས་གཉིས་རྟག་ཏུ་གཏང་། མི་བྱ། ཚོ །སྡིག་མེད་དགེ་བའི་སེམས་དང་ལྡན་དེ་ལ། བདེགཤེགས་རྣམས་ཀྱི་སྲས་སུ་དགོངས་པ་དང་། །ལྷ་མིར་འགོད་པ་ཀུན་གྱིས་ཕྱག་བྱའི་འོས། །མདོར་ན་བྱང་ཆུབ་རྣམ་པ་གསུམ་དགོསཏེ། །དངཔོར་སྨོན་དགོས་བར་དུ་འཇུག་ཀྱང་དགོས། འོ། ། ཐ་མར་དོན་དེ་ཉིད་ལ་གནསཔར་བྱ། ། ། །སྨོན་ཀྱང་མ་ཞུགས་དོན་ལ་མ་གནས་ན། ། ། །ཡིད་སྨོན་བྱས་པས་འབྲས་བུ་མི་ཐོབ་པས། ་ལ།བྱང་ཆུབ་སྤྱོད་པ་རླབས་ཆེན་སྤྱད་པར་བྱ། །རྒྱུ་ནིདཀའ་བ་དང་དུ་མ་བླངས་པར། །འབྲས་བུ་བྱང་ཆུབ་ཐོབ་པར་མི་འགྱུར་ཏེ། གང་ལ་སྦྱིན་པམཆོག་ཏུ་གཏང དགའ་བ། །ཡན་ལག་ཀུན་གྱི་དང་པོ་མགོ་བོཡིན:: །སྦྱིན་པར་བཏང་། བ་རྒྱལ་པོ་ཟླ་འོདཡིན། །དེ་བཞིན་སྤྱད་ན་བྱངཆུབ་ཐོབ་པར་འགྱུར། ཌཱ།དེ་ཕྱིར་འཁོར་བའི་ཆོས་འདི་གཏན་དུ་སྤང་། ། 》 །མྱ་ངན་འདས་པའི་ལམ་ལ་འཇུག་པར་བྱ། །སྦྱིན་པའི་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགསནཚོ་། ས་ནི་དང་པོ་རབ་ཏུ་དགའ་བ་ཐོབ། ་འ །ཚུལ་ཁྲིམས་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། །གཉིས་པ་དྲི་མ་མེད་པའི་ས་ཡང་ཐོབ། ། །བཟོད་པའི་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། །གསུམ་པ་འོད་བྱེད་པ་ཡི་ས་ཡང་ཐོབ། ལྔ་བ བརྩོན་འགྲུས་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། །བཞི་པ་འོད་འཕྲོ་བ་ཡི་ས་ཡངཐོབ། བསམ་གཏན་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། །ལྔ་པ་ཤིན་ཏུ་སྦྱངདཀའི་ས་ཡང་ཐོབ། །ཎ །ཤེས་རབ་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། །དྲུག་པ་མངོན་དུ་གྱུར་པའི་ས་ཡང་ཐོབ། ཟླ་ཚོ།ཐབས་ཀྱི་ཕ་རོལ ཕྱིན་པ་ཡོངས་རྫོགས་ན། བདུན་པ་རིང་དུ་སོང་བའི་ས་ཡང་ཐོབ། ། །སྨོན་ལམ་ཕ་རོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། ཁ །བརྒྱད་པ་མི་གཡོ་བ་ཡི་ས་ཡང་ཐོབ། ལྔ་།སྟོབས་ཀྱི་ཕ་རོལ་ཕྱིན$)—བསྟན་འགྱུར། རྒྱུད། ་༼ཚུ༽ པ་ཡོངས་རྫོགས་ན། །དགུ་པ་ལེགས་པའི་བློ་གྲོས་ས་ཡང་ཐོབ། །ད། །ཡེ་ཤེས་ཕརོལ་ཕྱིན་པ་ཡོངས་རྫོགས་ན། ། ལ།བཅུ་པ་ཆོས་ཀྱི་སྤྲིན་གྱི་ས་ཡང་ཐོབ། །འབྲས་བུ་སྐུ་གསུམ་ལྷུན་གྱིས་གྲུབ་པར་འགྱུར། ། ཇི་ལྟར་སྔོན་གྱི་བདེ་གཤེགསརྣམས་ཀྱིས་སུ། །བྱང་ཆུབ་མཆོག་ཏུ་ཐུགས་ནི་བསྐྱེད་པ་དང་། །བྱང་ཆུབསེམས་དཔའི་སྤྱོད་པ་རླབས་པོ་ཆེ། །རིམ་པར་གནས་ནས་འབྲས་བུ་ཐོབ་པ་ལྟར། དེ་བཞིན་བདག་དང་གཞན་ལ་ཕན་དོན་དུ། སེམས་བསྐྱེད་ནས་ནི་དགེ་བགཞུང་དུ་སྤྱོད། །དེ་ལྟར་དམ་ཆོས་གཞུང་དུ་སྤྱད་པ་ན། །ཉོན་མོངས་འཆི་བདག་བདུད་ནི་འཇོམས་པ་ཡི། །བདུད་རྩི་མཆོག་ཉིད་ཀྱང་ནི་དེ་ཡིན་ནོ། དེ་ཕྱིར་དམ་ཆོས་གཞུང་བཞིན་སྤྱད་པར་བྱ། ལ་།ཆོས་གཞུང་མི་སྤྱོད་ཆོས་པའི་སམིང་ཅི་ཡོད། །ཚུལ་ཁྲིམས་མི་སྲུང་། དགེ་སློང་ཅི་ལ་ཟེར། །དམ་ཚིག་མེད་ནསྔགས་པ་གང་ལ་ཟེར། ཁོས །སྡིག་ནི་ཆུང་ཡང་མི་བརྙས་སྤང་བར་བྱ། །དགེ་བཆུང་ཡང་དང་དུ་བླང་བར་བྱ།  །མདོར་ན་ཚད་མེད་བཞི་དང་བསྡུ་བ་བཞི། གཞན་དོན་སྤྱད་ན་བདག་དོན་ཆུད་མི་ཟ། ། །དེ་ཕྱིར་སྒྲིབ་གཉིས་སྤོངས་ལ་ཚོགསགཉིས་བསྒྲུབ། ། །སྟོན་པའི་བཀའ དང་དོན་དུ་མཐུན་བྱེད་ཅིང༌། །ཆགས་མེད་དགེ་བར་གཏོང་བའི་གང་ཟག་ནི། །བྱང་ཆུབ་སེམས་དཔའི་རིགས་ཅནཡིན་པས་ན། ། དེ་ལ་ཕྲག་དོག་དབང་གིས་སྒྲོ་སྐུར་ཞིང་། །ཐབས་ཀྱིས་སྨོདབྱེད་འཁོར་བའི་གཡང་སར་ལྟུང་། །དགེ་བའི་སྔོན་ལ་ཞུགས་པའི་གང་ཟགགིས། །ཉོན་མོངས་འདུ་འཛི་མེད་པའི་དགེ་གནས་ན། །མཚན་མོ་སྟོདསྨད་དགེ་ལ་གནས་ནས་ནི། ། །གུང་ལ་ཉལ་ཚེ་དགེ་དྲན་འཕྲོ་ལ་ཉལ། །དེ་ཡིསབག་ཆགས་དགེ་ལ་གནས་པར་འགྱུར། །བག་ཆགས་རྟག་པར་གནས་ན་དང་པོའི་དྲོད། །ཡིན་པས་དགེ་བ་ཉམས་འོག་ཆུད་དེ་བྱ། །ཟས་ཀྱང་ཆགས་ཕྱིར—དེ  ཞེན་ལོག་སྒྲོན་མ་བཞུགས། ་མ་ཡིན་བརྒྱགས་ཕྱིར་མིན། །དགེ་རྟག བྱ་ཕྱིར་ཚོད་ནི་ཟིན་པར་བཟའ། རྒོད་དང་འགྱོད་དང་གནོད་སེམས་རྨུགས་པ་དང༌། །ཡུལྔ།གཉིད་དང་འདོད་ལའདུན་དང་ཐེ་ཚོམ་སྟེ། །སྒྲིབ་པ་ལྔ་པོ་འདི་དག་དགེ་བའི་ནོར། ཚོ།འཕྲོག་པའིཆོམ་རྐུན་ལགས་པས་རིག་པར་མཛོད། ། སྟོན་པའི་བཀའ་ལ་བརྫུན་པ་མི་མངའབས།ཀ་། ཐེ་ཚོམ་ནེམ་ནུར་ཡིད་གཉིས་མི་བྱའོ། །དོན་ལ་སྤྱད་ཅིང་ཚིག་ཕྱིརམི་འབྲངས་ཏེ།། ། །དོན་ནི་གསེར་འདྲ་ཚིག་ནི་སྒྲོན་མ་བཞིན། །གསེར་རྙེདནས་ནི་སྒྲོན་མ་གདེག་མི་དགོས། །དོན་ལ་སྤྱད་ཅིང་བསྒོམས་ན་རང་རིག་སྐྱེ། ཚིག་ཕྱིར་ཉན་ཅིང་འཆད་པ་མང་བྱས་ཀྱང་། །རང་རིག་མི་སྐྱེ་ཉོན་མོངས་འཕེལབའི་རྒྱུ། །མདོར་ན་ཐེག་པ་ཀུན་ལ་བློ་སྦྱངས་ཏེ། * །ངེས་པ་གཅིག་ལ་ཞུགས་ཏེམི་སྒྲུབ་པར། །ཡེ་རྔམ་ཟོས་ནས་བྱང་ཆུབ་འཁྲུལ་པ་སྟེ། །དེས་ནི་འབྲསབུ་ཐོབ་པར་མི་འགྱུར་རོ། །ཆོས་ཀྱང་རང་གི་ཉམས་དང་དཔྱད་ཅིང་སྤྱད། འབྲས་བུ་གོང་མ་ཆེ་ལ་གཏད་བྱས་ན། །ཉམས་ཀྱིས་མི་ལྕོགས་བཞིན་དུ་ཁས་ལེནཔ།ཐང་སྟོང་སྡིག་པ་རང་གིས་བླངས་པ་ཡིན། ། །ཐེག་པའི་གྲངས་ནི་གཅིགཏུ་མ་ངེས་ལ། །ཐེག་པའི་ཁྱད་པར་ཤེས་རབ་ཆེ་ཆུང་སྟེ། །ཉམས་འོག་ཆུདཔར་སྤྱད་ན་ཆུད་མི་ཟ། ། །དམ་ཆོས་གཞུང་དུ་སྤྱོད་པའི་མི་དེ་ལ། །དམ་ཅན་ཀུན་ཡང་མ་བསྐུལ་རང་བཞིན་འགོ། : །མཁས་པ་ཆེན་པོའི་མཐུ་ནི་ཉི་ཤརའདྲ། །འགྲོ་བ་ཀུན་ཀྱང་དད་ཅིང་འདུན་པས་ན། ལྟ་ཞུ་།འདུས་བྱས་འདུས་མབྱས་ཀྱི་དགེ་བ་ལ། ་ཁ །ཐུན་མོང་དག་ཏུ་སྤང་ལེན་མེད་པར་བྱ། །སྤང་ལེནབྱས་ན་སྟོན་པའི་བཀའ་དང་འགལ། །ཐེག་ཆེན་སྒོར་ཞུགས་གང་ཟག་རྣམས་ཀྱིས་ཀྱང་། སྐུ་ལ་།ཤེས་རབ་སྙིང་རྗེ་གཉིས་ཀྱིས་གཞི་བཟུང་སྟེ། །ཐམས་ཅདདབྱེར་མེད་རྟོགསཔ་ཤེས་རབ་ཡིན། །ཏ་ཁ།ཀུན་ལ་བུ་བཞིན་བྱམས་པ་སྙིང་རྗེ -བསྟན་འགྱུར། རྒྱུད། ཚུ༽  ཡིན། །དེ་གཉིས་མི་ལྡན་ཐེག་ཆེན་ཁས་འཆེབ། །ལོང་སྤྲང་རྒྱལ་པོ་ཡིནཞེས་ཟེར་ཏེ་བརྫུན། །ད་ལྟའི་གང་ཟག་སྔགས་ཀྱི་སྒོར་ཞུགས་པས། །གསངསྔགས་ཟབ་མོ་ནོར་རྫས་བརྙན་མར་བཙོངས། །འཁོར་བའི་བུ་ཚ་ཕོ་མོ་ཁྱིམ་ཐབབྱེད་ང།གསང་སྔགས་ཤེས་པས་བདག་ཕུང་སྙིང་རེ་རྗེ། །དེས་ནི་ཙནྡན་སོལབར་འཚོང་བ་དང་། །གསེར་ཕྱེ་སྦང་ཕྱེར་འཚོང་དང་འདྲ་བར་གསུངས། གཞན་ཡང་དད་པ་ཅན་གྱི་ནོར་བླངས་ནས། ་།དགེ་བར་མ་སྤྱད་མི་ཆོས་ནོར་དུསྤྱད། ཞ་།འབྲས་བུ་དག་ནི་རྔ་མོའི་ལུས་སུ་སྐྱེ། ། །དེ་ཡི་ནོར་ལ་ནོར་བདགལོངས་སྤྱོད་བྱ། །སྤྱོད་རེས་སྡུག་བསྔལ་ཉམས་སུ་མྱོང་བར་འགྱུར། །ལྕགསརྡོ་དངུལ་རྡོ་བཞུ་རུ་བཏུབ་པས་ན། །དད་པ་ཅན་གྱི་ནོར་ནི་བཞུར་མི་བཏུབ། ཚིག་ཙམ་མ་ཡིན་སེམས་ཀྱིས་མཉམ་པ་ཉིད། །རྟོགས་ནས་ཀློང་དུ་གྱུར་ཅིང་དྲོདལྡན་དང༌། །ཆགས་མེད་དགེ་བར་སྤྱོད་པའི་གང་ཟག་གིས། ། མ་བྱིན་བླངས་ནས་སྤྱད་ཀྱང་མི་སྒྲིབ་བོ། །མདོར་ན་སྔགས་ཀྱི་རྩ་བ་དམ་ཚིག་བསྲུང༌། །བྱབའི་རྒྱུད་ཀྱི་དམ་ཚིག་བསྟན་པ་ནི། ་།ཐུན་མོང་བསླབ་པ་བཞི་པོ་ཉམས་མི་བྱ། བླ་མེད་རྣམ་གསུམ་མཆོད་ཅིང་བླ་མ་བཀུར། །གྲོགས་ལ་གདུང་ཞིང་མི་མཐུན་ཁཟས་སྤང་། །མཚན་མར་བྱས་ཤིང་ལྷ་ལ་ཕུལ་བ་ཡི་། །ཋ་།ཁྲུས་ལ་བརྩོན་པ་བསྐྱེད་ཅིང་ལུས་སྤྱོད་བཟང་། །མ་ཉམས་བསྲུངས་ན་རིགས གསུམ་དངོས་གྲུབཐོབ། །རྣལ་འབྱོར་ཕྱི་པའི་དམ་ཚིག་བསྟན་པ་ནི། ། སྤྱི་ཡི་དམ་ཚིག་གསུམ་པོཉམས་མི་བྱ། །རྩ་བའི་དམ་ཚིག་དྲུག་ནི་གཞུང་དུ་བསྲུང་། ། །ཡན་ལག་དམཚིག་གསུམ་དང་ལྡན་པར་བྱ། །མ་ཉམས་བསྲུངས་ན་འབྲས་བུ་ཐོབ་པར་འགྱུར། རྣལ་འབྱོར་ནང་པའི་དམ་ཚིག་བསྟན་པ་ནི། ལ།དམ་ཚིག་བྱེ་བྲག་བསམ་གྱིས་མིཁྱབ་ཀྱང་། །མདོ་རུ་བསྡུ་ན་རྩ་བ་གསུམ་དུ་འདུས། །ལུས་སྲོག་བསྐྱེད ཅད  ཞེན་ལོག་སྒྲོན་མ་བཞུགས། ཋགྲངཔའི་དུས་གསུམ་ཕ་མ་དང་།  ། མ་རིག་སྒོ་འབྱེད་འདྲེན་སྟོན་སློབ་དཔོན་དང་། དམ་པས་འབྲེལ་པའི་གྲོགས་མཆེད་རྣམས་ལ་ནི། །ལུས་ངག་ཡིད་གསུམ་དོནདང་མཐུན་པར་བསྲུང་། ་།ཡི་དམ་ལྷ་ཡི་ཏིང་འཛིན་གཡང་། མི་བྱ། །དེ་ཡི་སྔགས་དང་ཕྱག་རྒྱ་རྒྱུན་མི་བཅད ། །དམ་ཚིག་ཉམས་དང་མ་འདྲེས་སྣོད་མེདལ༄༅། །གསང་བའི་མན་ངག་སྤེལ་བར་མི་བྱའོ། །ད་ལྟ་གང་ཟག་སྔགས་པརཁས་འཆེས་ནས། །སྐུ་ཡི་དམ་ཚིག་གཞུང་དུ་མི་སྲུང་བར། །ཕ་མ་སྤོང་བརྡུངབླ་མ་སྨོད་མི་ཁུར ། །ཕྲག་དོག་དབང་གིས་གྲོགས་ལ་བརྩེ་གདུང་བཅད ། ངན་སོང་གནས་སུ་ཐེ་ཚོམ་མེད་པར་འགྲོ། །གསུང་གི་དམ་ཚིག་གཞུང་དུ་མི་སྲུངབར། །ཏིང་འཛིན་འདོད་སྒྲུབ་འཛབ་ཏུ་མནའ་འཆའ་ཞིང༌། ། ཕྱག་རྒྱམ་བྱིན་ཟས་ནོར་ལེན་བྱེད་པའི། །སྔགས་པར་ཁས་འཆེས་འཁོར་བའི་གཡང་སར་ལྟུང་། 《 །གསང་སྔགས་ཟབ་མོ་དགོན་པར་མ་༽ སྤྱོད་པར། །སྣོད་དང་མིལྡན་ཉན་མི་རྙེད་ཐད་ལ། །བེག་ཅོ་བཅས་ནས་མན་ངག་ལྷུག་པར་བཤད། ཐུགས་ཀྱི་དམ་ཚིག་ཉམས་པས་དམྱལ་བར་སྐྱེ། ཁོ་།གསང་སྔགས་གབ་པའི་དོནརྣམས་མ་རྟོགས་པར། །དམ་མེད་མི་ནག་ཁྲལ་ཐེག་ཀུན་བསྡུས་ནས། །སྦསཔའི་གསང་བ་མངོན་སུམ་འཆོལ་བར་སྤྱོད། ། །བཀའ་དང་འགལ་བས་འཁོར་བའིགཡང་སར་ལྟུང་། ། །དང་པོར་དམ་ཚིག་གཞུང་དུ་མི་སྲུང་དང༌། །བར་དུདབང་རྣམས་རིམ་པར་མི་ནོད་དང་། །ཐ་མར་བཀའ་གཞུང་བཞིན་དུ་མི་སྒྲུབཔརྋ་།སྔགས་པར་ཁས་འཆེ་གང་ཟག་སྙིང་རེ་རྗེ། རེ།བཟའ་ཤིང་ར་བརདུག་གི་སྡོང་སྐྱེས་ནས།  །དུག་དེས་བཟའ་ཤིང་བྱིན་ཡང་མེད་བྱེད་པསབདག་པོས་དུག་དེ་ངོ་ནི་ཤེས་མ་ཐགཎཥ །རྩ་ནས་ཕྱུང་སྟེ་མྱུར་དུ་མེད་བྱེད་ལྟར། གང་ཞིག་རྩ་བ་ཉམས་པའི་ཕྱལ་བ་དང་། །འདྲེས་ཉམས་འགྱུར་བ་ཉེས་པ་འཕེལ -བསྟན་འགྱུར། རྒྱུད། ཚུའུ ་འགྱུར་བས། །དོན་རིག་ནས་ནི་མྱུར་དུ་སྤང་བར་བྱ།  །དཔེར་ན་འོ་མ་ཉུངངུའི་ཐིགས་པ་གཅིག །རུལ་བས་འོ་མ་ཀུན་ཀྱང་རུལབ་བཞིན། ། །དམ་ཉམས་སྣོད་ངནདེ་དང་འདྲེས་པ་ན། །ཉམས་པ་དེ་དང་དེ་ནི་འདྲ་བརགསུངས། །དེ་བས་ལུང་གཅིག་ཆུ་ལའང་མི་བཏུང་ངོ་། །དམ་ཚིག་ས་གཞིའདྲ་སྟེ་ཀུན་གྱི་རྟེན། །དམ་ཚིག་ནོར་བུའདྲ་སྟེ་འདོད་པ་འབྱུང༌། །དམ་ཚིག་མཁར་དང་འདྲ་སྟེ་འཇིགས་པ་སྐྱོབས། །དམ་ཚིག་གྲུ་དང་འདྲ་སྟེ་ཆུ་བོསྒྲོལ། །མདོར་ན་ཆོས་ཀྱི་རྩ་བ་དམ་ཚིག་བསྲུང་། སྡེ་ཕྱི་རབས་རྣལ་འབྱོརརྣམས་ཀྱིསེམས་ལ་འཇོག །རྣམ་སྨིན་གཞན་ནི་ཁད་དང་རིམ་གྱིས་སྨིན། གསང་སྔགས་ཐུན་མོང་མ་ཡིན་རྣམ་སྨིན་དྲག །སྒྱུ་མར་སྦྲུལ་ཞུགས་བཞིནདུ་འགྲོ་བའོ་ཞེས། །གསང་བའི་རྒྱུད་རྣམས་ཀུན་ལས་གསལ་བར་འབྱུང༌། གང་ཟག་སྔགས་པར་ཁས་ལེན་རྣམས་ཀྱིས་ནི། ཁ་བ་རང་གཞན་ཏིང་འཛིན་གཟུགས་བརྙན་དཀྱིལ་འཁོར་གསུམ། །དམ་ཚིག་དབང་དང་མཆོད་པའི་སྒོརཞུགས་ནས། །རང་ཉིད་ཟབ་མོའི་དོན་དང་ལྡན་པར་བྱ། | །ཐེག་པ་འོག་མའི་སྒོར་ཞུགས་བསྲུང་སྡོམ་དང་། །ཆོས་སྤྱོད་བྱེད་པ་མི་དགག་མི་སྨོད་དོ། །ཅིཕྱིར་ཆོས་ལ་སྤང་ལེན་ཕྱོགས་རིས་དང་། །  །བཟང་ངན་གཉིས་སུ་སྟོན་པས་མ་གསུངས་པས། །འདི་ལྟར་སྣང་བའི་ཆོས་འདི་ཐམས་ཅད་ལ། །གཉིས་མེདམཉམ་པ་ཆེན་པོར་རྟོགས་པ་ནི། །འབད་རྩོལ་ཅི་བྱས་དགེ་བར་གྱུར་པས་ན། དགེ་བའི་རྩ་བ་ཕྱོགས་རིས་མེད་པར་བྱའོ། །ད་ལྟའི་གང་ཟག་ཐེག་པ་འོག་མཔའི། །རྒྱུ་འབྲས་དོན་ལ་ཞེན་པ་ཁ་ཅིག་ནི། །གསང་སྔགས་ཟབ་མོའི་དོནརྣམས་མ་རྟོགས་པར། །སྔགས་ཀྱི༔ཏན་གན་ཆོས་སྤྱོད་འཕྲིགས་མཐོང་ནས། བན་དེ་ལོག་ཆོས་སྤྱོད་ཅིང་རྩིང་ཆོས་བྱེད།  །པགསང་སྔགས་ཆོས་མིན་སྟོན -ཞེན་ལོག་སྒྲོན་མ་བཞུགས། ། བྱངཔས་མ་གསུངས་ཟེར། །མན་ངག་ཟབ་མོའི་དོན་ལ་སྐུར་བཏབ་པས། །འབྲསབུ་མནར་མེད་དམྱལ་བའི་ཡང་སྟེང་དུ། །སྐྱེ་བར་གྱུར་པས་རྨོངས་པ་སྙིང་རེ་རྗེ། སྟོན་པའི་ཞལ་ནས་ངེས་པར་གསུངས་པ་ཡི། །མན་ངག་དོན་དམ་གསང་སྔགསཟབ་མོ་ལ། །བཀའ་བསླད གཞུང་བཞིན་དང་དུ་མི་ལེན་པར། །མཚན་ཉིདཆོས་སྤྱོད་དག་ལ་བློ་ཞེན་པས། །བསྐལ་པ་གསུམ་དུ་སངས་རྒྱས་ཞལ་མི་མཐོང་། སྐལ་བ་ཆུང་ཞིང་ཤེས་རབ་ཞེན་པ་དང༌། །བློ་ངན་དོན་ལ་རྨོངས་པའི་གང་ཟགགིས། །ཟབ་མོའི་དོན་ལ་མ་ཞུགས་སྐུར་པ་བཏབ། །རྣམ་སྨིན་དཔག་གིསམི་ལང་བརྗོད་ལས་འདས། །གསང་སྔགས་ཟབ་མོའི་དོན་ལ་ཡིད་ཆེས་ནས། ལྟ་སྤྱོད་དམ་ཚིག་གཞུང་དུ་ལྡན་པ་དང་། །སྒྲུབ་ལ་ཤིན་ཏུ་བརྩོན་པའི་གང་ཟགནི། །སངས་རྒྱས་ས་ལ་ཐེ་ཚོམ་མེད་པར་འགྲོ། །སྐྱེ་བོ་མ་རིག་ཆོས་ལ་གཟུདཔ་དང་། །ཆོས་པ་ཆོས་དང་མི་ལྡན་པ་རྣམས་ཀྱི། །དོན་དུ་བདག་གིས་ཡི་གེ་འདི་བརྩམས་པས། །ཀུན་ཀྱང་དད་ཅིང་དོན་ལ་འཇུག་པར་སྨོན། །ཞེན་ལོགསྒྲོན་མ་ཞེས་བྱ་བ་སློབ་དཔོན་མཁས་པ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་རྫོགས་སོ།། ་
