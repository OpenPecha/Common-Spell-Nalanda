༄༅། །ལུགས་ཀྱི་བསྟན་བཅོས་སྐྱེ་བོ་གསོ་བའི་ཐིགས་པ།༄༅༅། །རྒྱ་གར་སྐད་དུ། །བྱ་ན་པུ་ཛ་ན་ཏུ་བྷི་དྷུ་པོ་ཤཱ་ནཱ་མ། བོད་སྐད་དུ། ལུགས་ཀྱི་བསྟན་ཆོས་སྐྱེ་བོ་གསོ་བའི་ཐིགས་པ་ཞེས་བྱ་བ། འཕགས་པ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་གྱི་འབྱུང་གནས་འཇམ་དཔལ་ཀུན་གྱིས་དཔལ། །མཐའ་དག་བྱམས་པས་ཁྱབ་པར་འཇམ་པའི་ངང་། །འཇམ་དང་དཔལ་གྱི་གྲགས་པས་ཐུགས་རྗེའི་མགོན། །འཇམ་དཔལ་ཀུན་གྱི་མགོན་ལ་ཕྱག་འཚལ་ཏེ། །བློ་དམན་སྐྱེ་བོ་མུན་ནག་ལྟ་བུ་ལ། །སྙན་ངག་སྒྲོན་མ་ལྟ་བུ་གསལ་བྱའི་ཕྱིར། །བདག་ལྟ་དམན་ལ་སྙན་སྔགས་སྒྲོན་མེད་ཀྱང་། །བླུན་པོ་གསོ་ཕྱིར་ཆུ་ཐིགས་ཙམ་ཞིག་བརྗོད། །གཞན་གྱི་སྐྱོན་རྣམས་བརྗོད་མི་བྱ། །གཞན་གྱི་སྐྱོན་རྣམས་ཅི་བརྗོད་པ། །དེ་བཞིན་དུ་དག་ལའང་ཡོད་པ་སྟེ། །དཔེར་ན་ཝ་དང་བུད་མེད་བཞིན། །བདག་ཉིད་ལ་ནི་བསྟོད་མི་བྱ། །བདག་ཉིད་ལ་ནི་གང་བསྟོད་པ། །དེ་ནི་བློ་ཆུང་ཤེས་རབ་ཞན། །དཔེར་ན་ཀ་ལན་ད་ཀ་འདྲ། །གྲོགས་ངན་བསྟེན་པར་མི་བྱ་སྟེ། །གྲོགས་པོ་ངན་པ་སུ་བསྟེན་པ། །ན་གུ་ཏ་ནི་བསྟེན་པ་ཡིས། །ཤིག་རྣམས་འཕུང་བར་གྱུར་ལ་ལྟོས། །ངན་པ་རྣམས་ལ་གྲོས་མི་གདབ། །ངན་པ་རྣམས་ལ་གྲོས་བཏབ་ན། །ལེགས་པར་སྨྲས་ཀྱང་ཉེས་པར་འགྲོ། །སྤྲེའུས་ཀ་ལན་ཀ་ཡི། །ཚང་བཤིག་བ་མི་ཚུལ་ལ་ལྟོས། །སྤྱོད་པ་རྣམས་ལ་བག་ཡོད་བྱ། །བག་མེད་རྒྱུན་དུ་སྤང་བར་བྱ། །གླང་ཆེན་མྱོས་པ་ལྟ་བུ་ནི། །ནམ་ཡང་བྱ་བ་མ་ཡིན་ནོ། །ཚིག་རྣམས་བརྗོད་འདོད་རྟག་ཏུ་ནི། །ཇི་ལྟར་བརྟགས་ཤིང་དེ་ལྟར་སྨྲས། །གང་ཞིག་མ་བརྟགས་པར་སྨྲས་བ། །དཔེར་ན་སྨྱོན་པའི་བསྟན་བཅོས་བཞིན། །དགོས་པ་མེད་པའི་ཚིག་རྩུབ་ནི། །ནམ་ཡང་བརྗོད་པར་མི་བྱ་སྟེ། །བདག་མི་ཕན་པར་གཞན་གནོད་པ། །དཔེར་ན་ནེ་ཙོའི་རྫུན་ཚིག་བཞིན། །རྫུན་གྱི་ཚིག་རྣམས་བརྗོད་མི་བྱ། །རྫུན་གྱི་ཚིག་རྣམས་བརྗོད་བྱས་ན། །མྱུར་བར་གཞན་གྱིས་ཤེས་འགྱུར་ཏེ། །དཔེར་ན་བྱི་ལའི་རྫུན་ཚིག་བཞིན། །ཚུལ་འཆོས་སྤྱོད་པ་བྱ་བ་མིན། །ཚུལ་འཆོས་སྤྱོད་པ་གང་བྱེད་པ། །དེ་ཡིས་བཙུན་པར་མི་འགྱུར་ཏེ། །བྲམ་ཟེ་ན་ཡ་རྒྱལ་མོ་བཞིན། །མཛའ་ཤེས་ངན་པའི་ཚིག་མི་སྙན། །མཛའ་ཤེས་ཉན་པའི་ཚིག་ཉན་པ། །
འདོད་དོན་མེད་དེ་བསླུས་པར་འགྱུར། །དཔེར་ན་རུས་སྦལ་སྤྲེའུ་བཞིན། །གཡོ་ཅན་ངག་འཇམ་ཚིག་མི་ཉན། །དེ་ཡི་ཚིག་རྣམས་ཉན་བྱེད་པ། །དཔེར་ན་དབང་པོའི་རྒྱལ་མཚན་གྱིས། །སྤྱང་ཀི་ལ་སོགས་བསླུས་ལ་བལྟོས། །གནོད་སེམས་ཅན་གྱི་ཚིག་མི་མཉན། །དེ་ཡི་ཚིག་རྣམས་ཉན་བྱེད་ན། །བདག་ལ་ཕན་པ་མི་འབྱུང་སྟེ། །ཝ་ནི་སྲེག་པས་ཕུང་བ་བཞིན། །སྡིག་པའི་གྲོགས་པོ་ཡིད་མཐུན་བཞིན། །དགའ་བ་སྣ་ཚོགས་ཚིག་བརྗོད་ཀྱང་། །སྡིག་ངན་དེ་ནི་བདུད་ཡིན་ཏེ། །དཔེར་ན་གླིང་གི་སྲིན་པོ་བཞིན། །ཕྱུག་པོ་ཀླ་ཀློ་ལ་ཡང་ཡོད། །དཔའ་བོ་དུད་འགྲོ་ལ་ཡང་ཡོད། །དོན་དང་མཐུན་པའི་གཏམ་སྨྲ་བ། །འཇིག་རྟེན་འདི་ན་ཤིན་ཏུ་དཀོན། །དེ་ཕྱིར་ཡོན་ཏན་དག་ལ་ནི། །སྲོག་དང་སྡོས་ཏེ་འབད་པར་བྱ། །གང་ཞིག་ཡོན་ཏན་མི་སློབ་པ། །དེ་ནི་བླུན་པོ་ཕྱུགས་ལྟ་བུ། །ཡོན་ཏན་མེད་པའི་ཁ་ཉིད་ནི། །ས་ཡི་བུ་ག་དག་དང་མཚུངས། །ཡོན་ཏན་རབ་ཀྱིས་བརྒྱན་པ་ཡི། །ཁ་ནི་པད་མོ་ཡིད་འོང་ལྡན། །གཞོན་པའི་དུས་སུ་རིག་པ་བསླབ། །ཞོ་འདོད་པ་ནི་དགུན་ནས་པ་ནི་རྒུན་ནས་གསོས། །དྲོད་གཤེར་ལྡན་དུས་ས་བོན་གདབ། །འདི་གསུམ་འབྲས་བུ་སྨིན་པའི་རྒྱུ། །ཡོན་ཏན་དག་ལ་འབད་བྱ་ཡི། །འགྱིང་ཚུལ་དགོས་པ་ཅི་ཞིག་ཡོད། །འོ་མ་མེད་པར་གྱུར་པའི་བ། །དྲིལ་བུ་བཏགས་ཀྱང་ཉོར་མི་འདོད། །བདག་ལ་ཕུན་སུམ་ཚོགས་ཡོད་ཀྱང་། །གཞན་ལ་བརྙས་པར་མི་བྱ་སྟེ། །ཞུན་མར་ལྟ་བུའི་སེམས་ཀྱིས་སུ། །ཡོན་ཏན་མཐའ་དག་ཁོང་དུ་སྦ། །ཤེས་རབ་ལྡན་པ་ལུས་ཆུང་ཡང་། །དོར་ཐབས་ཅན་གྱིས་བརྙས་མི་བྱ། །དཔེར་ན་བློ་གསལ་རི་བོང་གིས། །སེང་གེ་སྲོག་དང་ཕྲལ་བ་བཞིན། །ཀུན་ལ་མཐོན་དམན་མེད་པ་ཡི། །ཉི་མ་ལྟ་བུའི་ཕན་བྱས་ན། །ཀུན་གྱིས་ལན་རྣམས་ལྡོན་འགྱུར་ཏེ། །བྱི་བས་གླང་ཆེན་བསླངས་བ་བཞིན། །སེམས་ཅན་ཀུན་ལ་ཕན་སེམས་བསྐྱེད། །རེས་འགའ་རྩ་བཞིན་ཞུམ་པར་བྱ། །རེས་འགའ་མེ་བཞིན་སྤར་བར་གྱིས། །རེས་འགའ་གངས་རི་ལྟ་བུར་བགྱིས། །རེས་འགའ་དམག་ལ་བབ་པ་བཞིན། །གཞན་གྱིས་ཡུལ་ན་གནས་པ་དང་། །གཞན་གྱི་འཚོ་བས་འཚོ་བའི་ཚེ། །
ཇི་ལྟ་བུར་ནི་བརྙས་གྱུར་ཀྱང་། །ཁ་རོག་འདུག་པས་བདེ་བ་འཐོབ། །སྐྱེ་བོ་ངན་པ་ཡུན་རིང་བསྟེན་བྱས་ཤིང་། །བཟའ་བཏུང་ཞིམ་པོ་མང་པོས་ཕན་བཏགས་ཀྱང་། །ཕན་མེད་གྲོགས་པོ་སྤོང་གྱུར་ཏེ། །དཔེར་ན་ཁྲོན་པར་ལྷུང་བའི་མི་བཞིན་ནོ། །མི་གང་བློ་ནི་གཡོ་ཞིང་འགྱུར་བྱེད་པ། །དེ་ཡི་མཛའ་བཤེས་བརྟག་པར་མི་འགྱུར་ཏེ། །རབ་ཏུ་ཉོན་མོངས་མི་དེ་ཅི་འདྲ་ན། །ཤིང་ཤུན་ཚོས་ཀྱིས་རས་དཀར་བཙོས་པ་བཞིན། །གལ་ཏེ་བློ་ལྡན་མཁས་པས་དགྲ་བོ་དག །རབ་ཏུ་མི་དགའ་བར་ནི་བྱེད་འདོད་ན། །ཚིག་རྩུབ་མི་སྨྲ་རབ་ཏུ་ངག་འཇམ་གྱིས། །དེས་ནི་དེ་ལ་གནོད་པ་ཆེན་པོར་འགྱུར། །རྒྱ་མཚོ་དང་ནི་རི་རབ་གླིང་ལ་སོགས། །འཇིག་རྟེན་པ་རྣམས་བདག་གི་ཁུར་མིན་གྱི། །འཇིག་རྟེན་བྱས་པ་མི་བཟོ་གང་ཡིན་པ། །དེ་ནི་བདག་གི་ཁུར་ཆེན་དག་ཏུ་འབྱུང་། །མི་གང་སེམས་ནི་མི་རྒོད་ཅིང་། །བྱས་པ་བཟོ་དང་བྱས་པ་ཚོར། །བྱས་པ་ཆུད་བཟོན་མི་བྱེད་པ། །དེ་དག་མཁས་པས་རབ་ཏུ་བསྔགས། །ཆོས་བདེ་བྱམས་པ་བདེ་བ་སྟེ། །འཇུག་ངོགས་བདེ་བའི་ཆུ་ཀླུང་བཞིན། །གྲོགས་ལ་འཁུ་བ་མེད་པ་དང་། །ང་སྙེམས་ང་རྒྱལ་ཟད་པ་བདེ། །ཁེངས་དང་ཟོལ་བྱེད་ང་རྒྱལ་ཆེ། །སྡིག་པའི་གྲོགས་པོ་བསྟེན་པ་དང་། །སྟོབས་ལྡན་རྣམས་ལ་རྒོལ་བྱེད་པ། །དེ་དག་ཡིད་ནི་བདེ་མི་འགྱུར། །དེ་ཕྱིར་ཁེངས་པར་མི་བྱ་སྟེ། །ཁྲོ་བ་རྣམ་པར་སྤང་བར་བྱ། །ཁེངས་དང་ཁྲོ་བ་འོང་འགྱུར་ན། །དེ་དག་རྣམས་ཀྱི་གྲགས་པ་འཇིག །གང་ཞིག་རྒན་དང་བྱིས་པ་ནད་པ་དང་། །ཉམ་ཆུང་ཐབས་དོར་དེ་བཞིན་དམན་པ་དང་། །རྒྱལ་པོས་སྤངས་དང་སྐྱབས་མེད་པ་དག་ལ། །སྙིང་རྗེ་མེད་པ་དེ་དག་མིར་མི་རུང་། །རང་གི་ཡུལ་འཁོར་གནས་དཔགས་ཏེ། །གཞན་གྱི་ལྗོངས་སུ་ཕྱིན་པ་ན། །ཚིག་ངན་སོ་སོར་ཟློས་པ་ལ། །བཟོད་པ་ཆེན་པོ་བསྐྱེད་པར་བྱ། །དཀའ་ཐུབ་ཚུལ་ནི་བཟོད་པ་ཡིན། །གཙོ་བོའི་ཚུལ་ནི་དཀའ་སྤྱོད་ཡིན། །ཤར་བའི་ཚུལ་ནི་དཀའ་བཟློག་ཡིན། །བུད་མེད་ཚུལ་ནི་དཀའ་ཐུབ་ཡིན། །དེ་ལྟར་མ་ཡིན་རང་བཞིན་ངན་པའི་མི། །བྱམས་ལ་ལན་མེད་གཞན་གྱིས་ཕན་བྱས་བརྗེད། །སྡང་བས་གནོད་པ་ཇི་སྙེད་བྱས་གྱུར་ཀྱང་། །སྔང་་ཐབས་བྱས་པ་དེ་ཡང་ཆུ་བཞིན་འཐུང་། །ཚེ་དང་ལོངས་སྤྱོད་གཉེན་བཤེས་ཀུན་དར་དཀོན།སྐོན།དེ་དག་རྒུད་པ་ཤིན་ཏུ་རྙེད་པར་སླ། །དེ་ཕྱིར་སེམས་ཅན་དམས་པར་གྱུར་པའི་ཚེ། །
ཁྭ་བཞིན་མུ་ཅོར་མི་སྨྲ་ཁ་རོག་འདུག་ །གལ་ཏེ་ནོར་མེད་བདག་གི་ཕྱོགས་ཉུང་ལ། །མཛའ་བཤེས་སྟོབས་ཀྱི་ཕྱོགས་ཀྱང་ཆུང་གྱུར་ཞིང་། །དགྲ་བོའི་ཕྱོགས་ཀྱི་སྟོབས་ཚུལ་ལྡན་པའི་ཚེ། །ཚིག་ངན་མི་སྨྲ་ཁ་རོག་འདུག་པར་བྱ། །ཡ་རབས་ཚུལ་མེད་ཁེངས་པས་ང་རྒྱལ་ཤིན་ཏུ་ཆེ། །གཡོ་སྒྱུ་དད་མེད་ཁ་བཞིན་ཤིན་ཏུ་མུ་ཅོར་སྨྲ། །གཟུ་ལུམས་ཕྲག་དོག་སྐམས་དང་ཁྲིམས་མེད་ངོ་ཚ་བྲལ། །མི་དེ་བུར་གྱུར་ཀྱང་ནི་གཞན་བཞིན་ཤིན་ཏུ་སྤང་བར་བྱ། །དེ་ལྟ་བུ་ཡི་དམན་ཞིང་ངན་པ་དེ། །མཐོན་པོའི་སར་མི་དྲོང་ཞིང་དེར་མི་བཞག །ས་དེར་བཞག་ཀྱང་དྲེགས་དང་ཁེངས་མི་ཐུབ། །དཔེར་ན་པགས་པ་སྔོན་པོའི་ཝ་དང་འདྲ། །སྐྱེ་བོ་རྨོངས་པ་འགའ་ཞིག་ནི། །ལོག་པའི་ཆོས་ལ་རབ་ཞེན་པ། །འདོད་པའི་དོན་དེས་བསླུས་འགྱུར་ཏེ། །རུས་སྦལ་སོ་ཡིས་བྱ་བསླུས་བཞིན། །རབ་ཏུ་མི་བཟོའི་སྐྱེ་བོ་རྣམས། །སྡུག་བསྔལ་འབྱུང་བའི་རྒྱུ་དག་ལ། །དང་བླངས་དོན་དུ་གཉེར་བྱེད་པ། །གཅན་ཟན་གལ་སྒོར་བྱེད་པ་བཞིན། །དྲེགས་པའི་ལོངས་པ་འགའ་ཞིག་ནི། །རང་གི་སྐྱོན་རྣམས་མ་བསམ་པར། །ཡོན་ཏན་ཕྲ་མོས་རབ་ལ་བ། །ཁྲོན་པའི་སྦལ་པ་ཇི་བཞིན་ནོ། །རྨོངས་པའི་ལོག་པ་་བ་འགའ་ཞིག་ནི། །ཆོས་ཀྱི་དེ་ཉིད་མ་རྟོགས་པར། །ལོག་པའི་ཆོས་ཀྱིས་རབ་མགུ་བ། །བླུན་པོ་ནོར་བུ་ལེན་པ་བཞིན། །ཤིན་ཏུ་བླུན་པོ་ལ་ལ་དག །འཆི་བའི་རྣམ་པ་རབ་བརྗེད་ནས། །བྱ་བ་མང་པོ་རྩོམ་བྱེད་དེ། །བདུད་ཀྱིས་བསླུས་པར་རིག་པར་བྱ། །འགྲོ་བ་བཟང་པོ་བཟང་མོ་ལ། །གང་ཞིག་རབ་ཏུ་ཆགས་བྱེད་པ། །སྲོག་ཆགས་མེ་ལ་མཆོང་བ་བཞིན། །འདོད་པས་འཕུང་བར་འགྱུར་བར་ཡིན། །རྟ་དང་མ་ཧེ་གླང་པོ་སོགས། །འགའ་ཞིག་དེ་ལ་ཆགས་བྱེད་པ། །འདོད་པ་ཅན་གྱི་སྐྱེས་བུ་ཡིས། །རྨི་ལམ་ནོར་རྙེད་བསླུས་པ་བཞིན། །སྡིག་པའི་ལས་ལ་རབ་དགའ་ཞིང་། །དགེ་བ་དྲན་པར་མི་བྱེད་པ། །འཇིག་རྟེན་དག་ལ་རབ་ཏུ་ཆགས། །དུག་དང་འདྲ་བའི་སྐྱེས་བུ་དེས། །འདིར་ཡང་སྡུག་བསྔལ་ཆེན་པོ་ཅན། །ཕྱི་མ་མནར་མེད་རབ་ཏུ་འཇིགས། །བར་དོར་སྲིད་པའི་ལས་རྣམས་དྲན། །གསལ་བའི་མེ་ལོང་རབ་འཇིགས་པས། །ལས་ཀྱི་དགེ་སྡིག་རྣམ་པར་ཕྱེ། །རང་དབང་གནས་སུ་གནས་དབང་མེད། །མི་འདོད་པ་རུ་རྣམ་པར་ཁྲིད། །ལས་ཀྱི་རྒྱུ་འབྲས་མི་བསླུ་བ། །ཁོ་མོ་མཐོང་བའི་རྣམ་པའམ། །གཞན་ཡང་རབ་ཏུ་གྲངས་མང་ལྟོས། །དེ་ཕྱིར་དམ་པའི་སྐྱེས་རྣམས་ཀྱིས་། །
རྒྱུན་དུ་འཆི་བ་དྲན་པར་བྱ། །རང་གིས་ཅི་བྱས་ལས་རྣམས་བགྲང་། །ཡུད་ཙམ་པ་ཡི་ཚེ་འདི་ལ། །ཐང་ཅིག་འདུག་པའི་བློ་ཅན་དེ། །དེ་ནི་ཤིན་ཏུ་འཁྲུལ་པ་དེ། །མཁའ་འགྲོ་མ་ཡི་རྫུ་འཕྲུལ་བཞིན། །དེ་ཕྱིར་རྟག་པའི་སེམས་མི་བསྐྱེད། །ངན་སོང་གསུམ་པོ་དྲན་པར་བྱ། །སྲོག་གཅོད་ལ་སོགས་ལུས་ཀྱི་ནི། །ཉེས་པ་རྣམ་པར་བསྡམ་པར་བྱ། །རྫུན་སོགས་ངག་གི་ཉེས་པ་སྤང་། །དེ་བཞིན་ཡིད་ཀྱི་ཉེས་པ་སྤང་། །མ་ལུས་རབ་ཏུ་བསྡམ་པར་བྱ། །དེ་ནི་སྐྱེས་བུ་དམ་པའི་ཚུལ། །རབ་ཏུ་ཚ་ཞིང་དྲང་བ་ཡིན། །དམྱལ་བ་ཉི་མ་ཅིག་ག་བཞིན་དྲན། །བཀྲེས་སྐོམ་དག་གིས་རིད་པ་ཡི། །ཡི་དྭགས་རྣམས་ཀྱང་དྲན་པར་བྱ། །རྨོངས་པའི་སྡུག་བསྔལ་རབ་མང་བའི། །བྱོལ་སོང་ལྟ་ཞིང་དྲན་པར་བྱ། །དེ་རྒྱུ་རྣམ་སྤང་བདེ་རྒྱུ་སྤྱད། །ཐོབ་དཀའ་འཛམ་གླིང་མི་ཡི་ལུས། །དེ་ཐོབ་དུས་སུ་ངན་སོང་རྒྱུ། །རབ་ཏུ་བསྒྲིམས་ཏེ་བཅད་པར་བྱ། །འདིར་བདག་མི་ཡི་ལུས་ཐོབ་ནས། །ཕྱི་ནས་སྨོན་ལམ་འདེབས་བྱེད་པ། །དེ་ནི་བླུན་པོ་ལེ་ལོ་ཅན། །ངན་སོང་གནས་སུ་རབ་འགྱོད་དེ། །གྲོ་བཞིན་སྐྱེས་ཀྱི་བརྡོ་འཇོག་གི །ཡི་དྭགས་མཐོང་བ་ཇི་བཞིན་ནོ། །འདི་ཡི་ཚེ་ནི་མ་ངེས་པས། །དེ་རིང་སང་ངམ་ཟླ་བའམ། །ལེ་ལོའི་རྣམ་པས་འཆི་འགྱུར་ཏེ། །དེ་ཕྱིར་དགེ་ལ་བརྩོན་པར་བྱ། །དམ་པའི་ཆོས་ལ་རབ་འཇུག་ན། །རིགས་ངན་གཤན་མོ་ལྟ་བུ་དང་། །འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོ་ཡང་། །བྲན་མོ་བཙུན་མོར་ཐུབ་པས་གསུངས། །དམ་པའི་ཆོས་ལ་མི་འཇུག་ཅིང་། །འཇིག་རྟེན་ཁྲེལ་ལྟས་མི་སྲུང་བ། །ཤིན་ཏུ་སྨད་ཅིང་བཤུང་འགྱུར་ཏེ། །བྷི་ར་ལི་བའི་དཀའ་ཐུབ་བཞིན། །རི་བྲག་མདོམ་གང་སོགས་ཡོད་ཟབས། །དེ་ལའང་རབ་ཏུ་སྲུང་བྱེད་ན། །ཤིན་ཏུ་གཏིང་མེད་གྱུར་པ་ཡི། །གཡང་སར་ལྷུང་་བ་ཅིས་མི་རུང་། །ཡུངས་འབྲུ་ཙམ་གྱིས་རྨ་ཁུང་ཡང་། །ཤིན་ཏུ་བསྒྲིམས་ཏེ་འཆོས་བྱེད་ན། །འཚོ་མེད་ཚལ་པར་འགས་པ་ཡི། །རྨ་དེ་ཅི་ཕྱིར་འཆོས་མི་བྱེད། །ཉིན་པར་དགའ་ཟུང་ཙམ་དུ་ཡང་། །
ཟས་སྐོམ་གྱིས་དབུལ་མི་བཟོད་ན། །བར་གྱི་སྐལ་ལ་སོགས་སུ། །ཟས་གོས་མི་མྱོང་ཅི་ཕྱིར་བཟོད། །རང་ལུས་དག་ལ་ཚེར་མ་ཡི། །རྩེ་མོའང་བཟོད་པར་མི་འགྱུར་ན། །གཞན་གྱི་ལུས་ལ་མཚོན་དག་གིས། །སྲོག་དང་འབྲེལ་པ་ཅི་ཕྱིར་བཟོད། །རྨ་དྲེའུ་དུག་དང་་ཤིང་སྲིན་དང་། །ས་བོན་མྱུ་གུ་སྡོང་སྐྱེས་མེ། །དེ་བཞིན་ངན་པའི་མི་རྣམས་ནི། །རང་རྟེན་ཁོ་ན་འཕུང་བ་སྒྲུབ། །ཕྱི་མ་ཤེས་མ་འཛངས་མའི་ཚུལ། །ཉོན་མོངས་དགྲ་ཐུབ་དཔའ་བོའི་ཚུལ། །ཆགས་པ་མེད་པ་ཆོས་བྱེད་ཚུལ། །མཁས་པའི་ཚུལ་ནི་རྟོགས་པ་ཡིན། །རྟོགས་པའི་ཚུལ་ནི་བྱེད་པ་ཡིན། །སྐྱོ་བོ་མེད་པ་སྦྱིན་པའི་ཚུལ། །ཚུལ་ཁྲིམས་ཚུལ་ནི་མ་དཔྱས་པ། །བཟོད་པའི་ཚུལ་ནི་ཆོས་ཉིད་ཤེས། །བརྩོན་འགྲུས་ཚུལ་ནི་འཆི་བ་དྲན། །བསམ་གཏན་ཚུལ་ནི་གཡེངས་མེད་ཡིན། །རྣམ་པར་འབྱེད་ཤེས་ཤེས་རབ་ཡིན། །སངས་རྒྱས་འདོད་པ་དད་པའི་ཚུལ། །རྟོག་པ་ཆུང་བ་རྣལ་འབྱོར་ཚུལ། །ཟང་ཟིང་གཉེན་བྲལ་སྒོམ་མཁན་ཚུལ། །ཕན་སེམས་ཆེ་བ་སྒྲིན་པའི་ཚུལ། །དྲང་པོའི་ཚུལ་ནི་མི་བསླུ་བ། །བག་ཡོད་ཉམ་ང་མེད་པ་ནི། །དེ་ནི་རྒྱལ་པོའི་ཚུལ་དུ་བཤད། །དེ་ཡི་ཚུལ་དང་མི་ལྡན་པ། །དམན་ཞིང་སྨད་པའི་ཚུལ་བྱེད་པ། །ལས་ཀྱི་རྒྱུ་འབྲས་མི་བསླུའི་ཚུལ། །འོག་གི་ཚུལ་ནི་འདི་བཞིན་ནོ། །ཡོན་ཏན་ལྡན་ལ་སེར་སྣས་བཅིངས། །བསླངས་ན་ཁྲོ་བ་དང་བཅས་པ། །སྡུག་བསྔལ་ཆེན་པོས་ནོན་གྱུར་ཏེ། །ཤུ་མ་རི་ཡི་མི་སྡུག་བཞིན། །མཉམ་སོགས་དག་ལ་ཕྲག་མ་དོག །འདིར་ཡང་སྡུག་བསྔལ་ཆེ་འགྱུར་ལ། །ཕྱི་མར་རྣམ་སྨིན་ཆེ་འགྱུར་ཏེ། །དཔེར་ན་གཞོན་ནུ་ཝ་དང་འདྲ། །དྲེགས་མེད་ཤིན་ཏུ་དུལ་བར་བྱ། །ཀུན་ལ་གསོང་པོར་སྨྲ་བར་བྱ། །ཁྲོ་མེད་ང་རྒྱལ་སྤང་བར་བྱ། །ཀུན་ལ་འཛུམ་ཞིང་དྲང་པོར་སྨྲས། །དོན་མེད་ཅལ་ཅོལ་སྨྲ་མི་བྱ། །འདུ་བ་མང་པོའི་སར་མི་འགྲོ། །ཆང་སོགས་མྱོས་པ་བཏུང་མི་བྱ། །ཅི་འདོད་སྦྱིན་ལ་ཡངས་པར་བྱ། །བཤེས་མེད་མི་ལ་བྱམས་པར་བྱ། །དགེ་བའི་ཆོས་ལ་བརྩོན་པར་བྱ། །ཁྲིམས་སོགས་བླངས་ལ་བསྲུང་བར་བྱ། །དེ་ལྟར་བྱས་ན་མང་པོའི་མི། །ངང་གིས་བདེ་བ་འཐོབ་པར་བྱེད། །
ད་ལྟ་ཤིན་ཏུ་ངན་པའི་མིས། །རྒྱལ་བའི་བསྟན་པ་ནུབ་ཏུ་ཉེ། །ད་ནི་མན་བསྔགས་ངན་སྤྱོད་དང་། །ལོག་པར་ལྟ་བ་མང་བ་ཡི། །རྨོངས་པ་དག་གིས་བཅོམ་ལྡན་གྱི། །བསྟན་པའི་ལྗོན་ཤིང་བཅད་ཀྱིས་མེད། །ད་ནི་སྙིང་རྗེ་མེད་པ་ཡི། །ཆོས་ལུགས་བོར་ཤིང་རང་བཟོ་བྱེད། །བརྐམ་ཆགས་ཕ་རོལ་མ་མཐོང་བས། །བསྟན་པའི་རྒྱལ་མཚན་བརྙལ་གྱིས་མེད། །ད་ནི་གདུལ་ཀ་ངོ་ཚ་མེད། །སྡིག་ལ་དགའ་ཞིང་ཚུལ་འཆོས་བྱེད། །སྡིག་ཅན་རྣམས་ཀྱིས་ཐུབ་པ་ཡི། །བསྟན་པའི་ཟམ་པ་བཅག་གིས་མེད། །ད་ནི་མགོ་བྲེགས་ཉམས་མི་ལེན། །སྤྱོད་འཆལ་སྨད་པ་ལུས་ལ་འཆོས། །ངན་གཡོ་ཅན་གྱིས་ཐུབ་པ་ཡི། །བསྟན་པའི་ཚོང་འདུས་བཅོམ་གྱིས་མེད། །ད་ནི་ཁྲོ་དང་ང་རྒྱལ་དང་། །རྒྱགས་ཆོས་བྱེད་ཅིང་རྙེད་ལ་ཞེན། །སྡིག་པ་སྣ་ཚོགས་བྱེད་པའི་མིས། །ཆོས་ཀྱི་གྲུབ་པོ་བཅག་གིས་མེད། །ད་ནི་ཐུབ་པའི་བསྟན་པ་ཡང་། །ཅུང་ཟད་ཙམ་ཞིག་ལུས་པར་ཟད། །ཐུབ་པའི་བསྟན་པ་གངས་ཆེན་མཚོ། །རིང་པོར་མི་ཐོགས་སྐམ་པར་འགྱུར། །སེང་གེའི་སྒྲ་ཡི་བསྟན་པ་ནི། །དུས་ཀྱི་འཁོར་ལོས་རབ་བཅོམ་ནས། །མྱུར་བ་ཉིད་དུ་འཇིག་པར་འགྱུར། །གང་དག་འདི་འདྲའི་དུས་བབ་ཚེ། །བདེ་འདོད་རྣམས་ཀྱིས་ཕན་པའི་དོན། །དེ་ནི་ལེགས་པར་བརྟག་པའི་རིགས། །དུས་ཀྱི་འཁོར་ལོ་ངན་བྱེད་པ། །མ་རབས་ཤེས་རབ་འཆལ་བ་དང་། །ཚུལ་ཅན་དག་དང་བསྟེན་མི་ཤེས། །ཉེས་པ་བསྟེན་ཅིང་ཚུལ་ཁྲིམས་སྤངས། །དབེན་པས་འཇིགས་ཤིང་འདུ་འཛིན་འདོད། །མི་མཁས་བསླབ་ཉེས་སློབ་དཔོན་ནོར། །གདུག་རྩུབ་ང་རྒྱལ་བརྙས་པ་ཆེ། །ངན་པའི་མི་ལ་ཡིད་རྟོན་པས། །སངས་རྒྱས་བསྟན་པ་མ་རུངས་བྱས། །ཞི་ཞིང་བདེ་བའི་དུས་ནི་ཟད། །གདུག་རྩུབ་མི་ཟད་དུས་ནི་བྱུང་། །ཤེས་རབ་ཅན་གྱིས་མི་རྣམས་དང་། །དབང་གྱུར་སྐྱེ་བོ་རྣམས་ནི་ཟད། །དེ་ལྟ་བུ་ཡི་དུས་རྣམས་བྱུང་བའི་ཚེ། །རང་གི་ཡུལ་འཁོར་གནས་སྤངས་ཏེ། །གཞན་གྱི་ལྗོངས་སུ་ཕྱིན་བ་ནི། །དེ་ཡང་ནགས་འདབས་དབེན་པ་རུ། །སྨྲ་བ་སྐྱུངས་ཏེ་གནས་པར་བྱ། །དཀར་པོའི་ཆོས་རྣམས་ཡིད་ལ་མི་འཇོག་སྐྱེ་བོ་རྣམས་ལ་ནི། །མི་ཆོས་ཕུལ་དུ་བྱུང་བ་ཡིད་ལ་འཇོག་པ་དེ་ཉིད་ནི། །
ཅུང་ཟད་དགེ་བར་འགྱུར་སྙམ་ནས་ནི་བྲིས་པ་ཡིན། །དགེ་བས་བླུན་པོ་མ་ལུས་གྲོལ་ཞིང་བསྟན་ཆོས་ཡུལ་དུ་གནས་པར་ཤོག །ལུགས་ཀྱི་བསྟན་ཆོས་ཀྱི་སྙན་དངགས་སྐྱེ་བོ་གསོ་བའི་ཐིགས་པ་ཞེས་བྱ་བ། །སློབ་དཔོན་འཕགས་པ་ཀླུ་སྒྲུབ་ཀྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་ཤི་ལེནྡ་བོ་དྷི་དང་། ལོཙྪ་བ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་བའོ།། །།
