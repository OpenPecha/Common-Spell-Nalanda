༄། །སྦྱོར་བ་བརྒྱ་བ་བཞུགས།༄༅༅། །རྒྱ་གར་སྐད་དུ། ཡོ་ག་ཤ་ཏ་ཀ །བོད་སྐད་དུ། སྦྱོར་བ་བརྒྱ་པ། སངས་རྒྱས་སྨན་གྱི་བླ་བཻ་ཌཱུརྱ་འོད་ཀྱི་རྒྱལ་པོ་ལ་ཕྱག་འཚལ་ལོ། །འཕགས་པ་ཀླུ་སྒྲུབ་ལ་ཕྱག་འཚལ་ལོ། །རིང་དུ་གྲགས་པའི་གསོ་སྤྱད་ནས། །མཐའ་ཡས་རྒྱུད་ཀྱི་གཅེས་པ་བླངས། །སྨན་པ་མཁས་པར་རབ་མཆོད་པའི། །སྦྱོར་བ་བརྒྱ་ཡིས་བཅིངས་པར་བྱ། །བདུད་རྩི་ལྟ་བུའི་སྨན་སྦྱོར་གྱིས། །ལུས་ནི་ནད་མེད་འགྱུར་བ་ཡིན། །གསོ་དཔྱད་ཤེས་པའི་སྨན་པ་ཡིས། །ནད་རྒྱུ་མཚན་ཉིད་ཡོངས་སུ་བརྟག །སླེ་ཏྲེས་སླ་སྒང་བྱི་ཚེར་བཅའ་སྒ་དང་། །བྱི་ཚེར་གླ་སྒང་ཏིཀྟ་པརྦ་ཊ། བཱ་སཱ་བྱི་ཚེར་བཅའ་སྒ་གླ་སྒང་རྣམས། །བསྐོལ་ཐང་རླུང་མཁྲིས་བད་ཀན་རིམས་ལ་བཏུང་། །ཀྵུ་ཏྲ་སླེ་ཏྲེས་བཅའ་སྒ་པཥྐཽར། །བསྐོལ་བའི་ཐང་བྱས་བད་ཀན་རླུང་མཆོག་ལ། །ལུད་པ་དབུགས་མི་བདེ་བཅས་དང་རིམ་ལའང་བསྔགས། །དོང་ག་དྲོད་སྨན་ནར་རྩ་གླ་སྒང་དང་། །ཧོང་ལེན་ཨ་རུ་ར་རྣམས་བསྐོལ་བའི་ཐང་། །གཟེར་བཅས་ཚད་འཁྲུ་བད་རླུང་ལྡན་རིམས་ལ། །ཕན་ནོ་དྲོད་སྐྱེད་བཞུ་བ་བྱས་པས་ཀྱང་། །རྒུན་འབྲུམ་གླ་སྒང་ཧོང་ལེན་ཨ་རུ་ར། །དོང་ག་པརྤ་ཏ་ཀའི་ཐང་བྱས་ལ། །མཁྲིས་བྱུང་རིམས་དང་བསླབ་བཙོལ་བརྒྱལ་བ་དང་། །མགོ་འཁོར་ཚ་སྐམ་སྐོམ་དད་ལྡན་རྣམས་ལ། །ཀཎྜ་ཀཱ་རི་བཅའ་སྒ་སླེ་ཏྲེས་བསྐོལ། །དྲོ་སྨན་ནར་མོ་བསྲེས་པའི་ཐང་འཐུངས་པས། །གསུས་གཟེར་ལུད་པ་ཚད་རྙིང་དང་ག་མེད། །ཕོ་བའི་ནད་ཆབ་དབུགས་མི་བདེ་རྣམས་འཇོམས། །བྱི་ཚེར་པརྦ་ཊ་ཀ་ཨ་རུ་ར། །བ་ཤ་ཧོང་ལེན་ཏིཏྦ་པྲི་ཡངྐུ། །ཀ་ར་ཆེར་བཏབ་བསྐོལ་བའི་ཐང་འཐུངས་པས། །སྐོམ་དང་ཁྲག་མཁྲིས་རིམས་དང་ཚད་ལྡན་འཇོམས། །རིམས་ནད་གསོ་བའི་སྤྱད་དོ།། ༎སྐྱེར་རྩ་འབྲས་བུ་གསུམ་དང་བི་ཤཱ་ལཱ། །སྐྱེར་རྩ་ཕྱེ་མར་བྱས་ལ་ངེས་པར་བསྐོལ། །ཐང་ལ་སྦྲང་རྩི་ལྡན་པ་འཐུངས་པ་ཡིས། །གཅིན་ཁ་སྙི་བའི་ནད་སྐྱེས་མ་ལུས་འཇོམས། །ཨུ་དཱི་ཙྱ་དང་ཀ་བདེ་བོང་ང་དཀར། །གླ་སྒང་དུག་མོ་ཉུང་བཅས་ཐང་བྱས་པ། །ཚད་པ་ཟེར་བཅས་ཁྲག་འཁྲུ་དང་བཅས་ལ།ཡུན་རིང་ཞུགས་པའི་འཁྲུ་བ་རྣམས་ལ་ཕན། །གླ་སྒང་བཅའ་སྒ་བོང་ང་དཀར་པོར་བཅས། །སླེ་ཏྲེས་ཆ་མཉམ་ཆུ་ལ་བསྐོལ་འཐུངས་པས། །མེ་དྲོད་ཞན་པའི་རྒུན་གྱི་ནད་རྣམས་དང་། །ཚད་ཟིན་འཁྲུ་བའི་ནད་ཀྱང་སེལ་བར་འགྱུར། །པགས་ནད་ལྟོ་བ་ན་དང་སྐྱ་རྦབ་ན། །སྦོས་དང་ཆམ་པ་བད་ཀན་མགུལ་འགགས་ལ། །པུནརྣ་བཱ་ཐང་ཤིང་ཨ་རུ་ར། །སླེ་ཏྲེས་བ་གཅིན་གུ་གུལ་ལྡན་པ་བཏུང་། །བླ་སྒང་ཤི་གྲུ་ཐང་ཤིང་དནྟི་འབྲས། །
བི་ཏ་གྲོ་སྨན་ནར་ཕྱེ་བཅས་ཐང་གིས། །རྒྱུན་དུ་ཁ་རྐུབ་ནས་འབྱུང་སྲིན་ཕུ་དང་། །སྲིན་ལས་སྐྱེས་པའི་ནད་ཀྱང་འཇོམས་པར་འགྱུར། །སུཀྵྨེ་དྲོ་སྨན་ཤིང་མངར་ན་ལེ་ཤམ། །ཨསྨ་བྷེ་དང་ཨ་རཾ་བཱ་ཤ་གཟེ། །བྲག་ཞུན་གཙོ་ཆེར་བསྐོལ་བའི་ཐང་འཐུངས་ན། །རྡེའུ་གསེག་མ་དང་བཅས་གཅིན་སྲི་སེལ། །བྲག་ཞུན་ཨསྨ་བྷེ་དང་སུཀྵེ་དང་། །དྲོ་སྨན་ཕྱེ་མ་འབྲས་བཀྲུས་ཆུ་བཏབ་བཏུང་། །ཡང་ན་བུ་རམ་དང་བཅས་འཐུངས་ཙམ་ན། །གཅིན་སྲི་འཆི་ལ་ཐུག་པའང་འཆོ་བར་འགྱུར། །དོང་ག་དང་ནི་གཟེ་མ་ཨ་རུ་ར། །བྱི་ཚེར་ཨསྨ་བཱེ་ཏ་བ་ཤ་ཀ །བསྐོལ་བའི་ཐང་ལ་སྦྲང་རྩི་ལྡན་འཐུངས་ན། །ཚ་ཞིང་གཟེར་བཅས་གཅིན་སྲི་གྲོལ་བར་འགྱུར། །དོང་ག་དང་ནི་སླེ་ཏྲེས་བཱ་ཤ་རྣམས། །ཨེ་རཎ་མར་ཁུ་ལ་བསྐོལ་ཐང་འཐུངས་ན། །རླུང་བྱུང་རྣམ་འགྱུར་མ་ལུས་ཁྲག་གི་ནད། །སྟེང་འོག་ལུས་ཀུན་ནས་འཛག་རིམ་པས་འཇོམས། །ཏཎྜུ་ལ་ཀའི་རྩ་བ་ལིག་བུ་མིག །འབྲས་ཀྱི་ཆུན་ལ་སྦྲང་རྩི་ལྡན་འཐུངས་ན། །ཀུན་གྱིས་བསྐྱེད་པའི་ཁྲག་ནད་ངེས་པར་འཇོམས། །དབུགས་མི་བདེ་ལ་བྷཱརྒཱི་བཅའ་སྒས་སོ། །བི་ཧ་ཏི་གཉིས་ཨེ་རཎ་ཀ་པེད་དང་། །མ་ཏུ་ཨསྨ་ཀྵཱ་ར་ཤིང་ཀུན་གཟེ། །སཻནྡྷ་ཨེ་རཎ་མར་ཁུར་བསྲེས་བྱས་ཐང་། །རྐེད་པ་སྙིང་ནུ་ཕོ་མཚན་གཟེར་བས་བཏུང་། །ཤིང་ཀུན་ཤུ་དག་ཁ་རུ་བཅའ་སྒ་དང་། །གོ་སྙོད་ཨ་རུ་པུཥྐར་རུ་རྟ་རྣམས། །ཕྱི་མའི་ཆ་འཕེལ་ཕྱེ་མར་བྱས་རྣམས་ཀྱིས། །ལྟོ་སྐྲན་རིམས་རྙིང་གསུད་ནད་རྣམས་ལའོ། །བཅའ་སྒ་ཤིང་ཀུན་རུ་ཙ་ཆ་སྙམ་ཕྱེད་། །སྒ་ཆུ་ལ་བསྐོལ་བད་ཀན་རླུང་བྱུང་དང་། །རྩིབ་ལོགས་སྙིང་རྒྱབ་གསུས་གཟེར་གསུད་རྣམས་ལ། །བཏུང་ངོ་དེ་བཞིན་ནས་ཁྲུས་རྟུག་འགགས་ལའོ། །འཇམ་འབྲས་ལོ་མའི་བཱི་ཤཱ་ལཱ་ཙ་བྱ། །ཙི་ཏྲ་ཚ་གསུམ་ལན་ཚྭ་གཙོ་ཆེར་བཙེགས། །བསྲེགས་པའི་ཐལ་བ་ཞོ་ལ་སྦྱར་འཐུངས་པས། །དམུ་སྐྲན་སྐྱ་རྦབ་རྐུབ་ཀྱི་ནད་རྣམས་སེལ། །ཤིང་ཀུན་ཤུ་དག་བཅའ་སྒ་སྭརྩི་ཁ་རུ་ཕྱི་མ་ཉིས་འགྱུར་ཕྱེ་མར་བྱས། །ཆུ་དྲོན་བཏབ་འཐུངས་སྣ་ཚོགས་གསུད་གཟེར་དང་། །སྙིང་ནད་སྐྲན་དང་སྟོད་དུ་རླུང་འཚངས་འཇོམས། །ཤིང་རྒི་ཚ་གསུམ་འབྲས་གསུམ་ཀཎྜ་ཀ །བྷཱརྒཱི་པུཥྐར་ཛ་ཌཱ་ཚྭ་སྣ་ལྔ། །ཕྱེ་མ་ཆུ་དྲོན་བཏབ་འཐུངས་སྐྱིགས་བུ་ལུད། །དང་ཀ་མེད་ཆབ་སྟོད་རླུང་མི་བདེ་འཇོམས། །ཛམ་བུ་དུག་ཉུང་ཨརྒ་ཤིགྲུ་སྣུ་ཧ་བསེ་ཤིང་བྱཱ་གྷི་བྲིཧ་ཏཱི། པཱ་རི་ཀ་པེད་ཚོས་ཤིང་ཛ་ཌཱ་ཨ་བཱ་མརྒ་ནིམྦི་ཙེ་ཏྲ་ཀ །བ་ཥ་པུཥྐར་པ་ཌ་ལ་དང་སཻནྡ་ལྡན་པ་བསྲེགས་ཐལ་བསྐོལ་བ་ལ། །སྔར་བཤད་ཤིང་ཀུན་ལ་སོགས་པའི་ཁུ་བ་ལྟོ་སྐྲན་བད་ཀན་ནད་ལ་བཏུང་། །ཙ་བྱཱ་སྟར་བུ་ཚ་གསུམ་བསེ་ཡབ་བཅས། །བ་ལུ་ཛཱི་ར་བཅུ་གང་ཙི་ཏྲ་ཀ །ཕྱེ་མ་ཆ་མཉམ་བུར་བསྲེས་དྲི་གསུམ་ལྡན། །གྲེ་མཛེར་ཆམ་བད་དང་ག་མེད་ལ་བསྔགས། །བ་ལུ་ཙ་བྱན་ཤམ་མཚུངས་གཉིས་པ། །མཱུ་ལ་བཅུ་གང་དྲོ་ཙའོ་སྒ་གསུམ་འགྱུར། །
བུ་རམ་སུམ་འགྱུར་དྲི་བཟང་རིལ་བྱར་མུར། །གྲང་བཞི་གློ་རིམས་གཟེར་བའི་གཞང་འབྲུམ་སེལ། །སྒ་དྲོད་ནལ་ཤམ་ནཱ་འདབ་ཤིང་ཚ། །སུཀྵྨེ་མཐའ་རིམས་གང་འཕེལ་ཕྱེ་མར་བྱས། །འདི་དག་ཆ་འདྲའི་ཀར་འགམ་གཞང་མེད་ཞན། །དབུགས་དང་མི་བདེ་སྐྲན་མགུལ་སྙིང་ནད་ལའོ། །སེནྡྷེ་ཤིང་ཀུན་འབྲས་གསུམ་ལ་ལ་ཕུད། །ཚ་གསུམ་བུ་རམ་ཆ་རྣམས་རིལ་བུ་བྱས། །དེ་རྣམས་མུར་ཞིང་མྱངས་པས་ཚིམ་ཐོབ་ཅིང་། །མེ་དྲོད་ཞན་ལས་རབ་ཏུ་བྱུང་བའང་སེལ། །ལྕགས་ཕྱེ་ཚ་གསུམ་བརྒྱ་ཤུགས་ཆ་ཐམས་ཅད། །མཱཀྵི་ཕྱེ་མ་དང་མཉམ་དེ་རྣམས་བཟའ། །རྒུན་འབྲུམ་ལྡན་པའི་རྗེས་སུ་དར་འཐུངས་པས། །སྐྱ་རྦབ་ནད་ལ་ཕན་ཞིང་རིང་དུ་འགྲོ། །བུ་རམ་ལྡན་པའི་བཅའ་སྒའི་དྲོད་སྨན་ནར། །གསུམ་པ་ཨ་རུ་རའམ་བལ་པོའི་སེའུ། །བད་ཀན་ཚད་སྙིང་གཞང་འབྲུམ་ནད་རྣམས་དང་། །རྟུག་པ་འགགས་པ་རྣམས་ལའང་རྟག་ཏུ་བྱ། །བཅའ་སྒ་གླ་སྒང་ཨ་རུ་ར་ཕྱེ་ལ། །བུ་རམ་ལྡན་པ་ལ་ནི་རི་ལུ་བྱ། །ལུད་པ་དབུགས་མི་བདེ་དང་ཚོ་ཆེས་རྣམས། །ཁ་རུ་རི་ལུ་མུར་བས་བཟློག་པར་འགྱུར། །ནྱ་གྲོ་ཨུཏྤལ་རུ་རྟམཱ་མཱཀྵཱ་ཀ །ཕྱེ་མ་འབྲས་ཆེ་བཅས་པ་རིལ་བུ་བྱ། །སྲུ་པ་ཁ་རུ་མུར་པས་སྐོམ་དད་དང་། །རབ་ཏུ་འཕེལ་བ་མྱུར་དུ་འཇོམས་པར་འགྱུར། །ལྡོང་རོས་དྲོད་སྨན་ནར་མོ་ན་ལེ་ཤམ། །ཕྱེ་མ་ཀ་པིངྠ་སྐྱུར་ཁུ་བར་ལྡན། །འབྲས་ཕྱེ་ཆ་མཉམ་མཀྵི་ཀ་བསྲེས་པའི། །པུས་པས་ལན་ཅིག་མ་ཡིན་སྐྱུག་པ་འཇོམས། །སྣ་ཁྲག་འཛག་ལ་དུརྦའི་ཁུ་བའམ། །བལ་སེའུ་མེ་ཏོག་ཁུ་སྦྱར་ངེས་པར་བསྟན། །བུད་མེད་ནུ་ཞོ་རྒྱ་སྐྱེགས་ཁུ་བའམ། །བཱིཌ་མཀྵི་ཀས་ཀྱང་ཨི་ཀུ་འཇོམས། །སྐྱུ་རུ་སྤོས་དཀར་ཁུ་བ་ཁ་རུ་ཆ། །དེ་རྣམས་བསྲེས་པ་སོ་བི་རི་ལ་བཏགས། །དེས་འདིར་མྱུར་བར་གཞི་ནས་མེད་བྱད་ཅིང་། །ཇི་ལྟའི་སྐྱིགས་བུ་སླར་མི་འབྱུང་བ་ཡིན། །དུརྦ་སེནྡྷེ་ཐལ་ཀ་ཨ་རུ་ར། །ཀུ་ཐེ་ར་ཀཱ་ཀཉྫི་ཀ་དར་བཏགས། །གསུམ་པས་བྱུག་པའམ་ནུ་མའི་འོག་བཏགས་པས། །ཤུ་བ་དང་ནི་གཡན་པ་རྣམ་པར་འཇོམས། །སྣུ་ཧ་ཙི་ཏྲ་བྷི་ག་ཨརྐ་དང་། །མྱ་ངན་མེད་ཤུན་རུ་རྟ་ཚ་བ་གཅིན། །བཅོས་པའི་མར་ཁུས་ཤུ་བ་མཛེ་གླང་ཤུ། །གདུག་པའི་རྨ་དང་ཟ་འབྲུམ་སེ་ཁ་འཕྲོག །གུ་གུལ་མིག་སྨན་པྲ་ཚིལ་སིནྡྷུ་ར། །པང་མ་ཆ་མཉམ་ཚ་བའི་མར་ཁུས་བཙོ། །ཁྱི་རྡོ་ཟ་ཙག་གིས་ནི་ལུས་སྐམས་ལ། །ལན་ཅིག་བྱུགས་པའི་ནུས་པ་གཞི་ནས་འདོན། །པུ་བརྩི་ཀཱ་རུ་རྟ་མུརྦ་སྒ། །རྒྱ་སྐྱེགས་དང་ནི་བརྩོད་དང་ཡུང་བ་རྣམས། །དྲུག་འགྱུར་དར་བ་ལ་བསྐོལ་གྲུབ་པ་ན། །བྱུགས་པས་གྲང་དང་ཚ་བའི་རིམས་འཇོམས་སོ། །
ཀཎྜ་ཀཱ་རི་སླེ་ཏྲེས་པཱ་ཤ་ཀ །ཕྱེ་མར་ཞུན་མར་ལ་བསྐོལ་གྲུབ་པ་དེ། །འཐུངས་པས་ཚད་རྙིང་ལུད་གཟེར་དབུགས་མི་བདེ། །མེ་དྲོད་ཞན་པའི་ཕོ་བའི་ནད་ལ་ཕན། །བཱ་ཤ་སེང་ལྡེང་པ་ཊོ་ལ་པ་ཏྲ། །ནིམྦ་ཤིང་ཚ་སླེ་ཏྲེས་ཏ་མ་ལ། །ཀི་ཕྱེ་མར་སར་ལ་བསྐོལ་མྱུར་གྲུབ་ན། །མེ་དབལ་མཁྲིས་བསྐྲན་མཛེ་ལས་བརྒྱལ་བར་འགྱུར། །དྲུག་པས་ཁྲག་དང་བཅས་པ་བརྒྱ་ཡི་སྐྱོན། །རབ་འཁྲུགས་ལས་སྐྱེས་ཡན་ལག་ལུས་ཀུན་གཟེར། །བྱེ་བྲག་མེ་དབལ་ཆང་གིས་ར་རོ་བ། །རྨོངས་པ་ཆང་འཐུངས་ལས་སྐྱེས་རྣམ་འགྱུར་ལ། །ཀཎྜ་ཀ་རི་ནིམྤ་པ་ཏོ་ལ། །སླེ་ཏྲེས་འབྲས་གསུམ་འཇམ་འབྲས་བ་ཤ་ཀ །ཕྱེ་ཆུང་ར་རྙིང་ལ་ནི་ཆོ་གས་གཙོ། །མྱུར་དུ་གྲུབ་འདིས་མཛེ་ལས་མྱུར་དུ་རྒྱལ། །ཀར་ཀུ་ཤྨཎྜ་ཀ་ཆོ་ག་བཞིན་དུ་བཙོས་སྐམས་ནས། །མར་བསྐོལ་ཀར་བཏབ་ཚ་གསུམ་ཟིར་ལྡན་པ་རི་བཙོ། །འདི་བསྟེན་བྱིས་རྒན་ཁྲག་འཛག་རླུང་འཇོམས་བུད་མེད་ལ་ཕོད་ཅིང་། །ཟས་སྐོམ་ལུས་ཁྲག་མཁྲིས་སྐྱུགས་དབུགས་མི་བདེ་དང་གཞང་འབྲུམ་འཇོམས། །བ་གཅིན་སྐྱུ་རུ་མཱ་མཀྵི་ཀ་དྲོ་སྨན། །ཁ་རུ་པོ་སོན་དུ་བ་དནྟྲི་དང་། །རུ་རྟ་རྣམས་བཙོ་རེང་བུ་མཐེ་བོང་འདྲ། །མར་བསྐུས་རྐུབ་བཅུག་དེ་ཡི་ནད་འཕྲོག་བྱེད། །ལུས་གསོ་བའི་དཔྱད་རྫོགས་སོ།། །།ཀར་བཅས་གུར་གུམ་ཆ་མཉམ་མར་ལ་བཙོས། །ཆོ་ག་བཞིན་རླུང་ཁྲག་ལས་བྱུང་བའི། །སྨིན་མ་མུ་འགྲམ་རྣ་མིག་མགོ་བོ་ཡི། །ཕྱེད་གཟེར་ཉི་འཁོར་ལས་བྱུང་ནད་འཇོམས་སོ། །ཨ་རུ་བ་རུ་སྐྱུ་རུ་ནང་ས་བོན། །གསུམ་གཉིས་གཅིག་གི་ཆ་རྣམས་ཞོ་ལ་བཏགས། །མིག་སྨན་དེ་ཡིས་མཆི་མ་མང་འབྱུང་ཞིང་། །མིག་ཚ་རབ་ཏུ་འཁྲུག་པའང་འཇོམས་པར་འགྱུར། །ཨ་རུ་སེནྡྷེ་བ་དང་ལིག་བུ་མིག །བཙག་ཡུག་དང་བཅས་ཆུ་ལ་བསེས་བཏགས་ཏེ། །མིག་གི་ཕྱི་རོལ་རབ་ཏུ་བྱུགས་བྱས་པས། །འདི་ཡིས་མིག་གིས་ནད་ཀུན་ཞི་བར་འགྱུར། །སཽ་བྷི་ལ་སེནྡྷེ་སེང་ཕྲོམ་བཏགས། །དེ་ནས་མར་བཙོས་རས་དཀར་པོ་ལ་བཙགས། །དེ་ཡིས་མིག་ལ་བྱུག་བྱས་མཁྲིས་བསྐྱེད་ནད། །དམར་ཚག་དང་ནི་ཤ་མཛེར་ངེས་པར་འཇོམས། །ཁྲག་བསྐྱེད་ཀ་ར་རྒྱ་མཚོའི་སྤུ་བས་འཇོམས། །བ་བྱུང་གྲོ་སྨན་ནར་མོ་བྲོད་ལོང་ལའོ། །ལུག་མིག་ཚ་བ་གསུམ་གྱིས་མིག་འགྲིབས་ལ། །ལིང་ཏོག་དཀར་ལ་མཱ་མཱཀྵ་སྦྲང་རྩིས་སོ། །ནག་ཚུར་ལིག་མིག་དཀར་བཅས་རྒྱ་མཚོའི་སྦུ། །ནལ་ཤམ་སམ་ཁ་སཻནྡྷ་བཅག་ཡུག་དང་། །ལྡོང་རོས་ཆ་མཉམ་མཱཀྵི་ཀ་བསྲེས་བཏགས། །རབ་རིབ་ཁུས་བསྐྱེད་རྫི་ནད་ཚག་པོ་འཇོམས། །སླེ་ཏྲེས་པུ་ཤེལ་ལྕགས་ཕྱེ་དུག་ཉུང་ནལ་ཤམ་སེནྡྷེ་ཀཌྥ་ལ། །ལིག་མིག་རྡོ་དྲེག་སྤང་སྤོས་ཡུང་བ་སྤང་མ་སང་ཁ་རྒྱ་མཚོའི་ལྤུ། །མཚུངས་པ་ཕྱེ་མ་ལྕགས་སྣོད་དུ་བཞག་སྦྲང་རྩི་ལྡན་པས་ཤ་མཛེར་དང་། །རབ་རིབ་ལིང་ཏོག་ཁྲག་འབུར་མིག་ཤའི་ནད་ལ་ཉེ་བར་བྱུག་པའང་བསྔགས། །ཤིང་མངར་ཨུཏྤལ་རྒྱ་མཚོའི་ལྤུ་བ་ཤིང་ཚ་པུ་ཤེལ་གི་ཧང་བཙོད། །སྤང་པོས་ཙནྡན་སམ་ཁ་བཙག་ཡུག་བ་ལུ་ལིག་མིག་ནག་ཆུར་གདབ། །
དེ་ལྟར་ཐམས་ཅད་ཆ་མཉམ་མིག་སྨན་འདི་ཡིས་མིག་འབྲུམ་མིག་ནག་ཚ། །མཆི་ཁྲག་སྐྱ་རིབ་མིག་ཤ་གཟེར་ནད་འཕྲོག་ལ་བསྔགས་པས་རྟག་ཏུ་བྱུག །མིག་གི་སྨན་དཔྱད་རྫོགས་སོ།། ༎ཐང་ཤིང་བོང་ང་དཀར་པོ་སླེ་ཏྲེས་བཅས། །དུག་ཉུང་ཧོང་ལེན་མ་ནུ་པཱ་ཏྲའི་ཐང་། །བ་གཅིན་ལ་གྲུབ་སྦྲང་རྩི་དང་ལྡན་པ། །མགྲིན་པའི་ནད་དང་ཀུན་ལ་བཏུང་བར་བྱ། །ཀྵར་ཏེ་ཛ་བ་ཏ་པ་ཐཱ་དང་། །ལིག་མིག་སྐྱེར་རྩ་ཡུང་བ་གྲོ་སྨན་ནར། །རིལ་བུ་སྦྲང་རྩི་ལ་བྱས་ཁ་ནང་དུ། །མུར་བར་བྱའོ་མགྲིན་པའི་ནད་ཀུན་ལ། །སྐྱེར་རྩ་གླ་སྒང་ཛཱ་ཏིའི་མེ་ཏོག་དང་གི། །ལོ་གསར་རྒུན་འབྲུམ་དུག་ཉུང་འབྲས་གསུམ་བསྐོལ། །སྦྲང་རྩི་ལྡན་པས་ལུས་ཀྱི་ཁ་སྒོའི་ནད། །ལྕེ་ཡི་ནད་རྣམས་དུས་སུ་ཞི་བར་བྱེད། །རུ་རྟ་དང་ནི་སྐྱེར་པའི་རྩ་བ་སེང་ཕྲོམ་བཅོད། །ཡུང་ཧོང་གླ་སྒང་པཱ་ཐཱ་ཏེ་ཛ་ནི། །ཞིབ་བཏགས་སོ་རྣམས་ལྡེག་པར་བྱེད་པ་དང་། །གཟེར་ཞིང་འཁྱུག་ལ་ནང་ཁྲག་འབྱིན་པར་བྱེད། །སཽ་པྲི་ར་ཅུ་ཀྲ་སྒེའུ་གཤེར་མ་ཏུ། །ཁུ་བ་ཤ་རྣམས་གུ་གུལ་སེནྡྷེ་རྣམས། །གང་ཞིག་ཚ་བའི་མར་ཁུ་ལ་བསྐོལ་ཏེ། །རྣ་བར་བླུགས་པས་རྣ་བའི་ཟུག་རྔུ་སེལ། །བ་ཤ་ནིམྦ་པ་ཊོ་ལ་པརྦ་ཏ་ཀ་པེ་ཙ་བྱ་སྐྱེར་རྩ་དང་། །གླ་སྒང་ཏིག་ཏ་པུ་ཤེལ་བྱི་ཚེར་ཚ་གསུམ་ཏྲ་ཡིནྟ་ཀྟ་ཙནྡན་རྣམས། །ཞུན་མར་ལ་གྲུབ་སྟེང་འོག་ནས་སྐྱེས་རྣམ་འགྱུར་སྣ་དང་མིག་གི་གཟེར་ལ་སོགས། །པགས་ནད་རིམས་དང་ཁོང་འབྲས་རྨ་ཡི་ཟུག་རྔུ་མིག་ནད་སྐྱ་རིབ་རྣམས་ལ་བསྔགས། །སྐྱེར་ཙ་དུག་ཉུང་ནིམྦ་པརྤ་ཊ་དང་རོ་སྨན་བྱི་ཚེར་ཤིང་མངར་དང་། །ཏྲ་ཡནྟཱི་ཧོང་འབྲས་གསུམ་ཏིགྟ་བྲི་ཤ་པ་ཏྲོ་ལ་དང་གླ་སྒང་མངར། །དེ་རྣམས་ཕྱེ་མར་བསྐོལ་བ་ཞུན་མར་ལ་གྲུབ་འདི་ནི་མིག་ལ་ཕན་པ་ཡིན། །མཁྲིས་པ་ཁྲག་ལས་བྱུང་དང་སྐྱ་རིབ་རབ་རིབ་རྣམས་དང་སྟོད་ཀྱི་ནད་རྣམས་ལ། །ཨེ་རཎ་དཀར་པོའི་རྩ་བ་དང་ནི་ཤུ་དག་དང་། །ཤ་ཏ་པུཥྤ་བྲི་ཧ་ཏིའི་འབྲས་བུ་གཞན་ཡང་། །ཏ་ཁར་ཡི་མར་ཁུ་འོ་མར་བཅས་ལས་གྲུབ་། ་རབ་རིབ་སྟོད་དང་གཙོ་ཆེའི་ནད་དང་གཞན་པའང་འཇོམས། །ཟུག་རྔུ་ཟད་པར་བྱེད་པ་གཉིས་པ་རྫོགས་སོ།། །།དུང་དང་མུཤྐ་ཀ་དག་མེ་ལྕེས་བསྲེགས་པའི་ཐལ་བ་སྣོད་དུ་བླུག་པར་བྱས་ཏེ། །དེ་ཉིད་སླར་ཆུ་ལ་བཙོ་ཞིང་བསྐོལ་བའི་ཐལ་བ་མར་ཁུའི་ཆོ་ག་རྣམས། །སྦྱོར་བ་དེ་ཡིས་གདུག་པའི་རྨ་དང་རྨ་ཚ་སྐྲངས་འབུར་ཙ་ཡི་ནད་རྣམས་དང་། །རྣམ་པ་སྣ་ཚོགས་མཚན་བར་བརྡོལ་པའི་པགས་ནད་ཡུན་རིང་མགུལ་པའི་ནད་རྣམས་ལ། །ཤིང་མངར་དང་བཅས་སྐྱེར་རྩ་ཤུག་པ་ཨུཏྤལ་དང་། །པྲེ་ཡངྐུ་དང་སེང་ཕྲོམ་དཀར་པོ་ཙནྡན་རྣམས། །མར་ཁུ་དང་ནི་འོ་མར་གཙོ་ཞིང་སྦྱར་བར་བྱ། །རྨ་རྣམས་དང་ནི་གཟེར་དང་ཚ་བ་འཇོམས་པར་འགྱུར། །ཛཱ་ཏི་ནིམྤ་པ་ཊོ་ལ་ད་པ་ཊ་ཧོང་ལེན་སྐྱེར་རྩ་ཡུང་བ་དང་། །བཙོད་དང་ཐལ་ཏྲེས་ཨ་རུ་པུ་ཤེལ་པྲ་ཚིལ་པངས་མ་ཤིང་མངར་འཇམ་འབྲས་རྣམས། །མར་ལ་གྲུབ་པ་འདི་ཡིས་ཁ་ནད་ཕྲ་མོ་དང་ནི་གནད་གཟེར་ཙ་ཡི་ནད། །ནད་དམ་གཟེར་བཅས་འཕོ་ཞིང་འབྱུང་བའི་རྨ་ཡི་ནད་རྣམས་དང་ནི་འབྱང་བ་འབྱིན། །ཟུག་རྔུ་ཟད་བྱེད་གསུམ་པ་རྫོགས་སོ།། །།ནལ་ཤམ་དཀར་པོ་ཤ་རིའི། །མེ་ཏོག་ཁུ་བས་བརྒྱ་ཕྲག་བསྒོ་བར་བྱ། །སྦྲུལ་གྱིས་སོས་བཏབ་བརྒྱལ་བར་གྱུར་པ་ལ། །
རབ་ཏུ་སྦྱར་བྱ་བཏུང་དང་མིག་སྨན་ལ། །རྨ་བྱའི་མཁྲིས་པ་ཏནྡྱུལ་ཡི་ཀ །ཀཱ་ཀཎྚ་བཅས་མི་ཉུང་འཐུང་བྱས་པས། །འགྲོ་བའི་དུག་དང་བརྟན་མཆོག་དུག་གི་ནི། །འཚེ་བ་དང་བཅས་མྱུར་དུ་འབྱིན་པར་བྱེད། །ཏིལ་བཏགས་མར་ཁུ་ལ་ནི་བུ་རམ་དང་། །དེ་བཞིན་ཨརྐའི་ལོ་མ་ཆ་མཉམ་བཏུང་། །སྤྲིན་གྱི་ཚོགས་ལ་འཕྲལ་དུ་རླུང་བྱུང་ལྟར། །ཁྱི་སྨྱོའི་དུག་མཆོག་མྱུར་དུ་འབྱིན་པར་འགྱུར། །ཁང་པའི་དུ་བ་གུ་གུལ་དང་ལྡན་པ། །བཱ་ཛི་གནྡྷེ་ཏནྡྱུ་ལི་ཡ་བཅས། །བ་ཡི་གཅིན་ལ་བཏགས་བྱས་སྙིང་ནད་དང་། །བརྟན་མཆོག་འགྲོ་བའི་དུག་རྣམས་འཕྲོག་པར་འགྱུར། །སྤང་སྤོས་པུ་ཤེལ་བ་བླ་ཀཽནྟི་ཛ་ལ་སླེ་ཏྲེས་ལྡོང་རོས་གྷ་ཧི་པརྨ་ཀེ་ཤི་དང་། །སྤིགྐ་ཙཎྜ་ཡུང་བ་རུ་རྟ་ཀ་ར་ལ་རྟཱ་རྡོ་དྲག་སཱུཀྵྨེ་གུར་ཀུམ་ཤིང་ཤུག་པ། །ཆ་མཉམ་ཡུངས་ཀར་བརྒྱད་ཆ་ནཱ་ག་གེ་སར་མེ་ཏོག་ཆ་བཞིས་ལྡེ་གུས་ཀུན་ཏུ་ལོངས་སྤྱོད་ཅིང་། །ཕུན་སུམ་ཚོགས་བྱ་ཤ་ཟ་རིམ་དང་དུག་གི་ནད་རྣམས་འཇོམས་འགྱུར་ཟླ་བ་ཤར་བ་ལྟ་བུར་གྲགས། །དུག་ནད་སེལ་བའི་རྒྱུད་རྫོགས་སོ།། །།ཡུངས་དཀར་འབྲས་གསུམ་ཤི་རི་ཥ་ཀཏྲ་བྷཱི་ཤྭེ་འཇམ་འབྲས་སླེ་ཏྲེས་དང་། །བཙོད་དང་སེར་པོ་གཉིས་དང་ཚ་བ་གསུམ་དང་སྤང་སྤོས་ཤུ་དག་ཤིང་ཀུན་རྣམས། །ར་ཡི་གཅིན་ལ་བཏགས་བྱས་འཐུངས་སོགས་ལྟེན་པས་སྦྲུལ་གྱིས་སོས་བཏབ་གདོན་ཀུན་དང་། །གཤེད་བྱེད་སྨྱོ་བྱེད་ཤ་ཟ་དང་ནི་སྲིན་པོ་འཇོམས་ཤིང་བརླག་པར་བྱེད་པར་འགྱུར། །ཀརྤ་སའི་རུས་པ་དང་ནི་རྨ་བྱའི་འདབ་མ་བྲི་ཧ་ཧི་རྣམས་དང་། །ནིརྨ་ལྱ་པོ་སོན་ཆ་ཤིང་ཚ་ཅུ་གང་བྱི་ལའི་བརྟུག་པ་བཱ་ཤ་ཀཱ་དང་ཀེ་ཤ་སྦྲུལ་ལྤགས་དང་། །གླང་པོའི་སོ་དང་ཤྲིང་ག་ཤིང་ཀུན་ནལ་ཤམ་དཀར་པོ་ཆ་མཉམ་བདུག་བྱས་པས། །གཤེད་བྱེད་སྨྱོ་བྱེད་ཤ་ཟ་སྲིན་པོ་ལྷ་རྣམས་ཞུགས་པའི་རིམས་ནི་འཇོམས་པར་དྲན། །ཚ་བ་གསུམ་དང་ད་ལ་གུར་ཀུམ་དྲོ་སྨན་ནར་མོའི་རྩ་བ་ཀྵ་ར་སིང་ཧི་ཡུང་རུ་རྟ་ཉུངས་ཀར་གཉིས་དང་ཨམྦུ་དུག་ཉུང་སྒོག་སྐྱ་འབྲས་བུ་གསུམ་དང་པུ་ཤེལ་རྩེ་དང་ཧོ་ལན་ཤུ་ཏག་སྤངས་མ་ཤིང་མངར་བཱ་ལ་དམར་པོ་ལྡོང་རོས་ཤུག་པ་ཞེས་གྲགས་པ། །ད་རྟི་རྒྱ་སྤོས་མ་དྷུ་ཀ་སཱ་པྲི་ཡངྐུ་དང་བོང་ང་དཀར་པོ་ལིག་བུ་མིག་དང་ཙ་བྱ་ཐང་ཤིང་བཅས་པ་ཕྱེ་མར་བྱས་པ་མར་དང་སྲོག་ཆགས་བརྒྱད་ཀྱི་གཅིན་གྱི་ཆ་ལ་བསྐོལ་བ་བཤད་འཐུངས་པས་འབྱུང་པོ་འབྲོས་ཤིང་དེ་ཡི་གདོན་རྣམས་མཆོག་ཏུ་འཇོམས་པར་འགྱུར། །རྒྱ་སྤོས་ཤིང་མངར་འཇམ་འབྲས་རྒྱ་སྐྱེགས་པ་ཊོ་ལ་ཡི་ལོ་མ་རཀྟ་པ་ཌ་ཀཱ་ཤུ་དག་པ་ཏྲ་ལ་དང་ཤིན་ཀུན་ཡུངས་ཀར་སིཾ་ཧི་སེར་གཉིས་ལ་ཏཱ་ཧོང་ལེན་དང་། །པོ་སོན་ཚ་གསུམ་ཀཱ་ཀཱཎྜ་ཐང་ཤིང་པི་ཏི་ཨ་ང་གྷནྡྷ་མ་རཱ། །ཨརྐོ་ཌ་དང་ཀོཥྚོ་ཀཱི་ན་ཤམ་ནིམྦ་སླེ་ཏྲེས་དུག་མོ་ཉུང་། །རུ་རྟ་ཤུ་ཀ་ཏ་རུ་པུཥྤབཱ་ལ་ཤིང་མངར་ཨ་པ་ར་ཛི་ཏ། །དནྟི་ཙི་ཏྲ་ཀ་པེད་ཆ་མཉམ་ཕྱེ་བ་གཅིན་གྱི་སྡེ་དང་མར་ལ་བསྐོལ། །ཡུན་རིང་དུ་བཞག་ཀུན་གྱི་རིམ་པར་སྦྱོར་བས་མྱུར་དུ་འཇོམས་པར་འགྱུར་བ་ཡིན། །འབྱུང་པོ་ཆེན་པོའི་གདོན་ཀུན་དང་ནི་སྨྱོ་བྱེད་དེས་བྱས་མཛེ་རིམས་ཞི་བར་བཤད། །འབྱུང་པོ་ཞི་བར་བྱེད་པའི་རྒྱུད་ལྔ་པ་རྫོགས་སོ།། །།སྐྱེར་ཡུངས་དུག་མོ་ཉུང་གི་ས་བོན་དང་། །སི་ཧི་དང་ནི་ཤིང་མངར་ཆ་མཉམ་པོ། །དེ་རྣམས་ཐང་བྱས་བུ་ཆུང་མ་ཡི་ནི། །བུ་ནད་དང་ནི་འཁྲུ་བ་ཀུན་ལ་ཕན། །ཀ་པེད་པུཥྤ་དྷ་ཀ་ཛ་ལ། །སེང་ཕྲོམ་དང་བཅས་ག་ཛ་ཡིཔྤ་ལཱི། །ཐང་ལ་མཱཀྵི་ཀ་བསྲེས་བྱུག་པ་དང་། །འཁྲུ་བ་རྣམས་ལ་བྱིས་པ་ལ་སྦྱར་བྱ། །ཤྲི་གཱ་དྲོ་ནར་བོང་ང་དཀར་པོ་ཕྱི་མར་བྱས། །གོ་རིམ་སྐྱུག་པའི་ནད་རྣམས་འཇོམས་པ་ལ། །མཱཀྵི་ཀ་བཅས་བོང་དཀར་དེ་བཞིན་གཅིག །བྱིས་པའི་ནད་སེལ་བའི་རྒྱུད་དྲུག་པ་རྫོགས་སོ།། །།སྐྱུ་རུ་ར་ཕྱེ་དྲུག་ཅུ་རྩ་བཞིའི་སྲང་ལ་རང་གི་ཁུ་བསྒོས་སྦྲང་རྩི་མར་དང་ཆ་མཉམ་ཞིང་། །དྲོད་སྨན་ནར་མོ་སྲང་བརྒྱད་ཀ་ར་ཉིས་འགྱུར་ཆ་ལྡན་ཐལ་བའི་ཕུང་པོའི་ནང་དུ་གཞག་པར་བྱ། །དབྱར་གྱི་མཐའ་རུ་དེ་བརྟེན་སྐྲ་དཀར་དང་བྲལ་གཟུགས་དང་ཁ་དོག་དང་ནི་སྤོབས་པ་དང་ལྡན་ཞིང་། །ནད་མེད་བློ་དང་ཡིད་གཞུངས་དྲན་པ་ཚིག་བསལ་སྟོབས་དང་ལྡན་ཞིང་སེམས་སྟོབས་ལྡན་པར་འགྱུར་བ་ཡིན། །ཤིང་མངར་དང་ནི་སྦྲང་རྩི་མར་དང་ནི། །
འོ་མ་དང་སྦྱར་རྗེས་སུ་བསྟེན་པ་ཡིས། །ལུས་པོ་རྒྱག་པ་རབ་ཏུ་ཐོབ་གྱུར་ནས། །དགའ་ཞིང་འཁྲིག་པ་ཡང་ཡང་ནུས་པར་འགྱུར། །ཤིང་མངར་བཅུ་གང་སེནྡྷེ་དྲོ་སྨན་ནར། །ཀ་ར་དང་ནི་འབྲས་བུ་གསུམ་རབ་ལྡན། །ཚེ་སྦྱིན་རོ་ཙ་ཤིན་ཏུ་ཡིད་བཞུངས་ཤིང་། །རྒས་དང་ནད་རྣམས་རྣམ་པར་འཇོམས་པར་འགྱུར། །གཟེ་མ་སྐྱུ་རུ་སླེ་ཏྲེས་ཕྱེ་མར་བྱས་། །སྦྲང་རྩི་མར་དང་ལྡན་པ་བསྟེན་པ་ཡིས། །རོ་ཙ་བརྟན་ཞིང་སྡུག་བསྔལ་རྣམ་འགྱུར་ཞིང་། །སྐྲ་གནག་ཚེ་ལོ་བརྒྱར་ནི་འཚོ་བར་འགྱུར། །ནིམྦའི་མར་ཁུ་རང་བཞིན་གནས་བྱས་པ། །ཆོ་ག་ཇི་ལྟ་བཞིན་བཏུང་བ་དང་། །འོ་མ་མར་དང་རབ་སྦྱར་དེ་ཟླ་བས། །སྐྲ་དཀར་དང་ནི་རྒས་པ་རྣམ་པར་འབྱིན། །གཟེ་མ་ཀྵུ་ར་དང་རྩ་བ་བཅུ། །བཱ་ན་རཱ་ནཱ་གཱ་བཱ་ལ་བ་ཝ། །ཕྱེ་མ་འོ་ཐུགས་བྱས་འདི་བཏུང་བར་བྱ། །དེ་ཡིས་འཁྲིག་པ་ལན་བརྒྱར་ནུས་པ་ཡིན། །རྟ་ལྟར་བྱེད་པ་བདུན་པ་རྫོགས་སོ།། །།ཤིང་མངར་ཐང་བསྐོལ་སྐོལ་ེནྡྷེ་བ་ལྡན་ཞིང་། །དུག་ཉུང་དྲོ་ནར་པོ་སོན་ཕྱེ་མར་བསྲེ། །སྦྲང་རྩི་བཅས་པ་འདི་དག་སློན་ལ་སྦྱར། །མགྲིན་ནད་ཁ་དང་གྲེ་འགགས་ནད་རྣམས་ལ། །སཽ་བཱི་ར་ལ་ཨ་རུ་ར་ཐང་བསྐོལ། །དྲོ་ནར་ཨ་རུ་དནྟི་ཕྱེ་ཕྱེ་ལྡན། །ཨེ་ར་མར་ཁུ་དང་བཅས་འཁྲུ་ལ་དེ། །བཤལ་གྱི་སྦྱོར་བས་ནད་རྣམས་འཇོམས་པར་འགྱུར། །ར་སྣ་པོ་སོན་ཆ་དང་འབྲས་གསུམ་སླེ་ཏྲེས་རྩ་བ་བཅུ་དང་བཱ་ལཱ་དང་། །མར་ཁུ་སེནྡྷེ་བ་བཅས་པ་ལ་མམ་སི་བཙོས་ལྡན་སྦྲང་རྩི་བུར་མ་བཅས། །ཤ་ཏ་པུཥྤ་སླེ་ཏྲེས་རུ་རྟ་ཀ་བེཏ་བི་ཡཾ་ཀུ་དང་དྲོ་སྨན་ནར། །ཤུ་དག་ཕྱེ་མ་ཀཉྫི་བ་འོ་གཅིན་བཅས་མ་བཏང་རླུང་ནད་རྣམས་ལ་ཕན། །བ་ལ་ཕྱེ་མ་མར་ཁུ་ལ་བསྐོལ་དྲི་བཟང་དང་། །བཅུད་གྲུབ་པ་དང་འོ་མ་ཞོ་དང་སྐྱུ་ཆུ་རྣམས། །དེ་བཞིན་ས་ཧཱ་ཙ་རཱ་སྣཱ་སླེ་ཏྲེས་དང་། །བ་རི་རེ་རེར་བསྐོལ་བས་རླུང་འཇོམས་སོ། །སྦྱར་བར་བྱའོ་བཅའ་སྒའམ་བུ་རམ་ལ། །ཡང་ན་སེནྡྷེ་བ་བཅས་དྲོ་ནར་ལའོ། །སྣ་ཁ་འཇིམ་པ་འགྲམ་པ་དཔུང་པ་དང་། །རྒྱབ་དང་མགོ་མིག་མགུལ་འགགས་ནད་རྣམས་ལ། །ལས་ལྔའི་ལེའུ་བརྒྱད་པ་རྫོགས་སོ།། །།རིམས་ལ་གཙོ་ཆེར་པརྤ་པ་ཊ་ཀ་སྦྱར་བྱ་སྐོམ་དད་ཆེ་ལ་སྐྱུ་རུ་ར་དང་སྐྱུག་པ་ལ། །ཚ་བ་ལྔ་སྐྱེས་གཟེར་ལ་འཇམ་རྩི་རེངས་ལ་གི་རི་ཛ་དང་གཅིན་སྙི་སྐྱུ་རུ་ར་ཡི་ཁུ་བས་སོ། །རླུང་གི་ནད་ལ་རཱ་སྣཱ་དམུ་རྫིང་མཆེར་པའི་ནད་ལ་བ་གཅིན་ཆམ་པ་ལ། །དྲོ་ནར་འཕྱོ་ལ་བི་ཏཾ་དུག་ལ་ཤུ་ཀ་རྟ་རུ་སྐྲངས་ལ་གུ་གུལ་ལོ། །བ་ཤ་ཀ་ཡིས་ཁྲག་མཁྲིས་དུག་ཉུང་གིས། །འཁྲུ་ལ་འོ་བསྲེ་ཤིང་ཧེ་མི་ལྕི་གཟེར་ལ། །སྦོས་ལ་ཐང་ཤིང་སྲིན་ནད་བྱི་ཏཾ་ཀས། །སྐྱ་རྦབ་ཨ་རུ་ར་དང་འབྲུ་གཞན་གྱིས། །ཟད་བྱེད་ནད་ལ་འབྲས་བུ་གསུམ་བཤད་གཟེར་བ་ཀུན་ལ་དེ་ལྟར་ཁུ་བས་སོ། །མཛེ་ལ་རྟག་ཏུ་སེང་ལྡེང་ཁུ་བ་གློ་ལ་ཀཎྜ་ཀ་རི་དང་། །
ཚིགས་ནད་གཟེ་མ་མུ་ར་གཤེད་བྱེད་ནཱ་ག་བཱ་ལཱ་བ་ལས་སོ། །བཅུད་ལེན་ཀུན་ལ་སྐྱུ་རུ་ལྷ་བཅས་གདོན་ནད་ལ་བྲག་ཞུན་བཤད། །རྒྱུད་ཕྱི་མའོ།། །།རབ་ཏུ་གྲུབ་པའི་སྦྱོར་བའི་ཆོ་ག་རྫོགས་པར་བྱས། །གྲུབ་པའི་དོན་རྣམས་རྟོགས་པའི་སྨན་པ་མཁས་རྣམས་ཀྱིས། །རྟོག་པའི་སྨན་པ་མེད་པར་སུ་ཡིས་སྦྱོར་བ་མཐོང་། །འདི་དོན་སླར་ཡང་འདིར་ནི་ཅུང་ཟད་བཤད་པར་བྱ། །ཤུགས་བཀག་མ་ཟོས་གཉིད་ཆག་སྐད་ཆེས་རྩོལ་བ་དྲགས། །གཞོན་པ་ལ་ཞོན་ཚབ་ཁ་བསྐ་བ་རྩུབ་པ་རྣམས། །སེམས་བཟུང་འཁྲིག་དང་འཇིགས་དང་བསྙུང་གནས་མྱ་ངན་དྲངས། །དབྱར་གྱི་དུས་སུ་རླུང་ནི་རབ་ཏུ་འཁྲུག་པར་འགྱུར། །ཚ་སྐྱུར་ཆང་དང་ལན་ཚྭ་རྣོ་དྲོ་ཏིལ་མར་ཁུ། །ཁྲོ་དང་གདུང་བ་ལོ་མ་སྐམ་པོ་འཇིགས་ངལ་རྣམས། །ཐལ་སྨན་ལ་སོགས་མ་ཞུ་མ་འཕྲོད་ཟས་ཟོས་པ། །སྟོན་གྱི་དུས་སུ་མཁྲིས་པ་རབ་ཏུ་འཁྲུགས་པར་འགྱུར། །ཉིན་གཉིད་མངར་དང་གྲངས་ཆེས་ཉ་ཤ་ལྕི་བ་དང་། །ཤིང་ཏོག་འོ་མ་བུ་རམ་ཤིང་གི་རྣམ་འགྱུར་རྣམས། །སྣུམ་བག་འགྲངས་ཆེས་ཟོས་དང་ལན་ཚྭ་བུར་ཆུ་རྣམས། །དཔྱིད་ཀྱི་དུས་སུ་བད་ཀན་རབ་ཏུ་འཁྲུགས་པར་འགྱུར། །དེ་ལྟར་འདི་རྣམས་རིམ་པས་བྱེ་བྲག་སྐྱོན། །བདུག་པ་ཀུན་ནི་ལྡན་པ་བརྒྱད་དུ་ཤེས། །སྣ་ཚོགས་ནད་ནི་ལུས་ཀྱི་གྲངས་བཞིན་དུ། །མིང་རྣམས་མ་ལུས་རྗེས་སུ་རྟོགས་པར་བྱ། །རྩུབ་འཁུམས་འཕོ་ཞིང་གཟེར་དང་སྔོ་བསངས་མདོག །ཡན་ལག་ཟུག་ཆེ་ཡན་ལག་བརྐྱང་བསྐུམ་བྱེད། །ཤེས་རྨོང་གྲང་ཞིང་གྱོང་ལ་སྐོམ་པ་རྣམས། །རླུང་གི་ལས་སུ་མཁས་པས་རབ་ཏུ་བསྟན། །འཛག་ཅིང་རྡུལ་འབྱུང་ཚ་ཞིང་འདོད་ཆགས་སྐྱེ། །དྲི་ང་རྣག་དང་འཇུ་དང་ཁྲོ་བ་དང་། །བསླབ་བཅོལ་བརྒྱལ་དང་མགོ་འཁོར་ལུས་མདོག་སེར། །མཁྲིས་པའི་ལས་སུ་མཁས་པས་རབ་ཏུ་བསྟན། །ཁ་མིག་སྣ་ཆུ་གྲང་ལྕི་ལུས་གཡའ་རྡུལ། །ཤེས་པ་རྨོངས་ཤིང་སྣ་ཚོགས་འགགས་པ་དང་། །ཚོ་ཆེས་མཆིལ་སྣམས་ཡུན་རིང་འབྱུང་བྱེད་རྣམས། །བད་ཀན་ལས་སུ་མཁས་པས་རབ་ཏུ་བསྟན། །དེ་རྣམས་ཀྱིས་བྱས་འདི་དག་རྟོགས་རྣམས་ནི། །ནད་ཀུན་མིང་ནི་དུ་མའི་རྣམ་པ་ཡང་། །བྱེ་བྲག་ཁ་ཅིག་སྐྱེ་བའི་དབང་གིས་འགྱུར། །གང་གི་འདི་རྣམས་མིང་གཞན་ཀུན་དུ་འགྲོ། །རྒྱུད་ནི་ལེ་ལོ་ཡིད་ནི་རྣམ་མ་དག །སྐྱོན་ནི་རབ་འཇུག་བཀྲེས་དང་གཅིན་ཡང་འབྱུང་། །གསུས་ལྕི་ཤེས་པ་རྨོངས་ཤིང་དང་ག་མེད། །དེ་རྣམས་ནད་ལྡན་དེ་ལ་ནད་འབྱུང་འགྱུར། །བཅུད་དང་དྲོད་དང་བརྟན་དང་རོ་ཙ་སྟོབས་བསྐྱེད་ལན་ཚ་མངར་སྐྱུར་མར་ཁུ་དང་། །ཁྲུས་དང་དྲོད་བསྟེན་མས་བཏང་ཤེ་དང་བསྟེན་ཡན་ལག་ཀུན་ལ་དྲིལ་ཕྱིས་དང་། །སྣུམ་གྱི་དུགས་དང་འཇམ་རྩིས་ཞི་བྱས་བསྐུ་བྱུག་ཉེ་བར་བསྟེན་པ་ལ་སོགས་པ། །བཟའ་དང་བཏུང་བ་གནས་དང་སྨན་འདི་རྣམས་ཀྱིས་རླུང་ནི་རབ་ཏུ་ཞི་བར་འགྱུར། །ཁ་མངར་བསྐ་བ་གྲང་དང་རླུང་ཡབ་མཚན་མོ་ཟླ་བའི་གཟེར་ལ་གནས་པ་དང་། །ས་ཡི་ཁང་པ་ཆུ་ཡི་འཁྲུལ་འཁོར་ཆུ་ལ་བརྩེ་ཞིང་བུད་མེད་ལུས་ལ་རེག །མར་སར་ཀྵི་ར་དང་པོར་ལུས་ལ་བཀྲུ་སྨན་དང་ནི་ཁྲག་གི་གཏར་སྤྱད་བྱ། །བཟའ་དང་བཏུང་བ་གནས་དང་སྨན་འདི་རྣམས་ཀྱིས་མཁྲིས་པ་རབ་ཏུ་ཞི་བར་འགྱུར། །རྩུབ་དང་ཀྵ་ར་བསྐ་ཁ་ཚ་བ་དང་ནི་འབད་རྩོལ་དྲག་དང་ངར་སྣབ་སྦྱར། །བུད་མེད་བསྟེན་ཞིང་ལ་དུབ་གཉིད་བཅག་ཆུ་ལ་རྩེ་ཞིང་རྐང་པ་འཆག་པར་བྱ། །རེང་བུའི་དུད་བསྟེན་མགོ་བསྲོ་ཁྲུས་དང་སྐྱུག་སྨན་དུག་དང་མས་བཏང་ལ་སོགས་པ། །བཟའ་དང་བཏུང་བ་གནས་དང་སྨན་འདི་རྣམས་ཀྱིས་བད་ཀན་ཉེ་བར་ཞི་བར་འགྱུར། །
འཕྲོད་ཟས་བསྙུང་དང་ཚ་བའི་ཆུ་བཏུང་ཞིང་། །ཟས་དང་བྱ་ན་ཡང་ཞིང་ཁ་བའི་བཅུད། །རྩུབ་བྲལ་དུགས་དང་འཇུ་བ་བསྟེན་པ་དང་། །དེ་བཞིན་ལུས་ཀྱི་སྟེང་འོག་སྦྱང་བར་བྱ། །རྒུན་སྟོད་དབྱར་དང་རྒུན་སྨད་རླུང་གི་དུས། །མཁྲིས་པ་སྟོན་དང་སོ་ཁ་ཚ་བའི་དུས། །བད་ཀན་དཔྱིད་ཀ་མེ་ཏོག་འོང་དུས་འཁྲུག། །དེ་ལ་དེ་བཞིན་ཕན་པའི་སྨན་བསྟན་བྱ། །བད་ཀན་འཁྲུགས་ལ་སྐྱུག་སྨན་སྦྱར་བར་བྱ། །མཁྲིས་པའི་རྣམ་འགྱུར་བྱུང་ལ་བཀྲུས་སྨན་བཏང་། །རླུང་གི་བདག་ཉིད་འཇམ་རྩིས་འཇོམས་པར་འགྱུར། །འདུ་འཕྲོད་སྐྱེས་ལའང་འདྲེས་མ་དེ་དག་གོ། །རླུང་ལ་སྐྱུར་དང་མར་ཁུ་ལན་ཚྭ་དང་། །མངར་བའི་བཟའ་དང་བཏུང་བ་རྣམས་ཀྱིས་སོ། །མཁྲིས་པ་ལ་ནི་དེ་བཞིན་མངར་བ་དང་། །ཁ་དང་བསྐ་བ་གྲང་བ་རྣམས་ཀྱིས་སོ། །བད་ཀན་ལ་ཡང་ཁ་དང་ཚ་བ་དང་། །རྩུབ་དང་བསྐ་བ་རྣོ་བའི་རོ་རྣམས་ཀྱིས། །དེ་རྣམས་བཟའ་བས་ཉེ་བར་འགྲོ་འགྱུར་བ། །ནད་ནི་རབ་ཏུ་ཞི་བར་འགྱུར་པ་ཡིན། །བློ་གཞན་ཉིད་ཀྱིས་གང་དེ་མ་བསྟན་གང་ཞིག་བསྟན། །ནད་རྣམས་རང་གིས་རྟགས་ནས་ཇི་ལྟའི་ཚུལ། །སྨན་རྣམས་རྗེས་སུ་བློ་གྲོས་ཆུང་བ་རྣམས་ལ་བྱས། །བསྟན་བཅོས་སུ་ནི་ཅུང་ཟད་བཏན་ཕབ་འབྲས་བུར་བྱས། །ཡོན་ཏན་ལྷག་ཅན་སྦྱོར་བ་བརྒྱ་པ་བརྗོད་པ་ཡིས། །བསོད་ནམས་བླ་ན་མེད་པ་བདག་གིས་གང་ཐོབ་པ། །དེ་ཡིས་རྣམ་པ་སྣ་ཚོགས་ནད་ཀྱིས་གཟིར་གྱུར་པ། །མཐའ་ཡས་འགྲོ་བ་རྣམས་ཀྱི་ནད་ནི་མེད་གྱུར་ཅིག །སྦྱོར་བ་བརྒྱ་པ་ཞེས་བྱ་བའི་སྨན་གྱི་གཞུང་འཕགས་པ་ཀླུ་སྒྲུབ་ཞབས་ཀྱིས་མཛད་པ་རྫོགས་སོ།། །།བལ་པོའི་པཎྜི་ཏ་ཆེན་པོ་བྲམ་ཟེ་ཛེ་ཏ་ཀརྞཱའི་ཞལ་སྔ་ནས་དང་པོ་རྩ་འགྲེལ་ལན་ཅིག་གསུངས་ཤིང་། །དུས་ཕྱིས་རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་བུདྡྷ་ཤྲཱི་ཛྙཱ་ནའི་ཞལ་སྔ་ནས་སྟོད་ཅུང་ཟད་གསུངས་ནས། གྲུབ་པའི་ཞིང་སྐྱི་རོང་འཕགས་པ་འཇིག་རྟེན་དབང་པོའི་གཙུག་ལག་ཁང་ལ་ཅུང་ཟད་ལས་མི་རིང་བ། ཆེན་པོའི་ཆེན་པོ་རྣམས་ཀྱི་གདན་ས་དཔལ་ཤེར་གྱི་གཙུག་ལག་ཁང་དུ་སྡོམ་བརྩོན་དམ་པ་སྡེ་སྣོད་གསུམ་དང་། རྒྱུད་དང་རྒྱུད་ཀྱི་ཡན་ལག་རྣམས་ལས་ལེགས་པར་འདྲིས་ཤིང་། ཁྱད་པར་དུ་ཚེའི་རིག་བྱེད་ཡན་ལག་བརྒྱད་པའི་སྙིང་པོ་བསྡུས་པའི་གཞུང་དོན་ལེགས་པར་རྟོགས་པ་དགེ་སློང་ཨ་མོ་གྷའི་གསུང་གིས་བསྐུལ་ནས། མང་དུ་ཐོས་པའི་ལོ་ཙྪ་བ་ཤཱཀྱའི་དགེ་སློང་དཔལ་ཉི་མ་རྒྱལ་མཚན་བཟང་པོས་འགྲེལ་པ་གསུམ་དང་བསྟུན་ཏེ་བསྒྱུར་ཞིང་ལེགས་པར་གཏན་ལ་ཕབ་པའོ༎ །།གསོ་དཔྱད་གཞུང་ལུགས་བརྒྱ་ཕྲག་དུ་མའི་གནད། །སྣ་ཚོགས་མངོན་བརྗོད་ཐ་དད་སྡེབ་སྦྱོར་ཅན། །ཡ་མཚན་རྒྱན་ལ་ལེགས་སྦྱར་ཚིག་གིས་སྤེལ། །གང་འདི་བསྒྱུར་དགེས་ནད་རྣམས་ཞི་བར་ཤོག། ས་སྟེང་དུ་ཉི་མ་དང་ཟླ་བ་ལྟར་གྱུར་ཅིག། །།ལོ་ཙཱ་བའི་འགྱུར་བསྟན་ལ་དཔེ་ཕྱི་བྱས་ནས་བྲིས་ཏེ། ཞུས་དག་ལེགས་པར་བགྱིས་པ་འདིས།སེམས་ཅན་དཔག་ཏུ་མེད་པའི་དོན་བྱེད་པར་གྱུར་ཅིག། ༎
